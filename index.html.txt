  var pct=Math.min(Math.round(Math.abs(val)/Math.abs(tgt)*100),100);
  var ok=lower?val<=tgt:val>=tgt;
  var tc=ok?'ok':(pct>60?'warn':'bad'), bc=ok?'bg':(pct>60?'ba':'br');
  return'<div class="prow"><div class="phdr"><span class="plbl">'+lbl+'</span><span class="pnum '+tc+'">'+(isPct?fmtNum(val)+'%':fmtNum(val))+'</span></div><div class="pbg"><div class="pb '+bc+'" style="width:'+pct+'%"></div></div></div>';
}

function parse5M(text){
  var r={MO:'',MAT:'',METH:'',MIL:'',MGMT:''};
  if(!text) return r;
  var lines=text.split('\n'),curKey=null,cur=[];
  var map=[{keys:["main d'oeuvre","main-d'oeuvre","main d'œuvre","main :"],k:'MO'},
    {keys:['matière','matéri','matériel'],k:'MAT'},{keys:['méthode','methode'],k:'METH'},
    {keys:['milieu','environnement'],k:'MIL'},{keys:['management','machine','mesure'],k:'MGMT'}];
  lines.forEach(function(l){
    var lo=l.toLowerCase().replace(/[*#\\]/g,'').trim(),found=null;
    for(var i=0;i<map.length;i++){for(var j=0;j<map[i].keys.length;j++){if(lo.indexOf(map[i].keys[j])>-1){found=map[i].k;break;}}if(found)break;}
    if(found){if(curKey)r[curKey]=cur.join(' ').trim();curKey=found;cur=[l.replace(/^[-•*#\\: ]+/,'').trim()];}
    else if(curKey&&l.trim()) cur.push(l.replace(/^[-•*#\\ ]+/,'').trim());
  });
  if(curKey) r[curKey]=cur.join(' ').trim();
  return r;
}

function applyIncF(){
  if(!D) return;
  var p=document.getElementById('incPer').value;
  var mo=parseInt(document.getElementById('incMo').value)||4;
  var yr=parseInt(document.getElementById('incYr').value)||2026;
  var d1=document.getElementById('incD1').value,d2=document.getElementById('incD2').value;
  var filtered=(D.incidents||[]).filter(function(inc){
    if(!inc.id||inc._err) return false;
    if(!inc.victime&&!inc.description) return false;
    if(p==='all') return true;
    var iso=inc.isoDate; if(!iso) return p==='ytd';
    if(p==='ytd'){ var y=parseInt(iso.substring(0,4)); return y===2026||y===2025; }
    if(p==='month'){ var pts=iso.split('-'); return parseInt(pts[0])===yr&&parseInt(pts[1])===mo; }
    if(p==='custom'){ var ok=true; if(d1)ok=ok&&iso>=d1; if(d2)ok=ok&&iso<=d2; return ok; }
    return true;
  });
  renderIncidents(filtered);
}

function applyTagF(){
  if(!D) return;
  clearTagDirty();
  var p=document.getElementById('tagPer').value;
  var yr=parseInt(document.getElementById('tagYr').value)||2026;
  var mo=parseInt(document.getElementById('tagMo').value)||1;
  var d1=document.getElementById('tagD1').value,d2=document.getElementById('tagD2').value;
  var searchVal=(document.getElementById('tagSearch').value||'').toLowerCase().trim();
  var sortVal=document.getElementById('tagSort').value;
  var filterResp=document.getElementById('tagFilterResp').value;
  var filterZone=document.getElementById('tagFilterZone').value;

  var filtered=(D.tags||[]).filter(function(t){
    if(!t.id||t._err) return false;
    var isOpen=t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;
    if(tagStatFilter==='open'&&!isOpen) return false;
    if(tagStatFilter==='closed'&&isOpen) return false;
    if(filterResp!=='all'&&(t.responsable||'').trim()!==filterResp) return false;
    if(filterZone!=='all'){var z=(t.zone||t.emplacement||'').trim();if(z!==filterZone)return false;}
    if(p!=='all'){
      if(!t.isoDate) return p==='ytd';
      var pts=t.isoDate.split('-');
      if(p==='ytd'&&parseInt(pts[0])!==2026) return false;
      if(p==='month'&&(parseInt(pts[0])!==yr||parseInt(pts[1])!==mo)) return false;
      if(p==='custom'){if(d1&&t.isoDate<d1)return false;if(d2&&t.isoDate>d2)return false;}
    }
    if(searchVal){
      // Recherche dans tous les champs, SAUF "tag" et "tags" ignorés
      var hay=(t.description+' '+(t.zone||t.emplacement||'')+' '+t.type+' '
        +t.responsable+' '+t.id+' '+t.risque+' '
        // photoRef sans "tag" ni "tags"
        +(t.photoRef||'').replace(/\btags?\b/gi,'')).toLowerCase();
      if(hay.indexOf(searchVal)===-1) return false;
    }
    return true;
  });

  // Tri
  filtered.sort(function(a,b){
    if(sortVal==='date_desc'){if(!a.isoDate)return 1;if(!b.isoDate)return -1;return b.isoDate.localeCompare(a.isoDate);}
    if(sortVal==='date_asc'){ if(!a.isoDate)return 1;if(!b.isoDate)return -1;return a.isoDate.localeCompare(b.isoDate);}
    if(sortVal==='id_asc')  return String(a.id).localeCompare(String(b.id),undefined,{numeric:true});
    if(sortVal==='id_desc') return String(b.id).localeCompare(String(a.id),undefined,{numeric:true});
    if(sortVal==='zone_az') return (a.zone||a.emplacement||'').localeCompare(b.zone||b.emplacement||'');
    if(sortVal==='resp_az') return (a.responsable||'').localeCompare(b.responsable||'');
    return 0;
  });

  var total=D.tags.length;
  var openAll=D.tagsOpen.length,closedAll=total-openAll;
  var openF=filtered.filter(function(t){return t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;}).length;
  var closedF=filtered.length-openF;
  var cloPct=filtered.length>0?Math.round(closedF/filtered.length*100):0;

  document.getElementById('tagStatsRow').innerHTML=
    '<span class="tag-stat-pill tsp-total">Total : '+total+'</span>'
    +'<span class="tag-stat-pill tsp-open">Ouverts : '+openAll+'</span>'
    +'<span class="tag-stat-pill tsp-closed">Fermés : '+closedAll+'</span>'
    +(filtered.length!==total?'<span class="tag-stat-pill tsp-filtered">Filtrés : <strong>'+filtered.length+'</strong></span>':'');
  document.getElementById('tagListCount').textContent=filtered.length+' résultats';

  document.getElementById('tagKpi').innerHTML=[
    kpiCard('Filtrés',filtered.length,'b','TAGs','k-b',null,null),
    kpiCard('Fermés',closedF,'g','✓','k-g',cloPct,C.green),
    kpiCard('Ouverts',openF,'r','⚠','k-r',filtered.length>0?Math.round(openF/filtered.length*100):0,C.red),
    kpiCard('Clôture',fmtNum(cloPct)+'%','g','obj.95%','k-g',cloPct,cloPct>=95?C.green:cloPct>50?C.amber:C.red)
  ].join('');

  // Types de dangers
  var types={};
  filtered.forEach(function(t){if(t.type){var ty=t.type.substring(0,22);types[ty]=(types[ty]||0)+1;}});
  var tk=Object.keys(types).sort(function(a,b){return types[b]-types[a];}).slice(0,6);
  if(charts['dangerC'])charts['dangerC'].destroy();
  if(tk.length>0) new Chart(document.getElementById('dangerC'),{type:'doughnut',
    data:{labels:tk,datasets:[{data:tk.map(function(k){return types[k];}),backgroundColor:[C.amber,C.red,C.p2,C.teal,C.green,C.orange],borderWidth:0,hoverOffset:4}]},
    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:true,position:'right',labels:{color:C.txt,font:{size:9},boxWidth:8}}},cutout:'50%'}});

  // Zones TAGs
  var tzones={};
  filtered.forEach(function(t){if(t.zone||t.emplacement){var z=(t.zone||t.emplacement).substring(0,18);tzones[z]=(tzones[z]||0)+1;}});
  var tzk=Object.keys(tzones).sort(function(a,b){return tzones[b]-tzones[a];}).slice(0,6);
  mkChart('tagZoneC','bar',tzk,[{data:tzk.map(function(k){return tzones[k];}),backgroundColor:'rgba(9,99,126,.75)',borderRadius:4}],
    Object.assign({},BASE,{indexAxis:'y',scales:{x:{grid:{color:C.grid},ticks:{color:C.txt}},y:{grid:{display:false},ticks:{color:C.txt,font:{size:9}}}}}));

  // Table avec vignettes Avant / Après inline (chargement asynchrone)
  var tb=document.getElementById('tagBody'); tb.innerHTML='';
  if(filtered.length===0){
    var msg=searchVal?'Aucun TAG pour "'+esc(searchVal)+'"'
      :filterResp!=='all'?'Aucun TAG pour "'+esc(filterResp)+'"'
      :filterZone!=='all'?'Aucun TAG zone "'+esc(filterZone)+'"'
      :'Aucun TAG pour ce filtre';
    tb.innerHTML='<tr><td colspan="10" style="text-align:center;color:var(--text3);padding:20px">'+msg+'</td></tr>';
    return;
  }
  function hl(str){
    if(!searchVal||!str) return esc(str||'');
    var re=new RegExp('('+searchVal.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');
    return esc(str).replace(re,'<mark style="background:rgba(245,166,35,.35);border-radius:2px;padding:0 1px">$1</mark>');
  }
  var pendingTagThumbs=[];
  filtered.forEach(function(t,idx){
    var isOpen=t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;
    var respHtml=t.responsable
      ?'<span class="rbadge" style="color:var(--p4)">'+hl(t.responsable)+'</span>'
      :'<span style="color:var(--text3);font-size:9px">—</span>';

    // ── Vignettes Avant / Après — chargement asynchrone via google.script.run
    function tagThumbCell(label, has, fid, ref){
      var cellId='tagThumb_'+idx+'_'+label;
      if(!has||!fid){
        return '<td class="cln-thumb-cell"><div class="cln-thumb-ph" title="Pas de photo '+label+'">📷</div></td>';
      }
      pendingTagThumbs.push({fid:fid,cellId:cellId,label:label,ref:ref||'',tag:t});
      return '<td class="cln-thumb-cell"><div id="'+cellId+'" class="cln-thumb-load" title="Chargement photo '+label+' depuis Drive…"></div></td>';
    }
    // L'ordre des colonnes Avant puis Après est garanti côté HTML
    var avantCellHtml=tagThumbCell('Avant', t.hasPhotoAvant, t.driveFileIdAvant, t.photoClean||t.photoRef);
    var apresCellHtml=tagThumbCell('Après', t.hasPhotoApres, t.driveFileIdApres, t.photoCleanApres||t.photoRefApres||t.photoClean||t.photoRef);

    var tr=document.createElement('tr');
    tr.style.cursor='pointer';
    tr.innerHTML='<td class="mono" style="color:var(--p4)">'+esc(t.id)+'</td>'
      +'<td style="white-space:nowrap;font-size:10px">'+t.date+'</td>'
      +avantCellHtml+apresCellHtml
      +'<td style="font-size:10px;max-width:180px">'+hl(t.description.substring(0,80))+(t.description.length>80?'…':'')+'</td>'
      +'<td><span class="rbadge">'+hl((t.zone||t.emplacement)||'—')+'</span></td>'
      +'<td style="font-size:10px">'+hl(t.type)+'</td>'
      +'<td style="white-space:nowrap;font-size:10px">'+esc(t.dateCible||'—')+'</td>'
      +'<td style="font-size:10px">'+respHtml+'</td>'
      +'<td><span class="'+(isOpen?'s-open':'s-close')+'">'+(isOpen?'Ouvert':'Fermé')+'</span></td>';
    tr.addEventListener('click',function(){ openTagModal(t); });
    tb.appendChild(tr);
  });

  // ── Chargement asynchrone des vignettes (avec cache partagé) ──
  if(!window._tagPhotoCache) window._tagPhotoCache={};
  pendingTagThumbs.forEach(function(p){
    var slot=document.getElementById(p.cellId);
    if(!slot) return;
    var cached=window._tagPhotoCache[p.fid];
    if(cached){ replaceTagThumb(slot,cached,p); return; }
    google.script.run
      .withSuccessHandler(function(d){
        if(d){ window._tagPhotoCache[p.fid]=d; replaceTagThumb(slot,d,p); }
        else if(slot && slot.parentNode){ slot.outerHTML='<div class="cln-thumb-ph" title="Lecture impossible">⚠️</div>'; }
      })
      .withFailureHandler(function(){
        if(slot && slot.parentNode){ slot.outerHTML='<div class="cln-thumb-ph" title="Lecture impossible">⚠️</div>'; }
      })
      .getTagPhotoForClient(p.fid);
  });
}

function replaceTagThumb(slot, dataUrl, p){
  if(!slot||!slot.parentNode) return;
  var img=document.createElement('img');
  img.src=dataUrl;
  img.className='cln-thumb';
  img.alt=p.label;
  img.title=p.label+(p.ref?' · '+p.ref:'')+' (cliquer pour agrandir)';
  img.addEventListener('click',function(e){
    e.stopPropagation();
    openTagModal(p.tag);
    var photoTab=document.querySelectorAll('#tagModal .modal-tab')[1];
    if(photoTab) switchTab('mTag','photo',photoTab);
  });
  slot.parentNode.replaceChild(img,slot);
}

// État dérivé des photos :
//   - photo « Avant » seule           → en cours
//   - photos « Avant » ET « Après »   → terminé
//   - aucune photo                    → indéfini (ne compte pas)
function clnHasAvant(c){ return !!(c && c.hasPhotoAvant && c.driveFileIdAvant); }
function clnHasApres(c){ return !!(c && c.hasPhotoApres && c.driveFileIdApres); }
function isClnOpen(c){
  return clnHasAvant(c) && !clnHasApres(c);
}
function isClnClosed(c){
  return clnHasAvant(c) && clnHasApres(c);
}

function populateClnDropdowns(){

function applyClnF(){
  if (!D || !D.cleaning) return;
  var search    = (document.getElementById('clnSearch').value || '').trim().toLowerCase();
  var sort      = document.getElementById('clnSort').value;
  var fZone     = document.getElementById('clnFilterZone').value;

  var filtered = D.cleaning.filter(function(c){
    if (clnStatFilter === 'open'   && !isClnOpen(c)) return false;
    if (clnStatFilter === 'closed' && !isClnClosed(c)) return false;
    var empl = (c.emplacement || c.zone || '').trim();
    if (fZone !== 'all' && empl !== fZone) return false;
    if (search) {
      var hay = [c.id, empl].join(' ').toLowerCase();
      if (hay.indexOf(search) < 0) return false;
    }
    return true;
  });

  filtered.sort(function(a,b){
    if (sort === 'date_desc') return (b.isoDate||'').localeCompare(a.isoDate||'') || 0;
    if (sort === 'date_asc')  return (a.isoDate||'').localeCompare(b.isoDate||'') || 0;
    if (sort === 'id_asc' || sort === 'id_desc') {
      var na = parseInt(String(a.id).match(/\d+/),10), nb = parseInt(String(b.id).match(/\d+/),10);
      var diff = (isNaN(na)||isNaN(nb)) ? String(a.id).localeCompare(String(b.id)) : (na - nb);
      return sort === 'id_asc' ? diff : -diff;
    }
    if (sort === 'empl_az' || sort === 'zone_az') {
      return (a.emplacement||a.zone||'').localeCompare(b.emplacement||b.zone||'');
    }
    return 0;
  });

  // Stats pills (état dérivé des photos)
  var total  = D.cleaning.length;
  var open   = D.cleaning.filter(isClnOpen).length;
  var closed = D.cleaning.filter(isClnClosed).length;
  var statsRow = document.getElementById('clnStatsRow');
  if (statsRow) {
    statsRow.innerHTML =
      '<span class="tag-stat-pill tsp-total">'+total+' total</span>' +
      '<span class="tag-stat-pill tsp-open">'+open+' en cours</span>' +
      '<span class="tag-stat-pill tsp-closed">'+closed+' terminés</span>' +
      (filtered.length !== total ? '<span class="tag-stat-pill tsp-filtered">'+filtered.length+' filtrés</span>' : '');
  }
  document.getElementById('clnListCount').textContent = filtered.length + ' / ' + total;

  renderClnTable(filtered, search);
}
function renderClnTable(items, searchVal){
  var tb = document.getElementById('clnBody');
  if (!tb) return;
  if (items.length === 0) {
    tb.innerHTML = '<tr><td colspan="6" style="text-align:center;color:var(--text3);padding:20px">Aucune entrée pour ce filtre</td></tr>';
    return;
  }
  function hl(str){
    if (!searchVal || !str) return esc(str||'');
    var re = new RegExp('('+searchVal.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');
    return esc(str).replace(re,'<mark style="background:rgba(245,166,35,.35);border-radius:2px;padding:0 1px">$1</mark>');
  }
  tb.innerHTML = '';
  var pendingThumbs = [];   // { fid, imgId } → chargement asynchrone groupé
  items.forEach(function(c, idx){
    var hasAvant = clnHasAvant(c);
    var hasApres = clnHasApres(c);
    var closed   = hasAvant && hasApres;
    var open     = hasAvant && !hasApres;
    var statClass = closed ? 's-close' : (open ? 's-open' : '');
    var statTxt   = closed ? 'Terminé' : (open ? 'En cours' : '—');

    var emplTxt = c.emplacement || c.zone || '—';

    // ── Cellules vignettes Avant / Après ──────────────────────
    function thumbCell(label, has, fid, ref){
      var cellId = 'clnThumb_' + idx + '_' + label;
      if (!has || !fid) {
        return '<td class="cln-thumb-cell"><div class="cln-thumb-ph" title="Pas de photo '+label+'">📷</div></td>';
      }
      pendingThumbs.push({ fid: fid, cellId: cellId, label: label, ref: ref || '', cln: c });
      return '<td class="cln-thumb-cell"><div id="'+cellId+'" class="cln-thumb-load" title="Chargement photo '+label+' depuis Drive…"></div></td>';
    }
    var avantCellHtml = thumbCell('Avant', c.hasPhotoAvant, c.driveFileIdAvant, c.photoRefAvant);
    var apresCellHtml = thumbCell('Après', c.hasPhotoApres, c.driveFileIdApres, c.photoRefApres);

    var tr = document.createElement('tr');
    tr.style.cursor = 'pointer';
    tr.innerHTML =
      '<td><span class="rbadge">'+hl(emplTxt)+'</span></td>' +
      '<td>'+(c.zone ? '<span class="rbadge">'+hl(c.zone)+'</span>' : '<span style="color:var(--text3);font-size:9px">—</span>')+'</td>' +
      '<td style="white-space:nowrap;font-size:10px">'+esc(c.date||'—')+'</td>' +
      '<td>' + (statTxt === '—' ? '<span style="color:var(--text3);font-size:9px">—</span>' : '<span class="'+statClass+'">'+esc(statTxt)+'</span>') + '</td>' +
      avantCellHtml + apresCellHtml;

    tr.addEventListener('click', function(){ openCleanModal(c); });
    tb.appendChild(tr);
  });

  // ── Chargement asynchrone des vignettes (avec cache) ───────────
  if (!window._clnPhotoCache) window._clnPhotoCache = {};
  pendingThumbs.forEach(function(p){
    var slot = document.getElementById(p.cellId);
    if (!slot) return;
    var cached = window._clnPhotoCache[p.fid];
    if (cached) {
      replaceThumb(slot, cached, p);
      return;
    }
    google.script.run
      .withSuccessHandler(function(d){
        if (d) {
          window._clnPhotoCache[p.fid] = d;
          replaceThumb(slot, d, p);
        } else {
          slot.outerHTML = '<div class="cln-thumb-ph" title="Lecture impossible">⚠️</div>';
        }
      })
      .withFailureHandler(function(){
        slot.outerHTML = '<div class="cln-thumb-ph" title="Lecture impossible">⚠️</div>';
      })
      .getTagPhotoForClient(p.fid);
  });
}
function replaceThumb(slot, dataUrl, p){
  if (!slot || !slot.parentNode) return;
  var img = document.createElement('img');
  img.src = dataUrl;
  img.className = 'cln-thumb';
  img.alt = p.label;
  img.title = p.label + (p.ref ? ' · ' + p.ref : '') + ' (cliquer pour agrandir)';
  img.addEventListener('click', function(e){
    e.stopPropagation();
    openCleanModal(p.cln);
    var photoTab = document.querySelectorAll('#cleanModal .modal-tab')[1];
    if (photoTab) switchTab('mCln','photo',photoTab);
  });
  slot.parentNode.replaceChild(img, slot);
}

function openNotifFull(){
  // Fermer le petit panel
  document.getElementById('notifPanel').classList.remove('open');
  // Populer le sélecteur de semaines
  populateNotifWeeks();
  // Appliquer le filtre et afficher
  applyNotifFullFilter();
  document.getElementById('notifFullBg').classList.add('open');
  document.body.style.overflow='hidden';
}
function closeNotifFull(){
  document.getElementById('notifFullBg').classList.remove('open');
  document.body.style.overflow='';
}
function onNotifFullPeriod(){
  var p=document.getElementById('notifFullPeriod').value;
  document.getElementById('notifFullMonth').style.display=(p==='month')?'inline-block':'none';
  document.getElementById('notifFullWeek').style.display=(p==='week')?'inline-block':'none';
  if(p==='week') populateNotifWeeks();
  applyNotifFullFilter();
}
function populateNotifWeeks(){
  var sel=document.getElementById('notifFullWeek');
  var yr=parseInt(document.getElementById('notifFullYear').value)||2026;
  // Générer les semaines de l'année (lundi au dimanche)
  var opts='';
  var d=new Date(yr,0,1);
  // Aller au premier lundi
  while(d.getDay()!==1) d.setDate(d.getDate()+1);
  var wn=1;
  var today=new Date();
  var selectedWeek='';
  while(d.getFullYear()<=yr&&wn<=53){
    var end=new Date(d); end.setDate(end.getDate()+6);
    var startStr=d.toISOString().split('T')[0];
    var endStr=end.toISOString().split('T')[0];
    var label='S'+wn+' ('+fmtDateShort(d)+' → '+fmtDateShort(end)+')';
    opts+='<option value="'+startStr+'|'+endStr+'">'+label+'</option>';
    // Sélectionner la semaine courante
    if(today>=d&&today<=end) selectedWeek=startStr+'|'+endStr;
    d.setDate(d.getDate()+7);
    wn++;
  }
  sel.innerHTML=opts;
  if(selectedWeek) sel.value=selectedWeek;
}
function fmtDateShort(d){
  var dd=String(d.getDate()).padStart(2,'0');
  var mm=String(d.getMonth()+1).padStart(2,'0');
  return dd+'/'+mm;
}
function applyNotifFullFilter(){
  if(!D||!D.planning) return;
  var period=document.getElementById('notifFullPeriod').value;
  var yr=parseInt(document.getElementById('notifFullYear').value)||2026;
  var mo=parseInt(document.getElementById('notifFullMonth').value)||1;
  var weekVal=document.getElementById('notifFullWeek').value;

  var filtered=D.planning.filter(function(p){
    if(!p.isoDate) return false;
    var pts=p.isoDate.split('-');
    var pYr=parseInt(pts[0]),pMo=parseInt(pts[1]);

    if(period==='all'){
      return pYr===yr;
    }
    if(period==='month'){
      return pYr===yr&&pMo===mo;
    }
    if(period==='week'&&weekVal){
      var parts=weekVal.split('|');
      return p.isoDate>=parts[0]&&p.isoDate<=parts[1];
    }
    return true;
  });

  // Tri par date
  filtered.sort(function(a,b){return a.isoDate.localeCompare(b.isoDate);});

  document.getElementById('notifFullCount').textContent=filtered.length+' événement'+(filtered.length>1?'s':'');
  document.getElementById('notifFullStat').textContent=filtered.length+' événement'+(filtered.length>1?'s':'');

  var body=document.getElementById('notifFullBody');
  if(filtered.length===0){
    body.innerHTML='<div class="notif-full-empty">📭 Aucune notification pour cette période</div>';
    return;
  }

  var DOWS=['Dim','Lun','Mar','Mer','Jeu','Ven','Sam'];
  var MONTHS=['Jan','Fév','Mar','Avr','Mai','Jun','Jul','Aoû','Sep','Oct','Nov','Déc'];

  var html='';
  filtered.forEach(function(p){
    var d=new Date(p.isoDate+'T00:00:00');
    var day=d.getDate();
    var monthName=MONTHS[d.getMonth()];
    var dow=DOWS[d.getDay()];
    var isFumig=p.fumigation||p.activite.toLowerCase().indexOf('fumig')>-1;

    // Déterminer le statut
    var cls='nf-future', lblCls='nf-lbl-future', lblTxt='À venir';
    if(p.isoDate===TODAY_ISO){ cls='nf-today'; lblCls='nf-lbl-today'; lblTxt="Aujourd'hui"; }
    else if(p.isoDate===TOMORROW_ISO){ cls='nf-tomorrow'; lblCls='nf-lbl-tomorrow'; lblTxt='Demain'; }
    else if(p.isoDate<TODAY_ISO){ cls='nf-past'; lblCls='nf-lbl-past'; lblTxt='Passé'; }
    if(isFumig){ cls+=' nf-fumig'; lblCls='nf-lbl-fumig'; lblTxt=(p.isoDate===TODAY_ISO?"Fumigation aujourd'hui":p.isoDate===TOMORROW_ISO?'Fumigation demain':'Fumigation'); }

    html+='<div class="notif-full-item '+cls+'">'
      +'<div class="nf-date-col">'
      +'<div class="nf-date-day">'+day+'</div>'
      +'<div class="nf-date-month">'+monthName+'</div>'
      +'<div class="nf-date-dow">'+dow+'</div>'
      +'</div>'
      +'<div class="nf-content">'
      +'<span class="nf-act-label '+lblCls+'">'+lblTxt+'</span>'
      +'<div class="nf-act-title">'+(isFumig?'🦟 ':'')+esc(p.activite)+'</div>'
      +(p.details?'<div class="nf-act-details">'+esc(p.details)+'</div>':'')
      +'<div style="font-size:9px;color:var(--text3);margin-top:3px;font-family:\'JetBrains Mono\',monospace">📅 '+p.date+'</div>'
      +'</div></div>';
  });
  body.innerHTML=html;
}
document.addEventListener('keydown',function(e){
  if(e.key==='Escape'&&document.getElementById('notifFullBg').classList.contains('open')) closeNotifFull();
});
