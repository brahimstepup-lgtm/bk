function prog(lbl,val,tgt,lower,isPct){
  if(val===null||val===undefined) return'';
  if(!tgt||tgt===0) tgt=1;
  var pct=Math.min(Math.round(Math.abs(val)/Math.abs(tgt)*100),100);
  var ok=lower?val<=tgt:val>=tgt;
  var tc=ok?'ok':(pct>60?'warn':'bad'), bc=ok?'bg':(pct>60?'ba':'br');
  return'<div class="prow"><div class="phdr"><span class="plbl">'+lbl+'</span><span class="pnum '+tc+'">'+(isPct?val+'%':val)+'</span></div><div class="pbg"><div class="pb '+bc+'" style="width:'+pct+'%"></div></div></div>';
}

function cleanMonths(arr,hdrs){

}

function parse5M(text){
  var r={MO:'',MAT:'',METH:'',MIL:'',MGMT:''};
  if(!text) return r;
  var lines=text.split('\n'),curKey=null,cur=[];
  var map=[{keys:["main d'oeuvre","main-d'oeuvre","main d'œuvre","main :"],k:'MO'},
    {keys:['matière','matéri','matériel'],k:'MAT'},{keys:['méthode','methode'],k:'METH'},
    {keys:['milieu','environnement'],k:'MIL'},{keys:['management','machine','mesure'],k:'MGMT'}];
  lines.forEach(function(l){
    var lo=l.toLowerCase().replace(/[*#\\]/g,'').trim(),found=null;
    for(var i=0;i<map.length;i++){for(var j=0;j<map[i].keys.length;j++){if(lo.indexOf(map[i].keys[j])>-1){found=map[i].k;break;}}if(found)break;}
    if(found){if(curKey)r[curKey]=cur.join(' ').trim();curKey=found;cur=[l.replace(/^[-•*#\\: ]+/,'').trim()];}
    else if(curKey&&l.trim()) cur.push(l.replace(/^[-•*#\\ ]+/,'').trim());
  });
  if(curKey) r[curKey]=cur.join(' ').trim();
  return r;
}

  var p=document.getElementById('incPer').value;
  document.getElementById('incMo').style.display=p==='month'?'inline-block':'none';
  document.getElementById('incYr').style.display=p==='month'?'inline-block':'none';
  document.getElementById('incD1').style.display=p==='custom'?'inline-block':'none';
  document.getElementById('incSep').style.display=p==='custom'?'inline-block':'none';
  document.getElementById('incD2').style.display=p==='custom'?'inline-block':'none';
  applyIncF();
}
function applyIncF(){
  if(!D) return;
  var p=document.getElementById('incPer').value;
  var mo=parseInt(document.getElementById('incMo').value)||4;
  var yr=parseInt(document.getElementById('incYr').value)||2026;
  var d1=document.getElementById('incD1').value,d2=document.getElementById('incD2').value;
  var filtered=(D.incidents||[]).filter(function(inc){
    if(!inc.id||inc._err) return false;
    if(!inc.victime&&!inc.description) return false;
    if(p==='all') return true;
    var iso=inc.isoDate; if(!iso) return p==='ytd';
    if(p==='ytd'){ var y=parseInt(iso.substring(0,4)); return y===2026||y===2025; }
    if(p==='month'){ var pts=iso.split('-'); return parseInt(pts[0])===yr&&parseInt(pts[1])===mo; }
    if(p==='custom'){ var ok=true; if(d1)ok=ok&&iso>=d1; if(d2)ok=ok&&iso<=d2; return ok; }
    return true;
  });
  renderIncidents(filtered);
}
function renderIncidents(incs){
  document.getElementById('incStat').textContent=incs.length+' incidents';
  var s3=incs.filter(function(i){return String(i.gravite)==='3';}).length;
  var s2=incs.filter(function(i){return String(i.gravite)==='2';}).length;
  var s1=incs.filter(function(i){return String(i.gravite)==='1';}).length;
  var cl=incs.filter(function(i){return i.statut&&i.statut.toLowerCase().indexOf('clôt')>-1;}).length;
  document.getElementById('incKpi').innerHTML=[
    kpiCard('Total',incs.length,'r','incidents','k-r',null,null),
    kpiCard('Grave G3',s3,'r','Gravité 3','k-r',null,null),
    kpiCard('Modéré G2',s2,'a','Gravité 2','k-a',null,null),
    kpiCard('Faible G1',s1,'b','Gravité 1','k-b',null,null),
    kpiCard('Clôturés',cl,'g','soldés','k-g',null,null)
  ].join('');
  // Zones
  var zones={};
  incs.forEach(function(i){if(i.lieu){var z=i.lieu.substring(0,25);zones[z]=(zones[z]||0)+1;}});
  var zk=Object.keys(zones).sort(function(a,b){return zones[b]-zones[a];}).slice(0,5);
  mkChart('zoneC','bar',zk,[{data:zk.map(function(k){return zones[k];}),backgroundColor:[C.red,C.amber,C.p2,C.teal,C.purple],borderRadius:5}],
    Object.assign({},BASE,{indexAxis:'y',scales:{x:{grid:{color:C.grid},ticks:{color:C.txt}},y:{grid:{display:false},ticks:{color:C.txt,font:{size:9}}}}}));
  // Causes racines (statique)
  document.getElementById('causeGrid').innerHTML=[
    ['Sol mouillé / glissade'],['Bords tranchants / carton'],['Chute d\'objet / palette'],
    ['Mauvaise méthode'],['Manque vigilance'],['Collision engin']
  ].map(function(c,i){
    return'<div style="background:var(--bg3);border-radius:7px;padding:8px 10px;transition:background .3s">'
      +'<div style="font-size:17px;font-weight:800;font-family:\'JetBrains Mono\',monospace;color:var(--amber);margin-bottom:1px">'+(6-i)+'</div>'
      +'<div style="font-size:10px;color:var(--text3)">'+c[0]+'</div></div>';
  }).join('');
  // Liste
  var il=document.getElementById('incList'); il.innerHTML='';
  if(incs.length===0){il.innerHTML='<div style="padding:16px;text-align:center;color:var(--text3);font-size:12px">Aucun incident pour cette période</div>';return;}
  incs.forEach(function(inc){
    if(!inc.id||inc._err) return;
    var sev=String(inc.gravite||1);
    var stat=(inc.statut||'').toLowerCase().indexOf('clôt')>-1?'Clôturé':'En cours';
    var el=document.createElement('div');
    el.className='iitem is'+sev; el.tabIndex=0;
    el.innerHTML='<div class="ihdr"><span class="iid">INC-'+esc(inc.id)+(inc.date?' · '+inc.date:'')+'</span><span class="isev sv'+sev+'">G'+sev+'</span></div>'
      +'<div class="ivic">'+(inc.victime||'—')+'</div>'
      +(inc.description?'<div class="idesc">'+esc(inc.description.substring(0,90))+(inc.description.length>90?'…':'')+'</div>':'')
      +'<div class="itags">'+(inc.lieu?'<span class="itag">📍 '+esc(inc.lieu)+'</span>':'')+(inc.type?'<span class="itag">'+esc(inc.type)+'</span>':'')
      +'<span class="'+(stat==='Clôturé'?'s-close':'s-open')+'">'+stat+'</span>'
      +'<span class="iclick-hint">🔍 Voir détails</span></div>';
    el.onclick=function(){openIncModal(inc);};
    el.onkeydown=function(e){if(e.key==='Enter'||e.key===' ')openIncModal(inc);};
    il.appendChild(el);
  });
}

function populateTagDropdowns(){
  if(!D||!D.tags) return;
  var resps=new Set(),zones=new Set();
  D.tags.forEach(function(t){
    if(t.responsable&&t.responsable.trim()) resps.add(t.responsable.trim());
    var z=(t.zone||t.emplacement||'').trim(); if(z) zones.add(z);
  });
  var rSel=document.getElementById('tagFilterResp');
  rSel.innerHTML='<option value="all">Tous ('+resps.size+')</option>';
  Array.from(resps).sort().forEach(function(r){rSel.innerHTML+='<option value="'+esc(r)+'">'+esc(r)+'</option>';});
  var zSel=document.getElementById('tagFilterZone');
  zSel.innerHTML='<option value="all">Toutes ('+zones.size+')</option>';
  Array.from(zones).sort().forEach(function(z){zSel.innerHTML+='<option value="'+esc(z)+'">'+esc(z)+'</option>';});
}

function applyTagF(){
  if(!D) return;
  var p=document.getElementById('tagPer').value;
  var yr=parseInt(document.getElementById('tagYr').value)||2026;
  var mo=parseInt(document.getElementById('tagMo').value)||1;
  var d1=document.getElementById('tagD1').value,d2=document.getElementById('tagD2').value;
  var searchVal=(document.getElementById('tagSearch').value||'').toLowerCase().trim();
  var sortVal=document.getElementById('tagSort').value;
  var filterResp=document.getElementById('tagFilterResp').value;
  var filterZone=document.getElementById('tagFilterZone').value;

  var filtered=(D.tags||[]).filter(function(t){
    if(!t.id||t._err) return false;
    var isOpen=t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;
    if(tagStatFilter==='open'&&!isOpen) return false;
    if(tagStatFilter==='closed'&&isOpen) return false;
    if(filterResp!=='all'&&(t.responsable||'').trim()!==filterResp) return false;
    if(filterZone!=='all'){var z=(t.zone||t.emplacement||'').trim();if(z!==filterZone)return false;}
    if(p!=='all'){
      if(!t.isoDate) return p==='ytd';
      var pts=t.isoDate.split('-');
      if(p==='ytd'&&parseInt(pts[0])!==2026) return false;
      if(p==='month'&&(parseInt(pts[0])!==yr||parseInt(pts[1])!==mo)) return false;
      if(p==='custom'){if(d1&&t.isoDate<d1)return false;if(d2&&t.isoDate>d2)return false;}
    }
    if(searchVal){
      // Recherche dans tous les champs, SAUF "tag" et "tags" ignorés
      var hay=(t.description+' '+(t.zone||t.emplacement||'')+' '+t.type+' '
        +t.responsable+' '+t.id+' '+t.risque+' '
        // photoRef sans "tag" ni "tags"
        +(t.photoRef||'').replace(/\btags?\b/gi,'')).toLowerCase();
      if(hay.indexOf(searchVal)===-1) return false;
    }
    return true;
  });

  // Tri
  filtered.sort(function(a,b){
    if(sortVal==='date_desc'){if(!a.isoDate)return 1;if(!b.isoDate)return -1;return b.isoDate.localeCompare(a.isoDate);}
    if(sortVal==='date_asc'){ if(!a.isoDate)return 1;if(!b.isoDate)return -1;return a.isoDate.localeCompare(b.isoDate);}
    if(sortVal==='id_asc')  return String(a.id).localeCompare(String(b.id),undefined,{numeric:true});
    if(sortVal==='id_desc') return String(b.id).localeCompare(String(a.id),undefined,{numeric:true});
    if(sortVal==='zone_az') return (a.zone||a.emplacement||'').localeCompare(b.zone||b.emplacement||'');
    if(sortVal==='resp_az') return (a.responsable||'').localeCompare(b.responsable||'');
    return 0;
  });

  var total=D.tags.length;
  var openAll=D.tagsOpen.length,closedAll=total-openAll;
  var openF=filtered.filter(function(t){return t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;}).length;
  var closedF=filtered.length-openF;
  var cloPct=filtered.length>0?Math.round(closedF/filtered.length*100):0;

  document.getElementById('tagStatsRow').innerHTML=
    '<span class="tag-stat-pill tsp-total">Total : '+total+'</span>'
    +'<span class="tag-stat-pill tsp-open">Ouverts : '+openAll+'</span>'
    +'<span class="tag-stat-pill tsp-closed">Fermés : '+closedAll+'</span>'
    +(filtered.length!==total?'<span class="tag-stat-pill tsp-filtered">Filtrés : <strong>'+filtered.length+'</strong></span>':'');
  document.getElementById('tagListCount').textContent=filtered.length+' résultats';

  document.getElementById('tagKpi').innerHTML=[
    kpiCard('Filtrés',filtered.length,'b','TAGs','k-b',null,null),
    kpiCard('Fermés',closedF,'g','✓','k-g',cloPct,C.green),
    kpiCard('Ouverts',openF,'r','⚠','k-r',filtered.length>0?Math.round(openF/filtered.length*100):0,C.red),
    kpiCard('Clôture',cloPct+'%','g','obj.95%','k-g',cloPct,cloPct>=95?C.green:cloPct>50?C.amber:C.red)
  ].join('');

  // Types de dangers
  var types={};
  filtered.forEach(function(t){if(t.type){var ty=t.type.substring(0,22);types[ty]=(types[ty]||0)+1;}});
  var tk=Object.keys(types).sort(function(a,b){return types[b]-types[a];}).slice(0,6);
  if(charts['dangerC'])charts['dangerC'].destroy();
  if(tk.length>0) new Chart(document.getElementById('dangerC'),{type:'doughnut',
    data:{labels:tk,datasets:[{data:tk.map(function(k){return types[k];}),backgroundColor:[C.amber,C.red,C.p2,C.teal,C.green,C.orange],borderWidth:0,hoverOffset:4}]},
    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:true,position:'right',labels:{color:C.txt,font:{size:9},boxWidth:8}}},cutout:'50%'}});

  // Zones TAGs
  var tzones={};
  filtered.forEach(function(t){if(t.zone||t.emplacement){var z=(t.zone||t.emplacement).substring(0,18);tzones[z]=(tzones[z]||0)+1;}});
  var tzk=Object.keys(tzones).sort(function(a,b){return tzones[b]-tzones[a];}).slice(0,6);
  mkChart('tagZoneC','bar',tzk,[{data:tzk.map(function(k){return tzones[k];}),backgroundColor:'rgba(9,99,126,.75)',borderRadius:4}],
    Object.assign({},BASE,{indexAxis:'y',scales:{x:{grid:{color:C.grid},ticks:{color:C.txt}},y:{grid:{display:false},ticks:{color:C.txt,font:{size:9}}}}}));

  // Table avec responsable + indicateur photo Drive
  var tb=document.getElementById('tagBody'); tb.innerHTML='';
  if(filtered.length===0){
    var msg=searchVal?'Aucun TAG pour "'+esc(searchVal)+'"'
      :filterResp!=='all'?'Aucun TAG pour "'+esc(filterResp)+'"'
      :filterZone!=='all'?'Aucun TAG zone "'+esc(filterZone)+'"'
      :'Aucun TAG pour ce filtre';
    tb.innerHTML='<tr><td colspan="8" style="text-align:center;color:var(--text3);padding:20px">'+msg+'</td></tr>';
    return;
  }
  function hl(str){
    if(!searchVal||!str) return esc(str||'');
    var re=new RegExp('('+searchVal.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');
    return esc(str).replace(re,'<mark style="background:rgba(245,166,35,.35);border-radius:2px;padding:0 1px">$1</mark>');
  }
  filtered.forEach(function(t){
    var isOpen=t.statut.toLowerCase().indexOf('ouvert')>-1||t.statut.toLowerCase().indexOf('open')>-1;
    var respHtml=t.responsable
      ?'<span class="rbadge" style="color:var(--p4)">'+hl(t.responsable)+'</span>'
      :'<span style="color:var(--text3);font-size:9px">—</span>';
    // Bouton photo : 📸 si Drive résolu, 📷 sinon
    // Le serveur renvoie t.hasPhoto (bool) + t.driveFileId (ID Drive résolu)
    var hasPhoto=!!(t.hasPhoto&&t.driveFileId);
    var photoCls=hasPhoto?'tag-btn has-photo':'tag-btn no-photo';
    var photoIco=hasPhoto?'📸':'📷';
    var photoTitle=hasPhoto
      ?'Photo Drive disponible : '+(t.photoClean||t.photoRef)
      :(t.photoRef?'Pas de photo Drive : '+(t.photoClean||t.photoRef):'Pas de photo');
    var tr=document.createElement('tr');
    tr.innerHTML='<td class="mono" style="color:var(--p4)">'+esc(t.id)+'</td>'
      +'<td style="white-space:nowrap;font-size:10px">'+t.date+'</td>'
      +'<td style="font-size:10px">'+hl(t.type)+'</td>'
      +'<td><span class="rbadge">'+hl((t.zone||t.emplacement)||'—')+'</span></td>'
      +'<td style="font-size:10px;max-width:130px">'+hl(t.description.substring(0,55))+(t.description.length>55?'…':'')+'</td>'
      +'<td style="font-size:10px">'+respHtml+'</td>'
      +'<td><span class="'+(isOpen?'s-open':'s-close')+'">'+(isOpen?'Ouvert':'Fermé')+'</span></td>'
      +'<td><button class="'+photoCls+'" title="'+esc(photoTitle)+'">'+photoIco+'</button></td>';
    tr.querySelector('.tag-btn').addEventListener('click',function(e){
      e.stopPropagation();
      openTagModal(t);
      // Basculer directement sur l'onglet Photo
      var photoTab=document.querySelectorAll('#tagModal .modal-tab')[1];
      if(photoTab) switchTab('mTag','photo',photoTab);
    });
    tb.appendChild(tr);
  });
}

function applyAccF(){
  if(!D) return;
  var p=document.getElementById('accPer').value,mois=document.getElementById('accMo').value;
  var d1=document.getElementById('accD1').value,d2=document.getElementById('accD2').value;

function renderAccidents(accs){
  document.getElementById('accStat').textContent=accs.length+' accidents';
  var totalNJP=0; accs.forEach(function(a){totalNJP+=a.njp;});
  var njpM={Janvier:0,Février:0,Mars:0,Avril:0};
  accs.forEach(function(a){if(njpM[a.mois]!==undefined)njpM[a.mois]+=a.njp;});
  var maxMois=Object.keys(njpM).reduce(function(a,b){return njpM[a]>njpM[b]?a:b;},'Janvier');
  document.getElementById('accCount').textContent=accs.length+' accidents';
  document.getElementById('accSum').innerHTML=
    '<div class="acc-card"><div class="acc-val" style="color:var(--red)">'+accs.length+'</div><div class="acc-lbl">AT filtrés</div></div>'+
    '<div class="acc-card"><div class="acc-val" style="color:var(--amber)">'+totalNJP+'</div><div class="acc-lbl">NJP total</div></div>'+
    '<div class="acc-card"><div class="acc-val" style="color:var(--p4)">'+njpM[maxMois]+'</div><div class="acc-lbl">NJP max</div></div>'+
    '<div class="acc-card"><div class="acc-val" style="color:var(--green)">'+maxMois+'</div><div class="acc-lbl">Mois +accid.</div></div>';
  var njpL=[],njpD=[];
  ['Janvier','Février','Mars','Avril'].forEach(function(mo){if(njpM[mo]>=0){njpL.push(mo.substring(0,3));njpD.push(njpM[mo]);}});
  if(njpL.length>0) mkChart('njpC','bar',njpL,[{data:njpD,backgroundColor:[C.red,'rgba(232,68,90,.7)','rgba(232,68,90,.5)','rgba(232,68,90,.4)'],borderRadius:5}],BASE);

  if(zk.length>0) mkChart('accZC','bar',zk,[{data:zk.map(function(k){return zones[k];}),backgroundColor:'rgba(9,99,126,.75)',borderRadius:4}],
    Object.assign({},BASE,{indexAxis:'y',scales:{x:{grid:{color:C.grid},ticks:{color:C.txt}},y:{grid:{display:false},ticks:{color:C.txt,font:{size:9}}}}}));
  var causes={};accs.forEach(function(a){if(a.cause){var c=a.cause.substring(0,35);causes[c]=(causes[c]||0)+1;}});
  var ck=Object.keys(causes).sort(function(a,b){return causes[b]-causes[a];}).slice(0,5);
  if(ck.length>0) mkChart('accCC','bar',ck,[{data:ck.map(function(k){return causes[k];}),backgroundColor:[C.amber,C.red,C.p2,C.teal,C.purple],borderRadius:4}],
    Object.assign({},BASE,{indexAxis:'y',scales:{x:{grid:{color:C.grid},ticks:{color:C.txt}},y:{grid:{display:false},ticks:{color:C.txt,font:{size:9}}}}}));
  var foncs={};accs.forEach(function(a){if(a.fonction){var f=a.fonction.substring(0,25);foncs[f]=(foncs[f]||0)+1;}});
  var fk=Object.keys(foncs).sort(function(a,b){return foncs[b]-foncs[a];}).slice(0,5);
  if(fk.length>0){if(charts['accFC'])charts['accFC'].destroy();new Chart(document.getElementById('accFC'),{type:'doughnut',

    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:true,position:'right',labels:{color:C.txt,font:{size:9},boxWidth:8}}},cutout:'50%'}});}

  document.getElementById('envC').classList.toggle('on',m==='conso');
  document.getElementById('envR').classList.toggle('on',m==='ratio');
  if(D) updateEnvCharts();
}
function updateEnvCharts(){
  var m=D.metrics,isR=envMode==='ratio',hdrs=D.monthHeaders||[];
  document.getElementById('tElt').textContent=isR?'Ratio ELT kWh/UP':'Électricité kWh';
  document.getElementById('tGaz').textContent=isR?'Ratio GAZ m³/UP':'Gaz m³';
  document.getElementById('tEau').textContent=isR?'Ratio EAU m³/UP':'Eau m³';
  var eA=isR?m.relt_months:m.elt_months,gA=isR?m.rgaz_months:m.gaz_months,wA=isR?m.reau_months:m.eau_months;

function askG(prompt,display){
  var dp=display||prompt.substring(0,55)+'…';
  var resp=document.getElementById('aiResp'),st=document.getElementById('aiSt');
  var btn=document.getElementById('btnSend'),bta=document.getElementById('btnAuto');
  resp.innerHTML='<span class="typing">Gemini analyse…</span>';
  st.textContent='⏳'; btn.disabled=true; bta.disabled=true;
  google.script.run
    .withSuccessHandler(function(txt){
      resp.textContent=txt; st.textContent='✅'; btn.disabled=false; bta.disabled=false;
      aiH.unshift({q:dp.substring(0,55),a:txt.substring(0,90)+'…'});
      if(aiH.length>5) aiH.pop();
      document.getElementById('aiHist').innerHTML=aiH.map(function(x){
        return'<div style="background:var(--bg3);border-radius:7px;padding:8px 10px;transition:background .3s">'
          +'<div style="font-size:9px;color:var(--p4);margin-bottom:3px">❓ '+x.q+'</div>'
          +'<div style="font-size:9px;color:var(--text3)">→ '+x.a+'</div></div>';
      }).join('');
    })
    .withFailureHandler(function(err){resp.textContent='❌ '+err.message;st.textContent='❌';btn.disabled=false;bta.disabled=false;})
    .askGeminiForClient(buildCtx()+prompt);
}
