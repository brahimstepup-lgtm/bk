function doGet(e) {
  var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : '';
  if (action === 'data') return ContentService
    .createTextOutput(JSON.stringify(getData()))
    .setMimeType(ContentService.MimeType.JSON);
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('SOBCO 3 – HSE Master Dashboard')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// ── PROXY IMAGE BASE64 ───────────────────────────────────────
// Appelé depuis le client via google.script.run
// Retourne "data:image/jpeg;base64,..." ou ''
// ════════════════════════════════════════════════════════════
function getTagPhotoForClient(fileId) {
  if (!fileId || fileId === '') return '';
  try {
    var file = DriveApp.getFileById(fileId);
    var blob = file.getBlob();
    var b64  = Utilities.base64Encode(blob.getBytes());
    var mime = blob.getContentType() || 'image/jpeg';
    return 'data:' + mime + ';base64,' + b64;
  } catch(e) {
    Logger.log('getTagPhotoForClient error: ' + e.message);
    return '';
  }
}

// ── RÉSOLUTION PHOTO : référence → fileId Drive ──────────────
// Stratégie : extraire le numéro de la référence (ex: "431")
// puis chercher dans le dossier Drive avec title contains "431"
// Évite totalement le préfixe "tags/" et les accents
// ════════════════════════════════════════════════════════════
function resolveTagFileId(refRaw) {
  if (!refRaw || refRaw.trim() === '') return '';
  try {
    // 1. Nettoyer le préfixe tags/ ou tag/
    var clean = String(refRaw).replace(/^tags?\//i, '').trim();

    // 2. Extraire le premier numéro rencontré (accepte "431.xxx", "photo_431", "ref-431.jpg", etc.)
    var numMatch = clean.match(/(\d+)/);

    if (numMatch) {
      var num = numMatch[1];
      var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
      // Recherche Drive par numéro dans le titre — fonctionne avec accents
      var files = folder.searchFiles('title contains "' + num + '"');
      if (files.hasNext()) {
        return files.next().getId();
      }
    }
  } catch(e) {
    Logger.log('resolveTagFileId error: ' + e.message + ' | ref: ' + refRaw);
  }
  return '';
}

// ── SCAN INITIAL : pré-charger les IDs de tous les TAGs ──────
// Appelé une seule fois dans getData() pour éviter N appels Drive
// Construit un cache { "431": { avant:"fileId", apres:"fileId" }, ... }
// La classification AVANT / APRÈS se fait par mot-clé dans le nom :
//   - "apres" / "après" / "after"  → APRÈS
//   - tout le reste (par défaut)   → AVANT
// ════════════════════════════════════════════════════════════
function buildDriveTagsCache() {
  var cache  = {};  // num → { any: fileId, avant: fileId, apres: fileId }
  var count  = 0;
  try {
    var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
    var files  = folder.getFiles();
    while (files.hasNext()) {
      var file = files.next();
      var name = file.getName();
      var id   = file.getId();
      count++;
      // Extraire le premier numéro rencontré : "431.Référence...", "photo_431.jpg", "ref-431" → "431"
      var m = name.match(/(\d+)/);
      if (m) {
        var num = m[1];
        if (!cache[num]) cache[num] = { any: '', avant: '', apres: '' };
        // Tout premier fichier rencontré pour ce numéro → "any"
        if (!cache[num].any) cache[num].any = id;
        // Détection APRÈS — accents normalisés
        var low = String(name).toLowerCase()
          .replace(/[éèêë]/g,'e').replace(/[àâä]/g,'a').replace(/[ùûü]/g,'u');
        var isApres = /(^|[^a-z])(apres|after|aft)([^a-z]|$)/.test(low);
        if (isApres) {
          if (!cache[num].apres) cache[num].apres = id;
        } else {
          if (!cache[num].avant) cache[num].avant = id;
        }
      }
    }
    Logger.log('Drive cache: ' + count + ' fichiers, ' + Object.keys(cache).length + ' numéros');
  } catch(e) {
    Logger.log('buildDriveTagsCache error: ' + e.message);
  }
  return { cache: cache, count: count };
}

// ── Utilitaires ───────────────────────────────────────────────
function isNull(v) {
  if (v === null || v === undefined) return true;
  var s = String(v).trim(); return s === '' || s === '/' || s === '-';
}
function toNum(v) {
  if (isNull(v)) return null;
  var n = parseFloat(String(v).replace(/\s/g,'').replace(/,/g,''));
  return isNaN(n) ? null : n;
}
function toNumZ(v) { var n = toNum(v); return n === null ? 0 : n; }

function parseDate(v) {
  if (isNull(v)) return null;
  if (v instanceof Date && !isNaN(v.getTime())) return v;
  var s = String(v).trim();
  var m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})/);
  if (m) {
    var d = parseInt(m[1]), mo = parseInt(m[2]), yr = parseInt(m[3]);
    if (d > 12)  return new Date(yr, mo-1, d);
    if (mo > 12) return new Date(yr, d-1, mo);
    return new Date(yr, mo-1, d);
  }
  try { var r = new Date(v); return isNaN(r.getTime()) ? null : r; } catch(e2) { return null; }
}
function fmtDate(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return String(v).trim();
  try { return Utilities.formatDate(d, 'GMT+1', 'dd/MM/yyyy'); } catch(e) { return String(v).trim(); }
}
function getIso(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return '';
  try { return d.toISOString().split('T')[0]; } catch(e) { return ''; }
}
function isFumigation(a) { return a && a.toLowerCase().indexOf('fumig') > -1; }
function normalize(s) {
  if (!s) return '';
  return s.toLowerCase()
    .replace(/[àáâãäå]/g,'a').replace(/[èéêë]/g,'e')
    .replace(/[ìíîï]/g,'i').replace(/[òóôõö]/g,'o')
    .replace(/[ùúûü]/g,'u').replace(/[ç]/g,'c').replace(/[ñ]/g,'n')
    .replace(/\s+/g,' ').trim();
}

function readPlanning(ss) {
  var planning = [];
  try {
    var sheets = ss.getSheets();
    var planSheet = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === PLANNING_GID) { planSheet = sheets[i]; break; }
    }
    if (!planSheet) {
      for (var i = 0; i < sheets.length; i++) {
        var n = sheets[i].getName().toLowerCase();
        if (n.indexOf('planning') > -1 || n.indexOf('activit') > -1 || n.indexOf('suivi') > -1) {
          planSheet = sheets[i]; break;
        }
      }
    }
    if (!planSheet) { Logger.log('Planning sheet NOT found'); return planning; }
    Logger.log('Planning sheet found: ' + planSheet.getName());
    var data = planSheet.getDataRange().getValues();
    var startRow = 1;
    for (var i = 0; i < Math.min(data.length, 10); i++) {
      var h0 = String(data[i][0]||'').toLowerCase();
      if (h0.indexOf('date') > -1) { startRow = i + 1; break; }
    }
    for (var i = startRow; i < data.length; i++) {
      var r = data[i];
      if (isNull(r[0])) continue;
      var pd = parseDate(r[0]);
      if (!pd || isNaN(pd.getTime())) continue;
      var act = isNull(r[1]) ? '' : String(r[1]).trim();
      if (!act) continue;
      planning.push({
        date: fmtDate(r[0]),
        isoDate: pd.toISOString().split('T')[0],
        activite: act,
        details: isNull(r[2]) ? '' : String(r[2]).trim(),
        fumigation: isFumigation(act)
      });
    }
    Logger.log('Planning: ' + planning.length + ' entrées');
  } catch(e) { Logger.log('Planning error: ' + e.message); }
  return planning;
}

// ── CACHE PHOTOS NETTOYAGE ───────────────────────────────────
// Construit un index { numéro → { any, avant, apres } } pour le
// sous-dossier "Suivi de Nettoyage - Sobco 3_Images" du dossier
// parent CLEAN_PARENT_FOLDER_ID.
// Si le sous-dossier n'est pas trouvé, on scanne aussi le parent
// directement (et ses sous-dossiers en récursif, profondeur 2).
// La distinction AVANT / APRÈS suit la même règle que pour les
// TAGs : mot-clé apres / après / after dans le nom de fichier.
// ════════════════════════════════════════════════════════════
function buildCleaningPhotosCache() {
  var cache  = {};   // num → { any, avant, apres }
  var byName = {};   // nom de fichier (lowercased) → fileId
  var count  = 0;

  function scanFolder(folder, depth) {
    if (!folder || depth < 0) return;
    var files = folder.getFiles();
    while (files.hasNext()) {
      var file = files.next();
      var name = file.getName();
      var id   = file.getId();
      count++;

      // Index par nom (utile si la feuille référence le nom complet)
      byName[String(name).toLowerCase()] = id;

      // Extraire le premier numéro rencontré
      var m = name.match(/(\d+)/);
      if (m) {
        var num = m[1];
        if (!cache[num]) cache[num] = { any: '', avant: '', apres: '' };
        if (!cache[num].any) cache[num].any = id;
        var low = String(name).toLowerCase()
          .replace(/[éèêë]/g,'e').replace(/[àâä]/g,'a').replace(/[ùûü]/g,'u');
        var isApres = /(^|[^a-z])(apres|after|aft)([^a-z]|$)/.test(low);
        if (isApres) {
          if (!cache[num].apres) cache[num].apres = id;
        } else {
          if (!cache[num].avant) cache[num].avant = id;
        }
      }
    }
    if (depth > 0) {
      var subs = folder.getFolders();
      while (subs.hasNext()) scanFolder(subs.next(), depth - 1);
    }
  }

  try {
    var parent = DriveApp.getFolderById(CLEAN_PARENT_FOLDER_ID);

    // 1. Chercher le sous-dossier nommé "Suivi de Nettoyage - Sobco 3_Images"
    var targetFolder = null;
    var subs = parent.getFolders();
    while (subs.hasNext()) {
      var sf = subs.next();
      if (normalize(sf.getName()) === normalize(CLEAN_IMAGES_FOLDER_NAME)) {
        targetFolder = sf; break;
      }
    }
    if (targetFolder) {
      scanFolder(targetFolder, 1);
    } else {
      // Fallback : scanner le dossier parent + tous ses sous-dossiers
      Logger.log('Cleaning subfolder not found by name, scanning parent recursively');
      scanFolder(parent, 2);
    }
    Logger.log('Cleaning cache: ' + count + ' fichiers, ' + Object.keys(cache).length + ' numéros');
  } catch(e) {
    Logger.log('buildCleaningPhotosCache error: ' + e.message);
  }
  return { cache: cache, byName: byName, count: count };
}

// ── LECTURE FEUILLE "Nettoyage " ─────────────────────────────
// Détection automatique des colonnes depuis la ligne d'entête.
// Mappe ensuite chaque ligne sur un objet exposé au client.
// ════════════════════════════════════════════════════════════
function readCleaning(ss, photoCache) {
  var entries = [];
  try {
    // 1. Trouver la feuille (par GID puis par nom)
    var sheets = ss.getSheets();
    var sh = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === CLEAN_SHEET_GID) { sh = sheets[i]; break; }
    }
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('nettoyage') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('Cleaning sheet NOT found'); return entries; }
    Logger.log('Cleaning sheet found: ' + sh.getName());

    var data = sh.getDataRange().getValues();
    if (!data || data.length < 2) return entries;

    // 2. Détecter la ligne d'entête (première ligne contenant "date" ou "n°")
    var hdrRow = 0;
    for (var i = 0; i < Math.min(data.length, 6); i++) {
      var row = data[i];
      var foundKeys = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||''));
        if (h.indexOf('date') > -1 || h.indexOf('zone') > -1 ||
            h.indexOf('description') > -1 || h.indexOf('observation') > -1 ||
            h.indexOf('statut') > -1 || h.indexOf('responsable') > -1) {
          foundKeys++;
        }
      }
      if (foundKeys >= 2) { hdrRow = i; break; }
    }

    var hdr = data[hdrRow];
    var COL = {
      id:-1, date:-1, dateCible:-1, zone:-1, emplacement:-1,
      description:-1, observation:-1, action:-1, responsable:-1,
      statut:-1, photoAvant:-1, photoApres:-1, photoGen:-1, type:-1, frequence:-1
    };
    for (var c = 0; c < hdr.length; c++) {
      var h = normalize(String(hdr[c]||'').replace(/[\n\r#]/g,'').trim());
      if (!h) continue;

      // PHOTO APRÈS — tester en premier pour éviter le piège "photoRef"
      if ((h.indexOf('photo') > -1 || h.indexOf('image') > -1 || h.indexOf('reference') > -1) &&
          (h.indexOf('apres') > -1 || h.indexOf('after') > -1)) {
        if (COL.photoApres < 0) COL.photoApres = c;
        continue;
      }
      // PHOTO AVANT
      if ((h.indexOf('photo') > -1 || h.indexOf('image') > -1 || h.indexOf('reference') > -1) &&
          (h.indexOf('avant') > -1 || h.indexOf('before') > -1)) {
        if (COL.photoAvant < 0) COL.photoAvant = c;
        continue;
      }
      // PHOTO générique (sans précision avant/après)
      if (h.indexOf('photo') > -1 || h.indexOf('image') > -1) {
        if (COL.photoGen < 0) COL.photoGen = c;
        continue;
      }

      if      (h.match(/^n.?\s*(cas|tag|ordre|$)/) || h === 'n')    { if (COL.id < 0) COL.id = c; }
      else if (h.indexOf('date') > -1 && (h.indexOf('cible') > -1 || h.indexOf('echeance') > -1)) { if (COL.dateCible < 0) COL.dateCible = c; }
      else if (h.indexOf('date') > -1)                              { if (COL.date < 0) COL.date = c; }
      else if (h.indexOf('zone') > -1)                              { if (COL.zone < 0) COL.zone = c; }
      else if (h.indexOf('emplacement') > -1 || h.indexOf('lieu') > -1) { if (COL.emplacement < 0) COL.emplacement = c; }
      else if (h.indexOf('description') > -1)                       { if (COL.description < 0) COL.description = c; }
      else if (h.indexOf('observation') > -1 || h.indexOf('constat') > -1) { if (COL.observation < 0) COL.observation = c; }
      else if (h.indexOf('action') > -1 || h.indexOf('mesure') > -1 || h.indexOf('correctif') > -1) { if (COL.action < 0) COL.action = c; }
      else if (h.indexOf('respons') > -1 || h.indexOf('charg') > -1) { if (COL.responsable < 0) COL.responsable = c; }
      else if (h.indexOf('statut') > -1 || h.indexOf('etat') > -1)  { if (COL.statut < 0) COL.statut = c; }
      else if (h.indexOf('type') > -1 || h.indexOf('nature') > -1)  { if (COL.type < 0) COL.type = c; }
      else if (h.indexOf('frequen') > -1 || h.indexOf('periodicit') > -1) { if (COL.frequence < 0) COL.frequence = c; }
    }
    // Fallback : si on n'a pas trouvé d'ID, prendre la première colonne
    if (COL.id < 0) COL.id = 0;
    Logger.log('Cleaning COL: ' + JSON.stringify(COL));

    // 3. Helper : résoudre un fileId Drive depuis une référence photo
    function resolveCleanPhoto(refRaw, prefer) {
      if (isNull(refRaw)) return '';
      var ref = String(refRaw).trim();
      if (!ref) return '';
      // Recherche directe par nom de fichier
      var byNameKey = ref.toLowerCase();
      if (photoCache.byName[byNameKey]) return photoCache.byName[byNameKey];
      // Recherche tolérante (avec / sans extension)
      var keys = Object.keys(photoCache.byName);
      for (var k = 0; k < keys.length; k++) {
        if (keys[k].indexOf(byNameKey) > -1 || byNameKey.indexOf(keys[k]) > -1) {
          return photoCache.byName[keys[k]];
        }
      }
      // Recherche par numéro
      var m = ref.match(/(\d+)/);
      if (m && photoCache.cache[m[1]]) {
        var entry = photoCache.cache[m[1]];
        // Pour "après" : ne JAMAIS retomber sur un fichier avant/any.
        // Seul un fichier explicitement tagué "apres/after" est accepté.
        if (prefer === 'apres') return entry.apres || '';
        if (prefer === 'avant') return entry.avant || entry.any || '';
        return entry.any || entry.avant || '';
      }
      return '';
    }

    // 4. Parcourir les lignes de données
    for (var i = hdrRow + 1; i < data.length; i++) {
      var r = data[i];

      // Filtrer les lignes vides ou de type "total" / "sous-total"
      var idVal = (COL.id >= 0) ? r[COL.id] : '';
      var firstText = '';
      for (var c = 0; c < r.length; c++) {
        if (!isNull(r[c])) { firstText = String(r[c]).trim(); break; }
      }
      if (!firstText) continue;
      var lc = firstText.toLowerCase();
      if (lc.indexOf('total') === 0 || lc.indexOf('sous-total') > -1) continue;

      // Récupérer toutes les valeurs avec fallbacks
      function val(c) { return (c >= 0 && !isNull(r[c])) ? String(r[c]).trim() : ''; }

      var id          = val(COL.id) || ('NET-' + (i+1));
      var dateRaw     = (COL.date >= 0) ? r[COL.date] : '';
      var dateCible   = (COL.dateCible >= 0) ? r[COL.dateCible] : '';
      var zone        = val(COL.zone);
      var emplacement = val(COL.emplacement);
      var description = val(COL.description);
      var observation = val(COL.observation);
      var action      = val(COL.action);
      var responsable = val(COL.responsable);
      var statut      = val(COL.statut);
      var type        = val(COL.type);
      var frequence   = val(COL.frequence);

      // Photos : 3 colonnes possibles
      var refAvant = val(COL.photoAvant) || val(COL.photoGen);
      var refApres = val(COL.photoApres);

      // RÈGLE STRICTE : on n'affiche une photo "après" que si la colonne
      // « photo après » de la feuille est explicitement renseignée.
      // Aucun fallback automatique sur le numéro de la photo "avant".
      var fileIdAvant = resolveCleanPhoto(refAvant, 'avant');
      var fileIdApres = refApres ? resolveCleanPhoto(refApres, 'apres') : '';

      // Sauter les lignes complètement vides (pas de zone, description, ni photo, ni statut)
      if (!zone && !emplacement && !description && !observation && !statut && !fileIdAvant && !fileIdApres && !val(COL.id)) continue;

      var pd = parseDate(dateRaw);
      entries.push({
        id:          String(id),
        date:        fmtDate(dateRaw),
        isoDate:     getIso(dateRaw),
        dateCible:   fmtDate(dateCible),
        zone:        zone,
        emplacement: emplacement,
        type:        type,
        frequence:   frequence,
        description: description,
        observation: observation,
        action:      action,
        responsable: responsable,
        statut:      statut,
        photoRefAvant:    refAvant,
        photoRefApres:    refApres,
        driveFileIdAvant: fileIdAvant,
        driveFileIdApres: fileIdApres,
        hasPhotoAvant:    fileIdAvant !== '',
        hasPhotoApres:    fileIdApres !== '',
        hasPhoto:         (fileIdAvant !== '' || fileIdApres !== '')
      });
    }

    // Tri : date décroissante puis ID décroissant
    entries.sort(function(a, b) {
      if (a.isoDate && b.isoDate && a.isoDate !== b.isoDate) return b.isoDate.localeCompare(a.isoDate);
      var na = parseInt(String(a.id).match(/\d+/), 10);
      var nb = parseInt(String(b.id).match(/\d+/), 10);
      if (isNaN(na) && isNaN(nb)) return String(b.id).localeCompare(String(a.id));
      if (isNaN(na)) return 1;
      if (isNaN(nb)) return -1;
      return nb - na;
    });

    Logger.log('Cleaning entries: ' + entries.length);
  } catch(e) {
    Logger.log('readCleaning error: ' + e.message);
  }
  return entries;
}

// ── GETDATA ──────────────────────────────────────────────────

function getData() {
  var ss     = SpreadsheetApp.openById(SHEET_ID);
  var sheets = ss.getSheets();
  var byName = {};
  sheets.forEach(function(sh) {
    byName[normalize(sh.getName())] = sh;
  });
  function findSheet(kwds) {
    var keys = Object.keys(byName);
    for (var k = 0; k < kwds.length; k++)
      for (var i = 0; i < keys.length; i++)
        if (keys[i].indexOf(kwds[k]) > -1) return byName[keys[i]];
    return null;
  }

  var out = {
    ts: new Date().toISOString(),
    kpis:{}, incidents:[], tags:[], tagsOpen:[], accidents:[], planning:[],
    cleaning:[],
    monthHeaders:[],
    logoUrl: 'https://drive.google.com/uc?export=view&id=' + LOGO_FILE_ID
  };

  out.planning = readPlanning(ss);

  // ── Cache Drive Tags ─────────────────────────────────────
  var driveResult = buildDriveTagsCache();

function callGemini(prompt) {
  var models=['gemini-2.0-flash','gemini-1.5-flash'];
  for (var i=0;i<models.length;i++) {
    try {
      var url='https://generativelanguage.googleapis.com/v1/models/'+models[i]+':generateContent?key='+GEMINI_KEY;
      var resp=UrlFetchApp.fetch(url,{method:'post',contentType:'application/json',
        payload:JSON.stringify({contents:[{parts:[{text:prompt}]}],generationConfig:{temperature:0.3,maxOutputTokens:1024}}),muteHttpExceptions:true});
      var json=JSON.parse(resp.getContentText());
      if (json.candidates&&json.candidates[0]&&json.candidates[0].content) return json.candidates[0].content.parts[0].text;
      if (json.error&&json.error.code===404) continue;
      if (json.error) return 'Erreur API : '+json.error.message;
    } catch(ex) { return 'Erreur : '+ex.message; }
  }
  return 'Aucun modèle Gemini disponible.';
}

// ── Helpers ──────────────────────────────────────────────────
function normalizeName_(s) {
  return normalize(String(s || '')).replace(/\s+/g, ' ').trim();
}

function escHtml_(s) {
  return String(s == null ? '' : s)
    .replace(/&/g, '&amp;').replace(/</g, '&lt;')
    .replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}

// Résout l'email à partir d'un nom de responsable.
// Match strict puis tolérant (sous-chaîne) — retourne '' si rien trouvé.
function resolveResponsableEmail_(name) {
  if (!name) return '';
  var key  = normalizeName_(name);
  var keys = Object.keys(TAG_RESPONSABLES_EMAIL);

  // 1. Match exact (insensible casse/accents/espaces)
  for (var i = 0; i < keys.length; i++) {
    if (normalizeName_(keys[i]) === key) return TAG_RESPONSABLES_EMAIL[keys[i]];
  }
  // 2. Match partiel : la clé est contenue dans le nom (ou inverse)
  for (var i = 0; i < keys.length; i++) {
    var nk = normalizeName_(keys[i]);
    if (nk && (key.indexOf(nk) > -1 || nk.indexOf(key) > -1)) {
      return TAG_RESPONSABLES_EMAIL[keys[i]];
    }
  }
  return '';
}

function loadNotifiedIds_() {
  try {
    var raw = PropertiesService.getScriptProperties().getProperty(TAG_NOTIFIED_KEY);
    return raw ? (JSON.parse(raw) || {}) : {};
  } catch (e) { return {}; }
}
function saveNotifiedIds_(obj) {
  try {
    PropertiesService.getScriptProperties()
      .setProperty(TAG_NOTIFIED_KEY, JSON.stringify(obj));
  } catch (e) { Logger.log('saveNotifiedIds error: ' + e.message); }
}

// ════════════════════════════════════════════════════════════
// ── DÉTECTION + ENVOI ────────────────────────────────────────
// Scanne la feuille TAGs et envoie un email pour chaque nouvel
// enregistrement non encore notifié (dont le responsable et la
// description sont renseignés).
// ════════════════════════════════════════════════════════════
function notifyNewTags() {
  var lock = LockService.getScriptLock();
  if (!lock.tryLock(5000)) return;

  try {
    var ss     = SpreadsheetApp.openById(SHEET_ID);
    var sheets = ss.getSheets();

    // Trouver la feuille TAGs (même heuristique que getData)
    var tagSheet = null;
    for (var i = 0; i < sheets.length; i++) {
      var n = normalize(sheets[i].getName());
      if (n.indexOf('tag') > -1 || n.indexOf('danger') > -1 || n.indexOf('situation') > -1) {
        tagSheet = sheets[i]; break;
      }
    }
    if (!tagSheet) { Logger.log('notifyNewTags: feuille TAGs introuvable'); return; }

    var data = tagSheet.getDataRange().getValues();
    if (!data || data.length < 2) return;

    // Détection dynamique des colonnes (cohérent avec getData)
    var hdr = data[0];
    var COL = { id:0, dateCreate:1, dateCible:2, type:3, gravite:4,
                emplacement:5, zone:6, description:7, risque:8,
                action:9, propositions:10, photoRef:11, statut:12, responsable:13 };
    for (var c = 0; c < hdr.length; c++) {
      var h = normalize(String(hdr[c] || '').replace(/[\n\r#]/g, '').trim());
      if      (h.match(/^n.?\s*(cas|tag|$)/))                           COL.id          = c;
      else if (h.indexOf('date c') === 0 && h.indexOf('cible') < 0 && h.indexOf('cr') > -1) COL.dateCreate = c;
      else if (h.indexOf('cible') > -1 || h.indexOf('echeance') > -1)   COL.dateCible   = c;
      else if (h.indexOf('type') > -1 && h.indexOf('danger') > -1)      COL.type        = c;
      else if (h.indexOf('gravit') > -1)                                 COL.gravite     = c;
      else if (h.indexOf('emplace') > -1)                                COL.emplacement = c;
      else if (h === 'zone')                                             COL.zone        = c;
      else if (h.indexOf('description') > -1)                            COL.description = c;
      else if (h.indexOf('risque') > -1)                                 COL.risque      = c;
      else if (h.indexOf('action') > -1 && h.indexOf('urgente') > -1)    COL.action      = c;
      else if (h.indexOf('proposition') > -1)                            COL.propositions= c;
      else if (h.indexOf('reference') > -1 && h.indexOf('photo') > -1)   COL.photoRef    = c;
      else if (h.indexOf('statut') > -1)                                 COL.statut      = c;
      else if (h.indexOf('respens') > -1 || h.indexOf('respons') > -1)   COL.responsable = c;
    }

    var notified = loadNotifiedIds_();
    var changed  = false;
    var sent     = 0;

    for (var i = 1; i < data.length; i++) {
      var r  = data[i];
      var id = isNull(r[COL.id]) ? '' : String(r[COL.id]).trim();
      if (!id) continue;

      // Ligne incomplète : on attend que les champs essentiels soient saisis
      var responsable = isNull(r[COL.responsable]) ? '' : String(r[COL.responsable]).trim();
      var description = isNull(r[COL.description]) ? '' : String(r[COL.description]).trim();
      if (!responsable || !description) continue;

      // Déjà notifié → skip
      if (notified[id]) continue;

      var email = resolveResponsableEmail_(responsable);
      if (!email) {
        Logger.log('notifyNewTags: pas d\'email mappé pour "' + responsable + '" (TAG ' + id + ')');
        notified[id] = { ts: new Date().toISOString(), skipped: 'no_email_for:' + responsable };
        changed = true;
        continue;
      }

      var tag = {
        id:          id,
        date:        fmtDate(r[COL.dateCreate]),
        dateCible:   fmtDate(r[COL.dateCible]),
        type:        isNull(r[COL.type])         ? '' : String(r[COL.type]).trim(),
        gravite:     isNull(r[COL.gravite])      ? '' : String(r[COL.gravite]).trim(),
        emplacement: isNull(r[COL.emplacement])  ? '' : String(r[COL.emplacement]).trim(),
        zone:        isNull(r[COL.zone])         ? '' : String(r[COL.zone]).trim(),
        description: description,
        risque:      isNull(r[COL.risque])       ? '' : String(r[COL.risque]).trim(),
        action:      isNull(r[COL.action])       ? '' : String(r[COL.action]).trim(),
        propositions:isNull(r[COL.propositions]) ? '' : String(r[COL.propositions]).trim(),
        statut:      isNull(r[COL.statut])       ? '' : String(r[COL.statut]).trim(),
        responsable: responsable
      };

      try {
        sendTagEmail_(email, tag);
        notified[id] = { ts: new Date().toISOString(), email: email };
        changed = true;
        sent++;
      } catch (e) {
        Logger.log('notifyNewTags: envoi raté pour TAG ' + id + ' → ' + e.message);
      }
    }

    if (changed) saveNotifiedIds_(notified);
    Logger.log('notifyNewTags: ' + sent + ' email(s) envoyé(s)');
  } finally {
    lock.releaseLock();
  }
}

// ── Construction et envoi de l'email HTML ────────────────────
function sendTagEmail_(email, tag) {
  var subject = 'Nouveau TAG ' + tag.id + ' — ' +
                (tag.zone || tag.emplacement || 'Situation dangereuse');

  // Couleur selon gravité (heuristique tolérante)
  var g = String(tag.gravite || '').toLowerCase();
  var sevColor = /3|haut|elev|crit/.test(g) ? '#e8445a'
               : /2|moy/.test(g)            ? '#f5a623'
               : '#0a7a9a';

  var rows = [
    ['N° TAG',           tag.id],
    ['Date de création', tag.date],
    ['Date cible',       tag.dateCible],
    ['Type de danger',   tag.type],
    ['Gravité',          tag.gravite],
    ['Zone',             tag.zone],
    ['Emplacement',      tag.emplacement],
    ['Description',      tag.description],
    ['Risque principal', tag.risque],
    ['Action urgente',   tag.action],
    ['Propositions',     tag.propositions],
    ['Statut',           tag.statut],
    ['Responsable',      tag.responsable]
  ];

  var rowsHtml = rows.filter(function (r) { return r[1]; }).map(function (r) {
    return '<tr>' +
      '<td style="padding:6px 10px;border-bottom:1px solid #e6eef2;color:#557082;font-size:12px;font-weight:600;width:160px;vertical-align:top">' + escHtml_(r[0]) + '</td>' +
      '<td style="padding:6px 10px;border-bottom:1px solid #e6eef2;color:#142532;font-size:13px;line-height:1.5">' + escHtml_(r[1]) + '</td>' +
    '</tr>';
  }).join('');

  var html =
    '<div style="font-family:Inter,Arial,sans-serif;max-width:640px;margin:0 auto;color:#142532">' +
      '<div style="background:linear-gradient(135deg,#09637e,#0da8d2);padding:16px 20px;border-radius:10px 10px 0 0;color:#fff">' +
        '<div style="font-size:11px;letter-spacing:.6px;opacity:.85">SOBCO 3 — HSE</div>' +
        '<div style="font-size:18px;font-weight:700;margin-top:2px">Nouveau TAG ' + escHtml_(tag.id) + '</div>' +
        '<div style="font-size:12px;opacity:.9;margin-top:2px">Une nouvelle situation dangereuse vous est attribuée.</div>' +
      '</div>' +
      '<div style="background:#fff;border:1px solid #e6eef2;border-top:none;padding:16px 20px;border-radius:0 0 10px 10px">' +
        (tag.gravite ?
          '<div style="display:inline-block;padding:4px 10px;border-radius:14px;background:' + sevColor + ';color:#fff;font-size:11px;font-weight:700;margin-bottom:12px">' +
            'GRAVITÉ : ' + escHtml_(tag.gravite) +
          '</div>' : '') +
        '<table style="width:100%;border-collapse:collapse">' + rowsHtml + '</table>' +
        '<div style="margin-top:14px;padding:10px 12px;background:#f0f6f9;border-left:3px solid #0a7a9a;color:#284e64;font-size:12px;line-height:1.6">' +
          'Bonjour ' + escHtml_(tag.responsable) + ',<br>' +
          'Un TAG vient d\'être créé et requiert votre intervention. ' +
          'Merci de prendre en charge l\'action urgente avant la date cible indiquée.' +
        '</div>' +
        '<div style="margin-top:14px;font-size:10px;color:#8aa1b0">' +
          'Email automatique — SOBCO 3 HSE Master Dashboard.' +
        '</div>' +
      '</div>' +
    '</div>';

  var opts = { htmlBody: html, name: 'SOBCO 3 - HSE' };
  if (TAG_EMAIL_CC) opts.cc = TAG_EMAIL_CC;

  MailApp.sendEmail(email, subject, '', opts);
}

// ── Trigger onChange installable ─────────────────────────────
// Se déclenche à chaque ajout/édition/insertion de ligne dans
// le classeur. On ne filtre pas trop fin : notifyNewTags() est
// idempotent (Script Properties + LockService).
function onSheetChange(e) {
  try {
    notifyNewTags();
  } catch (err) {
    Logger.log('onSheetChange error: ' + err.message);
  }
}

// ── Installation du trigger (à exécuter UNE seule fois) ──────
function installTagTrigger() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  // Supprimer les anciens triggers onSheetChange pour éviter les doublons
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === 'onSheetChange') {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
  ScriptApp.newTrigger('onSheetChange').forSpreadsheet(ss).onChange().create();
  Logger.log('Trigger onSheetChange installé sur "' + ss.getName() + '"');
}

// ── Désinstallation (utilitaire) ─────────────────────────────
function uninstallTagTrigger() {
  var triggers = ScriptApp.getProjectTriggers();
  var n = 0;
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === 'onSheetChange') {
      ScriptApp.deleteTrigger(triggers[i]); n++;
    }
  }
  Logger.log('Triggers supprimés : ' + n);
}

// ── Test manuel : rejouer l'envoi pour un TAG donné ──────────
// Usage : testNotifyTag('123')
function testNotifyTag(tagId) {
  if (!tagId) { Logger.log('Usage : testNotifyTag("N° du TAG")'); return; }
  var notified = loadNotifiedIds_();
  delete notified[String(tagId).trim()];
  saveNotifiedIds_(notified);
  notifyNewTags();
}

// ── Réinitialisation complète de l'historique des envois ─────
// (à utiliser avec précaution — re-notifie TOUS les TAGs)
function resetTagNotificationHistory() {
  PropertiesService.getScriptProperties().deleteProperty(TAG_NOTIFIED_KEY);
  Logger.log('Historique des notifications TAG réinitialisé.');
}
