// ── Point d'entrée HTTP ───────────────────────────────────────
function doGet(e) {
  var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : '';
  if (action === 'data') return ContentService
    .createTextOutput(JSON.stringify(getData()))
    .setMimeType(ContentService.MimeType.JSON);
  return HtmlService.createHtmlOutputFromFile('Index')
    .setTitle('SOBCO 3 – HSE Master Dashboard')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// ── PROXY IMAGE BASE64 ───────────────────────────────────────
// Appelé depuis le client via google.script.run
// Retourne "data:image/jpeg;base64,..." ou ''
// ════════════════════════════════════════════════════════════
function getTagPhotoForClient(fileId) {
  if (!fileId || fileId === '') return '';
  try {
    var file = DriveApp.getFileById(fileId);
    var blob = file.getBlob();
    var b64  = Utilities.base64Encode(blob.getBytes());
    var mime = blob.getContentType() || 'image/jpeg';
    return 'data:' + mime + ';base64,' + b64;
  } catch(e) {
    Logger.log('getTagPhotoForClient error: ' + e.message);
    return '';
  }
}

// ── RÉSOLUTION PHOTO : référence → fileId Drive ──────────────
// Stratégie : extraire le numéro de la référence (ex: "431")
// puis chercher dans le dossier Drive avec title contains "431"
// Évite totalement le préfixe "tags/" et les accents
// ════════════════════════════════════════════════════════════
function resolveTagFileId(refRaw) {
  if (!refRaw || refRaw.trim() === '') return '';
  try {
    // 1. Nettoyer le préfixe tags/ ou tag/
    var clean = String(refRaw).replace(/^tags?\//i, '').trim();

    // 2. Extraire le premier numéro rencontré (accepte "431.xxx", "photo_431", "ref-431.jpg", etc.)
    var numMatch = clean.match(/(\d+)/);

    if (numMatch) {
      var num = numMatch[1];
      var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
      // Recherche Drive par numéro dans le titre — fonctionne avec accents
      var files = folder.searchFiles('title contains "' + num + '"');
      if (files.hasNext()) {
        return files.next().getId();
      }
    }
  } catch(e) {
    Logger.log('resolveTagFileId error: ' + e.message + ' | ref: ' + refRaw);
  }
  return '';
}

// ── SCAN INITIAL : pré-charger les IDs de tous les TAGs ──────
// Appelé une seule fois dans getData() pour éviter N appels Drive
// Construit un cache { "431": { avant:"fileId", apres:"fileId" }, ... }
// La classification AVANT / APRÈS se fait par mot-clé dans le nom :
//   - "apres" / "après" / "after"  → APRÈS
//   - tout le reste (par défaut)   → AVANT
// ════════════════════════════════════════════════════════════
function buildDriveTagsCache() {
  var cache  = {};  // num → { any: fileId, avant: fileId, apres: fileId }
  var count  = 0;
  try {
    var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
    var files  = folder.getFiles();
    while (files.hasNext()) {
      var file = files.next();
      var name = file.getName();
      var id   = file.getId();
      count++;
      // Extraire le premier numéro rencontré : "431.Référence...", "photo_431.jpg", "ref-431" → "431"
      var m = name.match(/(\d+)/);
      if (m) {
        var num = m[1];
        if (!cache[num]) cache[num] = { any: '', avant: '', apres: '' };
        // Tout premier fichier rencontré pour ce numéro → "any"
        if (!cache[num].any) cache[num].any = id;
        // Détection APRÈS — accents normalisés
        var low = String(name).toLowerCase()
          .replace(/[éèêë]/g,'e').replace(/[àâä]/g,'a').replace(/[ùûü]/g,'u');
        var isApres = /(^|[^a-z])(apres|after|aft)([^a-z]|$)/.test(low);
        if (isApres) {
          if (!cache[num].apres) cache[num].apres = id;
        } else {
          if (!cache[num].avant) cache[num].avant = id;
        }
      }
    }
    Logger.log('Drive cache: ' + count + ' fichiers, ' + Object.keys(cache).length + ' numéros');
  } catch(e) {
    Logger.log('buildDriveTagsCache error: ' + e.message);
  }
  return { cache: cache, count: count };
}

// ── Utilitaires ───────────────────────────────────────────────
function isNull(v) {
  if (v === null || v === undefined) return true;
  var s = String(v).trim(); return s === '' || s === '/' || s === '-';
}
function toNum(v) {
  if (isNull(v)) return null;
  var n = parseFloat(String(v).replace(/\s/g,'').replace(/,/g,''));
  return isNaN(n) ? null : n;
}
function toNumZ(v) { var n = toNum(v); return n === null ? 0 : n; }

function parseDate(v) {
  if (isNull(v)) return null;
  if (v instanceof Date && !isNaN(v.getTime())) return v;
  var s = String(v).trim();
  var m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})/);
  if (m) {
    var d = parseInt(m[1]), mo = parseInt(m[2]), yr = parseInt(m[3]);
    if (d > 12)  return new Date(yr, mo-1, d);
    if (mo > 12) return new Date(yr, d-1, mo);
    return new Date(yr, mo-1, d);
  }
  try { var r = new Date(v); return isNaN(r.getTime()) ? null : r; } catch(e2) { return null; }
}
function fmtDate(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return String(v).trim();
  try { return Utilities.formatDate(d, 'GMT+1', 'dd/MM/yyyy'); } catch(e) { return String(v).trim(); }
}

// ────────────────────────────────────────────────────────────
// Conversion sûre d'une valeur de cellule en chaîne d'affichage.
// Empêche le format laid « Sun Dec 31 1899 02:24:21 GMT+0009 »
// que produit String(date) sur les Date objects renvoyés par
// SpreadsheetApp.getValues().
//   - Cellule TIME (origine 30/12/1899 → année < 1901)
//       → « HH:mm » (ou « HH:mm:ss » si secondes non nulles)
//   - Date réelle  → « dd/MM/yyyy »
//   - Autre        → String(v).trim()
// ────────────────────────────────────────────────────────────
function safeText(v) {
  if (v === null || v === undefined) return '';
  if (v instanceof Date && !isNaN(v.getTime())) {
    if (v.getFullYear() < 1901) {
      try {
        var s = Utilities.formatDate(v, 'GMT+1', 'HH:mm:ss');
        return s.replace(/:00$/, '');
      } catch(e) { return ''; }
    }
    try { return Utilities.formatDate(v, 'GMT+1', 'dd/MM/yyyy'); } catch(e2) { return ''; }
  }
  return String(v).trim();
}
function getIso(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return '';
  // Format ISO en heure locale (GMT+1, Algérie) — évite le décalage
  // d'un jour provoqué par toISOString() qui convertit en UTC.
  try { return Utilities.formatDate(d, 'GMT+1', 'yyyy-MM-dd'); } catch(e) { return ''; }
}
function isFumigation(a) { return a && a.toLowerCase().indexOf('fumig') > -1; }
function normalize(s) {
  if (!s) return '';
  return s.toLowerCase()
    .replace(/[àáâãäå]/g,'a').replace(/[èéêë]/g,'e')
    .replace(/[ìíîï]/g,'i').replace(/[òóôõö]/g,'o')
    .replace(/[ùúûü]/g,'u').replace(/[ç]/g,'c').replace(/[ñ]/g,'n')
    .replace(/\s+/g,' ').trim();
}

// ── Planning ─────────────────────────────────────────────────
function readPlanning(ss) {
  var planning = [];
  try {
    var sheets = ss.getSheets();
    var planSheet = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === PLANNING_GID) { planSheet = sheets[i]; break; }
    }
    if (!planSheet) {
      for (var i = 0; i < sheets.length; i++) {
        var n = sheets[i].getName().toLowerCase();
        if (n.indexOf('planning') > -1 || n.indexOf('activit') > -1 || n.indexOf('suivi') > -1) {
          planSheet = sheets[i]; break;
        }
      }
    }
    if (!planSheet) { Logger.log('Planning sheet NOT found'); return planning; }
    Logger.log('Planning sheet found: ' + planSheet.getName());
    var data = planSheet.getDataRange().getValues();
    var startRow = 1;
    for (var i = 0; i < Math.min(data.length, 10); i++) {
      var h0 = String(data[i][0]||'').toLowerCase();
      if (h0.indexOf('date') > -1) { startRow = i + 1; break; }
    }
    for (var i = startRow; i < data.length; i++) {
      var r = data[i];
      if (isNull(r[0])) continue;
      var pd = parseDate(r[0]);
      if (!pd || isNaN(pd.getTime())) continue;
      var act = isNull(r[1]) ? '' : String(r[1]).trim();
      if (!act) continue;
      planning.push({
        date: fmtDate(r[0]),
        // ISO en heure locale (GMT+1) pour éviter qu'un événement
        // stocké à minuit local ne se retrouve daté de la veille
        // après conversion UTC via toISOString().
        isoDate: Utilities.formatDate(pd, 'GMT+1', 'yyyy-MM-dd'),
        activite: act,
        details: isNull(r[2]) ? '' : String(r[2]).trim(),
        fumigation: isFumigation(act)
      });
    }
    Logger.log('Planning: ' + planning.length + ' entrées');
  } catch(e) { Logger.log('Planning error: ' + e.message); }
  return planning;
}

// ── CACHE PHOTOS NETTOYAGE ───────────────────────────────────
// Construit un index { numéro → { any, avant, apres } } pour le
// sous-dossier "Suivi de Nettoyage - Sobco 3_Images" du dossier
// parent CLEAN_PARENT_FOLDER_ID.
// Si le sous-dossier n'est pas trouvé, on scanne aussi le parent
// directement (et ses sous-dossiers en récursif, profondeur 2).
// La distinction AVANT / APRÈS suit la même règle que pour les
// TAGs : mot-clé apres / après / after dans le nom de fichier.
// ════════════════════════════════════════════════════════════
function buildCleaningPhotosCache() {
  var cache  = {};   // num → { any, avant, apres }
  var byName = {};   // nom de fichier (lowercased) → fileId
  var count  = 0;

  function scanFolder(folder, depth) {
    if (!folder || depth < 0) return;
    var files = folder.getFiles();
    while (files.hasNext()) {
      var file = files.next();
      var name = file.getName();
      var id   = file.getId();
      count++;

      // Index par nom (utile si la feuille référence le nom complet)
      byName[String(name).toLowerCase()] = id;

      // Extraire le premier numéro rencontré
      var m = name.match(/(\d+)/);
      if (m) {
        var num = m[1];
        if (!cache[num]) cache[num] = { any: '', avant: '', apres: '' };
        if (!cache[num].any) cache[num].any = id;
        var low = String(name).toLowerCase()
          .replace(/[éèêë]/g,'e').replace(/[àâä]/g,'a').replace(/[ùûü]/g,'u');
        var isApres = /(^|[^a-z])(apres|after|aft)([^a-z]|$)/.test(low);
        if (isApres) {
          if (!cache[num].apres) cache[num].apres = id;
        } else {
          if (!cache[num].avant) cache[num].avant = id;
        }
      }
    }
    if (depth > 0) {
      var subs = folder.getFolders();
      while (subs.hasNext()) scanFolder(subs.next(), depth - 1);
    }
  }

  try {
    var parent = DriveApp.getFolderById(CLEAN_PARENT_FOLDER_ID);

    // 1. Chercher le sous-dossier nommé "Suivi de Nettoyage - Sobco 3_Images"
    var targetFolder = null;
    var subs = parent.getFolders();
    while (subs.hasNext()) {
      var sf = subs.next();
      if (normalize(sf.getName()) === normalize(CLEAN_IMAGES_FOLDER_NAME)) {
        targetFolder = sf; break;
      }
    }
    if (targetFolder) {
      scanFolder(targetFolder, 1);
    } else {
      // Fallback : scanner le dossier parent + tous ses sous-dossiers
      Logger.log('Cleaning subfolder not found by name, scanning parent recursively');
      scanFolder(parent, 2);
    }
    Logger.log('Cleaning cache: ' + count + ' fichiers, ' + Object.keys(cache).length + ' numéros');
  } catch(e) {
    Logger.log('buildCleaningPhotosCache error: ' + e.message);
  }
  return { cache: cache, byName: byName, count: count };
}

// ── LECTURE FEUILLE "Nettoyage " ─────────────────────────────
// Détection automatique des colonnes depuis la ligne d'entête.
// Mappe ensuite chaque ligne sur un objet exposé au client.
// ════════════════════════════════════════════════════════════
function readCleaning(ss, photoCache) {
  var entries = [];
  try {
    // 1. Trouver la feuille (par GID puis par nom)
    var sheets = ss.getSheets();
    var sh = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === CLEAN_SHEET_GID) { sh = sheets[i]; break; }
    }
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('nettoyage') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('Cleaning sheet NOT found'); return entries; }
    Logger.log('Cleaning sheet found: ' + sh.getName());

    var data = sh.getDataRange().getValues();
    if (!data || data.length < 2) return entries;

    // 2. Détecter la ligne d'entête (première ligne contenant "date" ou "n°")
    var hdrRow = 0;
    for (var i = 0; i < Math.min(data.length, 6); i++) {
      var row = data[i];
      var foundKeys = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||''));
        if (h.indexOf('date') > -1 || h.indexOf('zone') > -1 ||
            h.indexOf('description') > -1 || h.indexOf('observation') > -1 ||
            h.indexOf('statut') > -1 || h.indexOf('responsable') > -1) {
          foundKeys++;
        }
      }
      if (foundKeys >= 2) { hdrRow = i; break; }
    }

    var hdr = data[hdrRow];
    var COL = {
      id:-1, date:-1, dateCible:-1, zone:-1, emplacement:-1,
      description:-1, observation:-1, action:-1, responsable:-1,
      statut:-1, photoAvant:-1, photoApres:-1, photoGen:-1, type:-1, frequence:-1
    };
    for (var c = 0; c < hdr.length; c++) {
      var h = normalize(String(hdr[c]||'').replace(/[\n\r#]/g,'').trim());
      if (!h) continue;

      // PHOTO APRÈS — tester en premier pour éviter le piège "photoRef"
      if ((h.indexOf('photo') > -1 || h.indexOf('image') > -1 || h.indexOf('reference') > -1) &&
          (h.indexOf('apres') > -1 || h.indexOf('after') > -1)) {
        if (COL.photoApres < 0) COL.photoApres = c;
        continue;
      }
      // PHOTO AVANT
      if ((h.indexOf('photo') > -1 || h.indexOf('image') > -1 || h.indexOf('reference') > -1) &&
          (h.indexOf('avant') > -1 || h.indexOf('before') > -1)) {
        if (COL.photoAvant < 0) COL.photoAvant = c;
        continue;
      }
      // PHOTO générique (sans précision avant/après)
      if (h.indexOf('photo') > -1 || h.indexOf('image') > -1) {
        if (COL.photoGen < 0) COL.photoGen = c;
        continue;
      }

      if      (h.match(/^n.?\s*(cas|tag|ordre|$)/) || h === 'n')    { if (COL.id < 0) COL.id = c; }
      else if (h.indexOf('date') > -1 && (h.indexOf('cible') > -1 || h.indexOf('echeance') > -1)) { if (COL.dateCible < 0) COL.dateCible = c; }
      else if (h.indexOf('date') > -1)                              { if (COL.date < 0) COL.date = c; }
      else if (h.indexOf('zone') > -1)                              { if (COL.zone < 0) COL.zone = c; }
      else if (h.indexOf('emplacement') > -1 || h.indexOf('lieu') > -1) { if (COL.emplacement < 0) COL.emplacement = c; }
      else if (h.indexOf('description') > -1)                       { if (COL.description < 0) COL.description = c; }
      else if (h.indexOf('observation') > -1 || h.indexOf('constat') > -1) { if (COL.observation < 0) COL.observation = c; }
      else if (h.indexOf('action') > -1 || h.indexOf('mesure') > -1 || h.indexOf('correctif') > -1) { if (COL.action < 0) COL.action = c; }
      else if (h.indexOf('respons') > -1 || h.indexOf('charg') > -1) { if (COL.responsable < 0) COL.responsable = c; }
      else if (h.indexOf('statut') > -1 || h.indexOf('etat') > -1)  { if (COL.statut < 0) COL.statut = c; }
      else if (h.indexOf('type') > -1 || h.indexOf('nature') > -1)  { if (COL.type < 0) COL.type = c; }
      else if (h.indexOf('frequen') > -1 || h.indexOf('periodicit') > -1) { if (COL.frequence < 0) COL.frequence = c; }
    }
    // Fallback : si on n'a pas trouvé d'ID, prendre la première colonne
    if (COL.id < 0) COL.id = 0;
    Logger.log('Cleaning COL: ' + JSON.stringify(COL));

    // 3. Helper : résoudre un fileId Drive depuis une référence photo
    function resolveCleanPhoto(refRaw, prefer) {
      if (isNull(refRaw)) return '';
      var ref = String(refRaw).trim();
      if (!ref) return '';
      // Recherche directe par nom de fichier
      var byNameKey = ref.toLowerCase();
      if (photoCache.byName[byNameKey]) return photoCache.byName[byNameKey];
      // Recherche tolérante (avec / sans extension)
      var keys = Object.keys(photoCache.byName);
      for (var k = 0; k < keys.length; k++) {
        if (keys[k].indexOf(byNameKey) > -1 || byNameKey.indexOf(keys[k]) > -1) {
          return photoCache.byName[keys[k]];
        }
      }
      // Recherche par numéro
      var m = ref.match(/(\d+)/);
      if (m && photoCache.cache[m[1]]) {
        var entry = photoCache.cache[m[1]];
        // Pour "après" : ne JAMAIS retomber sur un fichier avant/any.
        // Seul un fichier explicitement tagué "apres/after" est accepté.
        if (prefer === 'apres') return entry.apres || '';
        if (prefer === 'avant') return entry.avant || entry.any || '';
        return entry.any || entry.avant || '';
      }
      return '';
    }

    // 4. Parcourir les lignes de données
    for (var i = hdrRow + 1; i < data.length; i++) {
      var r = data[i];

      // Filtrer les lignes vides ou de type "total" / "sous-total"
      var idVal = (COL.id >= 0) ? r[COL.id] : '';
      var firstText = '';
      for (var c = 0; c < r.length; c++) {
        if (!isNull(r[c])) { firstText = String(r[c]).trim(); break; }
      }
      if (!firstText) continue;
      var lc = firstText.toLowerCase();
      if (lc.indexOf('total') === 0 || lc.indexOf('sous-total') > -1) continue;

      // Récupérer toutes les valeurs avec fallbacks
      function val(c) { return (c >= 0 && !isNull(r[c])) ? safeText(r[c]) : ''; }

      var id          = val(COL.id) || ('NET-' + (i+1));
      var dateRaw     = (COL.date >= 0) ? r[COL.date] : '';
      var dateCible   = (COL.dateCible >= 0) ? r[COL.dateCible] : '';
      var zone        = val(COL.zone);
      var emplacement = val(COL.emplacement);
      var description = val(COL.description);
      var observation = val(COL.observation);
      var action      = val(COL.action);
      var responsable = val(COL.responsable);
      var statut      = val(COL.statut);
      var type        = val(COL.type);
      var frequence   = val(COL.frequence);

      // Photos : 3 colonnes possibles
      var refAvant = val(COL.photoAvant) || val(COL.photoGen);
      var refApres = val(COL.photoApres);

      // RÈGLE STRICTE : on n'affiche une photo "après" que si la colonne
      // « photo après » de la feuille est explicitement renseignée.
      // Aucun fallback automatique sur le numéro de la photo "avant".
      var fileIdAvant = resolveCleanPhoto(refAvant, 'avant');
      var fileIdApres = refApres ? resolveCleanPhoto(refApres, 'apres') : '';

      // Sauter les lignes complètement vides (pas de zone, description, ni photo, ni statut)
      if (!zone && !emplacement && !description && !observation && !statut && !fileIdAvant && !fileIdApres && !val(COL.id)) continue;

      var pd = parseDate(dateRaw);
      entries.push({
        id:          String(id),
        date:        fmtDate(dateRaw),
        isoDate:     getIso(dateRaw),
        dateCible:   fmtDate(dateCible),
        zone:        zone,
        emplacement: emplacement,
        type:        type,
        frequence:   frequence,
        description: description,
        observation: observation,
        action:      action,
        responsable: responsable,
        statut:      statut,
        photoRefAvant:    refAvant,
        photoRefApres:    refApres,
        driveFileIdAvant: fileIdAvant,
        driveFileIdApres: fileIdApres,
        hasPhotoAvant:    fileIdAvant !== '',
        hasPhotoApres:    fileIdApres !== '',
        hasPhoto:         (fileIdAvant !== '' || fileIdApres !== '')
      });
    }

    // Tri : date décroissante puis ID décroissant
    entries.sort(function(a, b) {
      if (a.isoDate && b.isoDate && a.isoDate !== b.isoDate) return b.isoDate.localeCompare(a.isoDate);
      var na = parseInt(String(a.id).match(/\d+/), 10);
      var nb = parseInt(String(b.id).match(/\d+/), 10);
      if (isNaN(na) && isNaN(nb)) return String(b.id).localeCompare(String(a.id));
      if (isNaN(na)) return 1;
      if (isNaN(nb)) return -1;
      return nb - na;
    });

    Logger.log('Cleaning entries: ' + entries.length);
  } catch(e) {
    Logger.log('readCleaning error: ' + e.message);
  }
  return entries;
}

// ════════════════════════════════════════════════════════════
// ── LECTURE FEUILLE "nettoyage 1" — PLAN DE NETTOYAGE ────────
// Renvoie un objet { entries, products } :
//
// 1) entries  → un tableau de zones (Étage -1, 0, 1, Voie externe)
//    Pour chaque zone on extrait :
//      - link         : lien du PLAN de nettoyage
//      - productsLink : lien des produits utilisés (si une colonne
//                       Produits existe dans le tableau des zones)
//      - productsText : libellé/description des produits
//    Détection souple des colonnes du tableau ZONES :
//      ZONE          : entête contient "zone"
//      LIEN PLAN     : "lien" / "link" / "url" / "document" /
//                      "fichier" / "plan"  (mais PAS "produit")
//      PRODUITS      : "produit" / "product" / "consommable" /
//                      "materiel" / "fds" / "fiche" / "detergent"
//                      "désinfect" / "chimique" / "savon" / "outil"
//                      "équipement"
//    Si AUCUNE entête n'est détectée → A=zone, B=lien plan, C=produits.
//
// 2) products → un tableau global { name, link } extrait d'un
//    sous-tableau « PRODUITS UTILISÉS » présent ailleurs sur la
//    même feuille, avec colonnes typiquement « NOM DE PRODUIT »
//    + « LIEN DE FICHE » (FDS).  Détection tolérante : on accepte
//    toute paire (NOM/PRODUIT, LIEN/FICHE/FDS/DOCUMENT/URL).
//    Lecture jusqu'à rencontrer 2 lignes consécutives totalement
//    vides (= fin du sous-tableau).
//
// Lecture par formule (=HYPERLINK), RichText puis valeur brute
// afin de capturer tous les types de liens hypertexte.
// ════════════════════════════════════════════════════════════
function readCleaningPlan(ss) {
  var entries  = [];
  var products = [];
  try {
    var sheets = ss.getSheets();
    var sh = null;

    // 1. Cherche par GID
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === CLEAN_PLAN_SHEET_GID) { sh = sheets[i]; break; }
    }
    // 2. Sinon par nom exact "nettoyage 1"
    if (!sh) {
      var target = normalize(CLEAN_PLAN_SHEET_NAME);
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()) === target) { sh = sheets[i]; break; }
      }
    }
    // 3. Sinon premier onglet contenant "nettoyage" + "1"
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('nettoyage') > -1 && n.indexOf('1') > -1 && n.indexOf('10') < 0) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('Cleaning PLAN sheet NOT found'); return { entries: entries, products: products }; }
    Logger.log('Cleaning PLAN sheet found: ' + sh.getName());

    var range    = sh.getDataRange();
    var data     = range.getValues();
    var rich     = null;
    var formulas = null;
    try { rich     = range.getRichTextValues(); } catch(e0) { rich = null; }
    try { formulas = range.getFormulas();        } catch(eF) { formulas = null; }
    if (!data || data.length < 1) return { entries: entries, products: products };

    // ── Helper : extrait l'URL d'une cellule ─────────────────
    // Ordre : 1) formule =HYPERLINK("url",...)   2) RichText
    //         3) objet Drive (getUrl)            4) URL en clair
    function cellUrl(rIdx, cIdx) {
      // 1) Formule =HYPERLINK("url", "libellé")  — souvent invisible aux RichText
      if (formulas && formulas[rIdx] && formulas[rIdx][cIdx]) {
        var f = String(formulas[rIdx][cIdx] || '');
        if (f) {
          var mH = f.match(/=\s*HYPERLINK\s*\(\s*["']([^"']+)["']/i);
          if (mH) return mH[1].trim();
          // URL nue dans une formule générique
          var mU = f.match(/(https?:\/\/[^\s"'),]+)/i);
          if (mU) return mU[1].trim();
        }
      }
      // 2) RichText (capture les hyperlinks « masqués » dans le texte)
      if (rich && rich[rIdx] && rich[rIdx][cIdx]) {
        var rt = rich[rIdx][cIdx];
        try {
          var u = rt.getLinkUrl && rt.getLinkUrl();
          if (u) return String(u).trim();
          // Parcourir les "runs" pour trouver un lien
          var runs = rt.getRuns && rt.getRuns();
          if (runs && runs.length) {
            for (var k = 0; k < runs.length; k++) {
              var ru = runs[k].getLinkUrl && runs[k].getLinkUrl();
              if (ru) return String(ru).trim();
            }
          }
        } catch(eR) {}
      }
      // 3) Objet Drive natif
      var v = data[rIdx][cIdx];
      if (isNull(v)) return '';
      if (typeof v === 'object' && v && typeof v.getUrl === 'function') {
        try { return String(v.getUrl() || '').trim(); } catch(eU) {}
      }
      // 4) URL en clair dans la valeur brute
      var s = String(v).trim();
      if (/^https?:\/\//i.test(s)) return s;
      // Certaines cellules contiennent l'URL au milieu d'un texte
      var mIn = s.match(/(https?:\/\/[^\s,;]+)/i);
      if (mIn) return mIn[1].trim();
      return '';
    }
    function cellText(rIdx, cIdx) {
      var v = data[rIdx][cIdx];
      return isNull(v) ? '' : safeText(v);
    }
    // Vrai si la colonne c contient au moins une donnée sous la ligne hdr
    function colHasData(c, fromRow) {
      for (var r = fromRow; r < data.length; r++) {
        if (c < (data[r] || []).length && !isNull(data[r][c])) return true;
      }
      return false;
    }

    // ── Détecter la ligne d'entête + colonnes ──
    // Mots-clés ÉLARGIS pour la colonne « Produits utilisés » :
    //   produit / product / consommable / materiel / matériel
    //   fds / fiche (de données) / detergent / désinfectant / désinfection
    //   chimique / savon / outil / equipement / équipement
    var PROD_KW = ['produit','product','consommable','materiel',
                   'fds','fiche','detergent','desinfect','chimique',
                   'savon','outil','equipement'];
    function isProdHeader(h) {
      for (var k = 0; k < PROD_KW.length; k++) {
        if (h.indexOf(PROD_KW[k]) > -1) return true;
      }
      return false;
    }

    var hdrRow = -1, zoneCol = -1, linkCol = -1, prodCol = -1;
    for (var i = 0; i < Math.min(data.length, 6); i++) {
      var row = data[i];
      var zc = -1, lc = -1, pc = -1;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||''));
        if (!h) continue;
        // Produits — testé en premier pour éviter de matcher « lien produits » comme « lien »
        if (pc < 0 && isProdHeader(h)) { pc = c; continue; }
        if (zc < 0 && h.indexOf('zone') > -1) { zc = c; continue; }
        if (lc < 0 && (h.indexOf('lien') > -1 || h.indexOf('link') > -1 ||
                       h.indexOf('url') > -1  || h.indexOf('document') > -1 ||
                       h.indexOf('fichier') > -1 || h.indexOf('plan') > -1)) {
          lc = c; continue;
        }
      }
      if (zc >= 0 && (lc >= 0 || pc >= 0)) {
        hdrRow = i; zoneCol = zc; linkCol = lc; prodCol = pc; break;
      }
    }
    // Fallback total : pas d'entête détectée → A=zone, B=lien plan, C=produits
    if (hdrRow < 0) { hdrRow = -1; zoneCol = 0; linkCol = 1; prodCol = 2; }

    // Fallback ciblé : entête trouvée MAIS colonne Produits manquante.
    // On choisit la première colonne « libre » (≠ zoneCol, ≠ linkCol) qui
    // contient des données sous la ligne d'entête. Cela couvre les cas où
    // la colonne s'appelle « Détergents », « FDS », n'a pas d'entête, etc.
    if (prodCol < 0) {
      var startRow = (hdrRow >= 0) ? hdrRow + 1 : 0;
      var maxCol   = 0;
      for (var rr = 0; rr < data.length; rr++) {
        if ((data[rr] || []).length > maxCol) maxCol = data[rr].length;
      }
      // 1) Préférer une colonne avec un entête non vide
      for (var c = 0; c < maxCol; c++) {
        if (c === zoneCol || c === linkCol) continue;
        var hdrTxt = (hdrRow >= 0 && data[hdrRow]) ? String(data[hdrRow][c] || '').trim() : '';
        if (!hdrTxt) continue;
        if (colHasData(c, startRow)) { prodCol = c; break; }
      }
      // 2) Sinon n'importe quelle colonne libre avec des données
      if (prodCol < 0) {
        for (var c = 0; c < maxCol; c++) {
          if (c === zoneCol || c === linkCol) continue;
          if (colHasData(c, startRow)) { prodCol = c; break; }
        }
      }
      Logger.log('Cleaning PLAN: prodCol auto-fallback → ' + prodCol);
    }

    // Log diagnostic des entêtes détectées (utile en cas d'erreur "Aucune info produits")
    if (hdrRow >= 0 && data[hdrRow]) {
      Logger.log('Cleaning PLAN headers row[' + hdrRow + '] : ' +
                 JSON.stringify(data[hdrRow].map(function(x){ return String(x||''); })));
    }
    Logger.log('Cleaning PLAN cols : zone=' + zoneCol +
               ' link=' + linkCol + ' prod=' + prodCol + ' hdrRow=' + hdrRow);

    // ── Détection du sous-tableau « PRODUITS UTILISÉS » ───────
    // Cherche n'importe où sur la feuille (sauf la ligne d'entête
    // des zones) une paire d'entêtes (Nom de produit / Lien de fiche).
    // Renvoie { hdrRow, nameCol, linkCol } ou null.
    function findProductsTable(skipRow) {
      function classify(h) {
        if (!h) return '';
        if (h.indexOf('zone') > -1) return '';   // jamais le tableau des zones
        var hasNom  = h.indexOf('nom') > -1;
        var hasProd = h.indexOf('produit') > -1 || h.indexOf('product') > -1;
        var hasLink = h.indexOf('lien') > -1 || h.indexOf('link') > -1 ||
                      h.indexOf('url') > -1  || h.indexOf('fiche') > -1 ||
                      h.indexOf('fds') > -1  ||
                      h.indexOf('document') > -1 || h.indexOf('fichier') > -1;
        if (hasLink) return 'link';
        if ((hasNom && hasProd) || h === 'nom') return 'name';
        if (hasProd) return 'name';   // "Produit", "Produits"
        if (h.indexOf('designation') > -1 || h.indexOf('libelle') > -1 ||
            h.indexOf('intitule')   > -1) return 'name';
        return '';
      }
      for (var r = 0; r < data.length; r++) {
        if (r === skipRow) continue;
        var row = data[r] || [];
        var nc = -1, lc = -1;
        for (var c = 0; c < row.length; c++) {
          var h = normalize(String(row[c]||''));
          var t = classify(h);
          if      (t === 'name' && nc < 0) nc = c;
          else if (t === 'link' && lc < 0) lc = c;
        }
        if (nc >= 0 && lc >= 0 && nc !== lc) {
          return { hdrRow: r, nameCol: nc, linkCol: lc };
        }
      }
      return null;
    }
    var prodTbl = findProductsTable(hdrRow);
    if (prodTbl) {
      Logger.log('Cleaning PLAN sub-table « PRODUITS UTILISÉS » found at row ' +
                 prodTbl.hdrRow + ' (name=' + prodTbl.nameCol + ' link=' + prodTbl.linkCol + ')');
    } else {
      Logger.log('Cleaning PLAN sub-table « PRODUITS UTILISÉS » : not found');
    }

    // ── Lecture des lignes ZONES ─────────────────────────────
    // (on s'arrête au début du sous-tableau produits pour ne pas
    //  consommer ses lignes comme si c'étaient des zones)
    for (var i = hdrRow + 1; i < data.length; i++) {
      if (prodTbl && i >= prodTbl.hdrRow) break;
      var zone = cellText(i, zoneCol);
      if (!zone) continue;

      var link         = (linkCol >= 0) ? cellUrl(i, linkCol)  : '';
      var productsLink = (prodCol >= 0) ? cellUrl(i, prodCol)  : '';
      var productsText = (prodCol >= 0) ? cellText(i, prodCol) : '';
      // Si la cellule produits était une URL brute, productsText == productsLink → on vide le texte
      if (productsLink && productsText === productsLink) productsText = '';

      entries.push({
        zone:         zone,
        link:         link,
        productsLink: productsLink,
        productsText: productsText,
        key:          normalize(zone)
      });
    }
    Logger.log('Cleaning PLAN entries : ' + entries.length);

    // ── Lecture du sous-tableau PRODUITS UTILISÉS ────────────
    // Format attendu : NOM DE PRODUIT | LIEN DE FICHE (FDS).
    // S'arrête après 2 lignes consécutives totalement vides.
    if (prodTbl) {
      var blanks = 0;
      for (var i = prodTbl.hdrRow + 1; i < data.length; i++) {
        var pName    = cellText(i, prodTbl.nameCol);
        var pLink    = cellUrl(i, prodTbl.linkCol);
        var pLinkRaw = cellText(i, prodTbl.linkCol);
        if (!pName && !pLink && !pLinkRaw) {
          blanks++;
          if (blanks >= 2) break;
          continue;
        }
        blanks = 0;
        if (!pName) continue;            // ignore les lignes sans nom
        // Si une URL est dans le texte brut, prends-la
        if (!pLink && /^https?:\/\//i.test(pLinkRaw)) pLink = pLinkRaw;
        products.push({ name: pName, link: pLink || '', text: pLink ? '' : pLinkRaw });
      }
      Logger.log('Cleaning PLAN products : ' + products.length);
    }
  } catch(e) {
    Logger.log('readCleaningPlan error: ' + e.message);
  }
  return { entries: entries, products: products };
}

// ════════════════════════════════════════════════════════════
// ── LECTURE FEUILLE "procédures" ─────────────────────────────
// Renvoie un objet { entries, extras } :
//
//   entries → liste des procédures HSE :
//     [{ titre, reference, dateVersion, url }]
//   extras  → liens spéciaux trouvés dans la feuille (typiquement
//     « fiche d'identité processus HSE ») :
//     [{ label, url }]
//
// Détection souple des colonnes du tableau :
//   TITRE       : entête contient « titre » / « procédure » / « nom »
//   RÉFÉRENCE   : « réf » / « code »
//   DATE VERSION: « date » / « version »
// Si aucune entête détectée → A=titre, B=référence, C=date.
//
// Les hyperliens sont extraits dans cet ordre :
//   1) =HYPERLINK("url",…)
//   2) RichTextValues.getLinkUrl()
//   3) URL en clair dans la valeur brute
// ════════════════════════════════════════════════════════════
function readProcedures(ss) {
  var entries = [];
  var extras  = [];
  try {
    var sheets = ss.getSheets();
    var sh = null;

    // 1. Cherche par GID
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === PROCEDURES_SHEET_GID) { sh = sheets[i]; break; }
    }
    // 2. Sinon par nom exact "procédures"
    if (!sh) {
      var target = normalize(PROCEDURES_SHEET_NAME);
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()) === target) { sh = sheets[i]; break; }
      }
    }
    // 3. Sinon premier onglet contenant "procedure"
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('procedure') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('Procédures sheet NOT found'); return { entries: entries, extras: extras }; }
    Logger.log('Procédures sheet found: ' + sh.getName());

    var range    = sh.getDataRange();
    var data     = range.getValues();
    var rich     = null;
    var formulas = null;
    try { rich     = range.getRichTextValues(); } catch(e0) { rich = null; }
    try { formulas = range.getFormulas();        } catch(eF) { formulas = null; }
    if (!data || data.length < 1) return { entries: entries, extras: extras };

    // ── Helper : extrait l'URL d'une cellule ─────────────────
    function cellUrl(rIdx, cIdx) {
      // 1) Formule =HYPERLINK("url", "libellé")
      if (formulas && formulas[rIdx] && formulas[rIdx][cIdx]) {
        var f = String(formulas[rIdx][cIdx] || '');
        if (f) {
          var mH = f.match(/=\s*HYPERLINK\s*\(\s*["']([^"']+)["']/i);
          if (mH) return mH[1].trim();
          var mU = f.match(/(https?:\/\/[^\s"'),]+)/i);
          if (mU) return mU[1].trim();
        }
      }
      // 2) RichText
      if (rich && rich[rIdx] && rich[rIdx][cIdx]) {
        var rt = rich[rIdx][cIdx];
        try {
          var u = rt.getLinkUrl && rt.getLinkUrl();
          if (u) return String(u).trim();
          var runs = rt.getRuns && rt.getRuns();
          if (runs && runs.length) {
            for (var k = 0; k < runs.length; k++) {
              var ru = runs[k].getLinkUrl && runs[k].getLinkUrl();
              if (ru) return String(ru).trim();
            }
          }
        } catch(eR) {}
      }
      // 3) Objet Drive natif
      var v = data[rIdx][cIdx];
      if (isNull(v)) return '';
      if (typeof v === 'object' && v && typeof v.getUrl === 'function') {
        try { return String(v.getUrl() || '').trim(); } catch(eU) {}
      }
      // 4) URL en clair dans la valeur brute
      var s = String(v).trim();
      if (/^https?:\/\//i.test(s)) return s;
      var mIn = s.match(/(https?:\/\/[^\s,;]+)/i);
      if (mIn) return mIn[1].trim();
      return '';
    }
    function cellText(rIdx, cIdx) {
      if (cIdx < 0) return '';
      var v = data[rIdx][cIdx];
      return isNull(v) ? '' : safeText(v);
    }

    // ── Détecter la ligne d'entête + colonnes ──
    var hdrRow = -1, titleCol = -1, refCol = -1, dateCol = -1;
    for (var i = 0; i < Math.min(data.length, 6); i++) {
      var row = data[i];
      var tc = -1, rc = -1, dc = -1;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||''));
        if (!h) continue;
        if (tc < 0 && (h.indexOf('titre') > -1 || h.indexOf('procedure') > -1 ||
                       h.indexOf('intitul') > -1 || h.indexOf('nom') > -1)) { tc = c; continue; }
        if (rc < 0 && (h.indexOf('reference') > -1 || h.indexOf('ref') === 0 ||
                       h.indexOf('code') > -1)) { rc = c; continue; }
        if (dc < 0 && (h.indexOf('date') > -1 || h.indexOf('version') > -1)) { dc = c; continue; }
      }
      if (tc >= 0 && (rc >= 0 || dc >= 0)) {
        hdrRow = i; titleCol = tc; refCol = rc; dateCol = dc; break;
      }
    }
    // Fallback : pas d'entête détectée → A=titre, B=référence, C=date
    if (hdrRow < 0) { hdrRow = -1; titleCol = 0; refCol = 1; dateCol = 2; }
    Logger.log('Procédures cols: titre=' + titleCol + ' ref=' + refCol + ' date=' + dateCol);

    // ── Parcourir les lignes de données ──
    var firstDataRow = (hdrRow >= 0) ? hdrRow + 1 : 0;
    for (var i = firstDataRow; i < data.length; i++) {
      var titre = cellText(i, titleCol);
      var ref   = cellText(i, refCol);
      var dv    = cellText(i, dateCol);

      // ── Détection d'un lien spécial sur la ligne (fiche d'identité processus HSE) ──
      var lineText = '';
      var rowLen   = (data[i] || []).length;
      for (var c = 0; c < rowLen; c++) {
        var v = data[i][c]; if (!isNull(v)) lineText += ' ' + String(v);
      }
      var lt = normalize(lineText);
      var isFiche = (lt.indexOf('fiche') > -1) &&
                    (lt.indexOf('processus') > -1 || lt.indexOf('identite') > -1);
      if (isFiche) {
        var fLabel = '', fUrl = '';
        for (var c = 0; c < rowLen; c++) {
          var t = cellText(i, c);
          var u = cellUrl(i, c);
          if (!fLabel && t) fLabel = t;
          if (!fUrl   && u) fUrl   = u;
          if (fLabel && fUrl) break;
        }
        extras.push({
          label: fLabel || "Fiche d'identité processus HSE",
          url:   fUrl   || ''
        });
        continue;
      }

      // Sauter les lignes vides
      if (!titre && !ref && !dv) continue;

      // Sauter une éventuelle ré-impression de l'entête
      var lcTitre = normalize(titre);
      if (lcTitre.indexOf('titre de la procedure') > -1 ||
          lcTitre === 'procedure' || lcTitre === 'titre') continue;

      // URL associée — on cherche d'abord sur la cellule du titre,
      // puis sur la cellule de la référence (souvent porteuse aussi)
      var url = cellUrl(i, titleCol);
      if (!url && refCol >= 0) url = cellUrl(i, refCol);

      entries.push({
        titre:       titre,
        reference:   ref,
        dateVersion: dv,
        url:         url
      });
    }

    Logger.log('Procédures: ' + entries.length + ' entrées, ' + extras.length + ' extras');
  } catch(e) {
    Logger.log('readProcedures error: ' + e.message);
  }
  return { entries: entries, extras: extras };
}

// ════════════════════════════════════════════════════════════
// ── LECTURE FEUILLE "Suivi des Contrats et Liens Drive" ──────
// Renvoie un tableau de contrats :
//   [{ id, type, intitule, tiers, dateDebut, dateFin, statut,
//      url, commentaire, raw[] }]
//
// Détection tolérante des colonnes via mots-clés :
//   ID / N°       : « n° » / « numero » / « id » / « ref »
//   TYPE          : « type » / « categorie » / « nature » / « objet »
//                   (« objet » sert de fallback intitulé si rien d'autre)
//   INTITULÉ      : « intitule » / « libelle » / « designation » /
//                   « titre » / « contrat » / « description »
//   TIERS         : « tiers » / « fournisseur » / « client » /
//                   « partenaire » / « societe » / « entreprise » /
//                   « prestataire » / « organisme »
//   DATE DÉBUT    : « debut » / « signature » / « effet » / « date d »
//   DATE FIN      : « fin » / « echeance » / « expiration » /
//                   « validite » / « renouvellement »
//   STATUT        : « statut » / « etat »
//   LIEN          : « lien » / « link » / « url » / « document » /
//                   « fichier » / « drive » / « pdf »
//   COMMENTAIRE   : « commentaire » / « remarque » / « note » /
//                   « observation »
//
// Les hyperliens sont extraits dans cet ordre :
//   1) =HYPERLINK("url",…)
//   2) RichTextValues.getLinkUrl()
//   3) Objet Drive natif (getUrl)
//   4) URL en clair dans la valeur brute
// Si aucune colonne « lien » dédiée n'est trouvée, on scanne TOUTES
// les cellules de la ligne et on retient la première URL rencontrée.
// ════════════════════════════════════════════════════════════
// ════════════════════════════════════════════════════════════
// ── LECTURE FEUILLE EPI ──────────────────────────────────────
// Renvoie un objet :
//   { entries:[{...}], headers:[...], sheetName:'...' }
//
// Chaque entry contient :
//   - id, designation, categorie, taille, unite, qty, qtyMin,
//     dateRecep, dateExp, statut, responsable, photoRef, url,
//     commentaire, raw:[…toutes les colonnes…], stockState:'ok|low|out|expired|na'
// Les hyperliens sont extraits dans cet ordre :
//   1) =HYPERLINK("url",…)  2) RichText  3) Drive natif  4) URL en clair
// ════════════════════════════════════════════════════════════
function readEPI(ss) {
  var entries = [];
  var rawHeaders = [];
  var sheetName  = '';
  try {
    var sheets = ss.getSheets();
    var sh = null;

    // 1. Cherche par GID
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === EPI_SHEET_GID) { sh = sheets[i]; break; }
    }
    // 2. Sinon par nom exact
    if (!sh) {
      var target = normalize(EPI_SHEET_NAME);
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()) === target) { sh = sheets[i]; break; }
      }
    }
    // 3. Sinon premier onglet contenant "epi"
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('epi') > -1 ||
            n.indexOf('protection individuelle') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('EPI sheet NOT found'); return { entries: entries, headers: rawHeaders, sheetName: '' }; }
    sheetName = sh.getName();
    Logger.log('EPI sheet found: ' + sheetName);

    var range    = sh.getDataRange();
    var data     = range.getValues();
    var rich     = null;
    var formulas = null;
    try { rich     = range.getRichTextValues(); } catch(e0) { rich = null; }
    try { formulas = range.getFormulas();        } catch(eF) { formulas = null; }
    if (!data || data.length < 1) return { entries: entries, headers: rawHeaders, sheetName: sheetName };

    // ── Helpers — extraction URL / texte d'une cellule ──
    function cellUrl(rIdx, cIdx) {
      if (cIdx < 0) return '';
      if (formulas && formulas[rIdx] && formulas[rIdx][cIdx]) {
        var f = String(formulas[rIdx][cIdx] || '');
        if (f) {
          var mH = f.match(/=\s*HYPERLINK\s*\(\s*["']([^"']+)["']/i);
          if (mH) return mH[1].trim();
          var mU = f.match(/(https?:\/\/[^\s"'),]+)/i);
          if (mU) return mU[1].trim();
        }
      }
      if (rich && rich[rIdx] && rich[rIdx][cIdx]) {
        var rt = rich[rIdx][cIdx];
        try {
          var u = rt.getLinkUrl && rt.getLinkUrl();
          if (u) return String(u).trim();
          var runs = rt.getRuns && rt.getRuns();
          if (runs && runs.length) {
            for (var k = 0; k < runs.length; k++) {
              var ru = runs[k].getLinkUrl && runs[k].getLinkUrl();
              if (ru) return String(ru).trim();
            }
          }
        } catch(eR) {}
      }
      var v = data[rIdx][cIdx];
      if (isNull(v)) return '';
      if (typeof v === 'object' && v && typeof v.getUrl === 'function') {
        try { return String(v.getUrl() || '').trim(); } catch(eU) {}
      }
      var s = String(v).trim();
      if (/^https?:\/\//i.test(s)) return s;
      var mIn = s.match(/(https?:\/\/[^\s,;]+)/i);
      if (mIn) return mIn[1].trim();
      return '';
    }
    function cellText(rIdx, cIdx) {
      if (cIdx < 0) return '';
      var v = data[rIdx][cIdx];
      return isNull(v) ? '' : safeText(v);
    }
    function anyUrlOnRow(rIdx) {
      var rowLen = (data[rIdx] || []).length;
      for (var c = 0; c < rowLen; c++) {
        var u = cellUrl(rIdx, c); if (u) return u;
      }
      return '';
    }

    // ── Détection ligne d'entête + colonnes ──
    var COL = { id:-1, designation:-1, categorie:-1, taille:-1, unite:-1,
                qty:-1, qtyMin:-1, dateRecep:-1, dateExp:-1,
                statut:-1, responsable:-1, photoRef:-1, lien:-1,
                commentaire:-1 };
    var hdrRow = -1;
    for (var i = 0; i < Math.min(data.length, 8); i++) {
      var row = data[i];
      var cur = { id:-1, designation:-1, categorie:-1, taille:-1, unite:-1,
                  qty:-1, qtyMin:-1, dateRecep:-1, dateExp:-1,
                  statut:-1, responsable:-1, photoRef:-1, lien:-1,
                  commentaire:-1 };
      var hits = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||'').replace(/[\n\r#]/g,'').trim());
        if (!h) continue;

        // Lien (testé tôt pour ne pas être volé par d'autres mots-clés)
        if (cur.lien < 0 &&
            (h.indexOf('lien') > -1 || h.indexOf('link') > -1 ||
             h.indexOf('url') > -1  || h.indexOf('drive') > -1 ||
             h.indexOf('document') > -1 || h.indexOf('fichier') > -1 ||
             h.indexOf('fiche') > -1)) {
          cur.lien = c; hits++; continue;
        }
        // Photo / référence photo
        if (cur.photoRef < 0 &&
            ((h.indexOf('photo') > -1) ||
             (h.indexOf('reference') > -1 && h.indexOf('photo') > -1) ||
             h.indexOf('image') > -1)) {
          cur.photoRef = c; hits++; continue;
        }
        // Stock minimum / seuil d'alerte
        if (cur.qtyMin < 0 &&
            ((h.indexOf('stock') > -1 && (h.indexOf('min') > -1 || h.indexOf('seuil') > -1 || h.indexOf('alert') > -1)) ||
             h.indexOf('seuil') > -1 || h.indexOf('alerte') > -1 ||
             h.indexOf('minimum') > -1)) {
          cur.qtyMin = c; hits++; continue;
        }
        // Stock / quantité actuelle
        if (cur.qty < 0 &&
            (h.indexOf('stock') > -1 || h.indexOf('quantite') > -1 ||
             h.indexOf('qte') > -1 || h === 'qty' || h.indexOf('disponible') > -1)) {
          cur.qty = c; hits++; continue;
        }
        // Date de réception
        if (cur.dateRecep < 0 &&
            (h.indexOf('date') > -1 && (h.indexOf('recep') > -1 ||
                                         h.indexOf('entree') > -1 ||
                                         h.indexOf('achat') > -1 ||
                                         h.indexOf('livraison') > -1))) {
          cur.dateRecep = c; hits++; continue;
        }
        // Date d'expiration / péremption
        if (cur.dateExp < 0 &&
            ((h.indexOf('date') > -1 && (h.indexOf('expir') > -1 || h.indexOf('peremp') > -1 ||
                                          h.indexOf('validite') > -1 || h.indexOf('fin') > -1)) ||
             h.indexOf('expiration') > -1 || h.indexOf('peremption') > -1)) {
          cur.dateExp = c; hits++; continue;
        }
        // Statut / état
        if (cur.statut < 0 && (h.indexOf('statut') > -1 || h.indexOf('etat') > -1)) {
          cur.statut = c; hits++; continue;
        }
        // Responsable / affecté à
        if (cur.responsable < 0 &&
            (h.indexOf('respons') > -1 || h.indexOf('charg') > -1 ||
             h.indexOf('affect') > -1 || h.indexOf('attribut') > -1 ||
             h.indexOf('beneficiaire') > -1 || h.indexOf('porteur') > -1 ||
             h.indexOf('utilisateur') > -1)) {
          cur.responsable = c; hits++; continue;
        }
        // Catégorie / type / famille
        if (cur.categorie < 0 &&
            (h.indexOf('categorie') > -1 || h.indexOf('famille') > -1 ||
             h.indexOf('type') > -1 || h.indexOf('classe') > -1 ||
             h.indexOf('groupe') > -1)) {
          cur.categorie = c; hits++; continue;
        }
        // Taille / pointure
        if (cur.taille < 0 &&
            (h.indexOf('taille') > -1 || h.indexOf('size') > -1 ||
             h.indexOf('pointure') > -1 || h.indexOf('dimension') > -1)) {
          cur.taille = c; hits++; continue;
        }
        // Unité
        if (cur.unite < 0 &&
            (h === 'unite' || h.indexOf('u.m') > -1 ||
             h.indexOf('mesure') > -1)) {
          cur.unite = c; hits++; continue;
        }
        // ID / N°
        if (cur.id < 0 && (h.match(/^n.?\s*$/) || h === 'no' || h === 'num' ||
            h.indexOf('numero') > -1 || h === 'id' ||
            (h.indexOf('ref') > -1 && h.indexOf('reference') < 0 && h.length < 10))) {
          cur.id = c; hits++; continue;
        }
        // Désignation / intitulé / nom de l'EPI
        if (cur.designation < 0 &&
            (h.indexOf('designation') > -1 ||
             h.indexOf('libelle') > -1 ||
             h.indexOf('intitule') > -1 ||
             h.indexOf('nom') > -1 || h.indexOf('article') > -1 ||
             h.indexOf('produit') > -1 || h.indexOf('epi') > -1 ||
             h.indexOf('equipement') > -1)) {
          cur.designation = c; hits++; continue;
        }
        // Commentaire / observation / remarque
        if (cur.commentaire < 0 &&
            (h.indexOf('commentaire') > -1 || h.indexOf('remarque') > -1 ||
             h.indexOf('observation') > -1 || h.indexOf('note') > -1)) {
          cur.commentaire = c; hits++; continue;
        }
      }
      if (hits >= 2) { hdrRow = i; COL = cur; break; }
    }

    // Fallback total : pas d'entête détectée → mapping positionnel
    if (hdrRow < 0) {
      hdrRow = 0;
      COL.id          = 0;
      COL.designation = 1;
      COL.categorie   = 2;
      COL.taille      = 3;
      COL.qty         = 4;
      COL.qtyMin      = 5;
      COL.dateRecep   = 6;
      COL.dateExp     = 7;
      COL.statut      = 8;
      COL.responsable = 9;
    }
    Logger.log('EPI COL: ' + JSON.stringify(COL) + ' hdrRow=' + hdrRow);

    // Capturer les entêtes brutes pour l'affichage générique côté client
    if (hdrRow >= 0 && data[hdrRow]) {
      for (var c = 0; c < data[hdrRow].length; c++) {
        rawHeaders.push(String(data[hdrRow][c] || '').trim());
      }
      Logger.log('EPI headers: ' + JSON.stringify(rawHeaders));
    }

    // ── Parcourir les lignes de données ──
    var firstDataRow = (hdrRow >= 0) ? hdrRow + 1 : 0;
    for (var i = firstDataRow; i < data.length; i++) {
      var row    = data[i] || [];
      var rowLen = row.length;

      // Skip lignes vides
      var anyVal = false;
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { anyVal = true; break; }
      }
      if (!anyVal) continue;

      var idVal       = cellText(i, COL.id);
      var desigVal    = cellText(i, COL.designation);
      var catVal      = cellText(i, COL.categorie);
      var tailleVal   = cellText(i, COL.taille);
      var uniteVal    = cellText(i, COL.unite);
      var statutVal   = cellText(i, COL.statut);
      var respVal     = cellText(i, COL.responsable);
      var photoRef    = cellText(i, COL.photoRef);
      var commentVal  = cellText(i, COL.commentaire);
      var qtyNum      = (COL.qty    >= 0) ? toNum(row[COL.qty])    : null;
      var qtyMinNum   = (COL.qtyMin >= 0) ? toNum(row[COL.qtyMin]) : null;
      var dateRecepR  = (COL.dateRecep >= 0) ? row[COL.dateRecep] : '';
      var dateExpR    = (COL.dateExp   >= 0) ? row[COL.dateExp]   : '';

      // URL : colonne dédiée d'abord, puis désignation, puis n'importe où
      var url = cellUrl(i, COL.lien);
      if (!url) url = cellUrl(i, COL.designation);
      if (!url) url = anyUrlOnRow(i);

      // Sauter ré-impression de l'entête
      var firstTxt = '';
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { firstTxt = String(row[c]).trim(); break; }
      }
      var lcFirst = normalize(firstTxt);
      if (lcFirst === 'n' || lcFirst === 'no' || lcFirst === 'num' ||
          lcFirst === 'id' || lcFirst.indexOf('total') === 0 ||
          lcFirst.indexOf('designation') === 0) continue;

      // ── État du stock : ok / low / out / expired ──
      var stockState = 'na';
      if (qtyNum !== null) {
        if (qtyNum <= 0) stockState = 'out';
        else if (qtyMinNum !== null && qtyNum <= qtyMinNum) stockState = 'low';
        else stockState = 'ok';
      }
      // Override par statut explicite
      var lcStat = String(statutVal || '').toLowerCase();
      if (lcStat.indexOf('rupture') > -1 || lcStat.indexOf('epuis') > -1) stockState = 'out';
      else if (lcStat.indexOf('faible') > -1 || lcStat.indexOf('alerte') > -1 || lcStat.indexOf('critique') > -1) stockState = 'low';
      // Date d'expiration dépassée → expired
      var expIso = getIso(dateExpR);
      if (expIso) {
        var todayStr = Utilities.formatDate(new Date(), 'GMT+1', 'yyyy-MM-dd');
        if (expIso < todayStr) stockState = 'expired';
      }

      // Capturer la ligne brute pour affichage générique côté client
      var rawCells = [];
      for (var c = 0; c < rowLen; c++) {
        rawCells.push(safeText(row[c]));
      }

      entries.push({
        id:           idVal,
        designation:  desigVal,
        categorie:    catVal,
        taille:       tailleVal,
        unite:        uniteVal,
        qty:          qtyNum,
        qtyMin:       qtyMinNum,
        dateRecep:    fmtDate(dateRecepR),
        isoDateRecep: getIso(dateRecepR),
        dateExp:      fmtDate(dateExpR),
        isoDateExp:   expIso,
        statut:       statutVal,
        responsable:  respVal,
        photoRef:     photoRef,
        commentaire:  commentVal,
        url:          url,
        stockState:   stockState,
        raw:          rawCells
      });
    }

    Logger.log('EPI: ' + entries.length + ' entrees');
  } catch(e) {
    Logger.log('readEPI error: ' + e.message);
  }
  return { entries: entries, headers: rawHeaders, sheetName: sheetName };
}

function readContracts(ss) {
  var entries = [];
  try {
    var sheets = ss.getSheets();
    var sh = null;

    // 1. Cherche par GID
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === CONTRACTS_SHEET_GID) { sh = sheets[i]; break; }
    }
    // 2. Sinon par nom exact
    if (!sh) {
      var target = normalize(CONTRACTS_SHEET_NAME);
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()) === target) { sh = sheets[i]; break; }
      }
    }
    // 3. Sinon premier onglet contenant "contrat"
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('contrat') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) { Logger.log('Contrats sheet NOT found'); return entries; }
    Logger.log('Contrats sheet found: ' + sh.getName());

    var range    = sh.getDataRange();
    var data     = range.getValues();
    var rich     = null;
    var formulas = null;
    try { rich     = range.getRichTextValues(); } catch(e0) { rich = null; }
    try { formulas = range.getFormulas();        } catch(eF) { formulas = null; }
    if (!data || data.length < 1) return entries;

    // ── Helper : extrait l'URL d'une cellule ─────────────────
    function cellUrl(rIdx, cIdx) {
      if (cIdx < 0) return '';
      // 1) Formule =HYPERLINK("url", "libellé")
      if (formulas && formulas[rIdx] && formulas[rIdx][cIdx]) {
        var f = String(formulas[rIdx][cIdx] || '');
        if (f) {
          var mH = f.match(/=\s*HYPERLINK\s*\(\s*["']([^"']+)["']/i);
          if (mH) return mH[1].trim();
          var mU = f.match(/(https?:\/\/[^\s"'),]+)/i);
          if (mU) return mU[1].trim();
        }
      }
      // 2) RichText
      if (rich && rich[rIdx] && rich[rIdx][cIdx]) {
        var rt = rich[rIdx][cIdx];
        try {
          var u = rt.getLinkUrl && rt.getLinkUrl();
          if (u) return String(u).trim();
          var runs = rt.getRuns && rt.getRuns();
          if (runs && runs.length) {
            for (var k = 0; k < runs.length; k++) {
              var ru = runs[k].getLinkUrl && runs[k].getLinkUrl();
              if (ru) return String(ru).trim();
            }
          }
        } catch(eR) {}
      }
      // 3) Objet Drive natif
      var v = data[rIdx][cIdx];
      if (isNull(v)) return '';
      if (typeof v === 'object' && v && typeof v.getUrl === 'function') {
        try { return String(v.getUrl() || '').trim(); } catch(eU) {}
      }
      // 4) URL en clair dans la valeur brute
      var s = String(v).trim();
      if (/^https?:\/\//i.test(s)) return s;
      var mIn = s.match(/(https?:\/\/[^\s,;]+)/i);
      if (mIn) return mIn[1].trim();
      return '';
    }
    function cellText(rIdx, cIdx) {
      if (cIdx < 0) return '';
      var v = data[rIdx][cIdx];
      return isNull(v) ? '' : safeText(v);
    }
    // Cherche une URL n'importe où sur la ligne (toutes colonnes)
    function anyUrlOnRow(rIdx, exceptCol) {
      var rowLen = (data[rIdx] || []).length;
      for (var c = 0; c < rowLen; c++) {
        if (c === exceptCol) continue;
        var u = cellUrl(rIdx, c);
        if (u) return u;
      }
      return '';
    }

    // ── Détecter la ligne d'entête + colonnes ──
    var COL = { id:-1, type:-1, intitule:-1, tiers:-1, dateDebut:-1,
                dateFin:-1, statut:-1, lien:-1, commentaire:-1 };
    var hdrRow = -1;
    for (var i = 0; i < Math.min(data.length, 8); i++) {
      var row = data[i];
      var cur = { id:-1, type:-1, intitule:-1, tiers:-1, dateDebut:-1,
                  dateFin:-1, statut:-1, lien:-1, commentaire:-1 };
      var hits = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||'').replace(/[\n\r#]/g,'').trim());
        if (!h) continue;

        // Lien (testé tôt pour ne pas être volé par "objet" / autres)
        if (cur.lien < 0 &&
            (h.indexOf('lien') > -1 || h.indexOf('link') > -1 ||
             h.indexOf('url') > -1  || h.indexOf('drive') > -1 ||
             h.indexOf('document') > -1 || h.indexOf('fichier') > -1 ||
             h.indexOf('pdf') > -1 || h.indexOf('contrat scan') > -1)) {
          cur.lien = c; hits++; continue;
        }
        // Commentaire / observation
        if (cur.commentaire < 0 &&
            (h.indexOf('commentaire') > -1 || h.indexOf('remarque') > -1 ||
             h.indexOf('observation') > -1 || h.indexOf('note') > -1)) {
          cur.commentaire = c; hits++; continue;
        }
        // Dates : début vs fin (échéance / validité / expiration / fin)
        if (cur.dateFin < 0 &&
            (h.indexOf('fin')        > -1 || h.indexOf('echeance')      > -1 ||
             h.indexOf('expiration') > -1 || h.indexOf('validite')      > -1 ||
             h.indexOf('renouvellement') > -1)) {
          cur.dateFin = c; hits++; continue;
        }
        if (cur.dateDebut < 0 &&
            (h.indexOf('debut')     > -1 || h.indexOf('signature') > -1 ||
             h.indexOf('effet')     > -1 || h === 'date' ||
             h.indexOf('date de')   > -1 || h.indexOf('date du')   > -1)) {
          cur.dateDebut = c; hits++; continue;
        }
        // Statut
        if (cur.statut < 0 && (h.indexOf('statut') > -1 || h.indexOf('etat') > -1)) {
          cur.statut = c; hits++; continue;
        }
        // ID / N°
        if (cur.id < 0 && (h.match(/^n.?\s*$/) || h === 'no' || h === 'num' ||
            h.indexOf('numero') > -1 || h === 'id' ||
            (h.indexOf('ref') > -1 && h.indexOf('reference') < 0 && h.length < 10))) {
          cur.id = c; hits++; continue;
        }
        // Tiers / partenaire / fournisseur / etc.
        if (cur.tiers < 0 &&
            (h.indexOf('tiers') > -1 || h.indexOf('fournisseur') > -1 ||
             h.indexOf('client') > -1 || h.indexOf('partenaire') > -1 ||
             h.indexOf('societe') > -1 || h.indexOf('entreprise') > -1 ||
             h.indexOf('prestataire') > -1 || h.indexOf('organisme') > -1 ||
             h.indexOf('contractant') > -1)) {
          cur.tiers = c; hits++; continue;
        }
        // Type / catégorie / nature
        if (cur.type < 0 &&
            (h.indexOf('type') > -1 || h.indexOf('categorie') > -1 ||
             h.indexOf('nature') > -1)) {
          cur.type = c; hits++; continue;
        }
        // Intitulé / objet / désignation / titre / description
        if (cur.intitule < 0 &&
            (h.indexOf('intitule') > -1 || h.indexOf('libelle') > -1 ||
             h.indexOf('designation') > -1 || h.indexOf('titre') > -1 ||
             h.indexOf('objet') > -1 || h.indexOf('description') > -1 ||
             h.indexOf('contrat') > -1)) {
          cur.intitule = c; hits++; continue;
        }
      }
      // Heuristique : on accepte cette ligne comme entête si on a au moins
      // 2 colonnes reconnues (typiquement : intitulé + lien, ou tiers + date)
      if (hits >= 2) { hdrRow = i; COL = cur; break; }
    }

    // Fallback total : pas d'entête détectée → mapping positionnel raisonnable
    //   A=ID  B=intitulé  C=tiers  D=dateDébut  E=dateFin  F=statut  G=lien
    if (hdrRow < 0) {
      hdrRow = -1;
      COL.id        = 0;
      COL.intitule  = 1;
      COL.tiers     = 2;
      COL.dateDebut = 3;
      COL.dateFin   = 4;
      COL.statut    = 5;
      COL.lien      = 6;
    }
    Logger.log('Contrats COL: ' + JSON.stringify(COL) + ' hdrRow=' + hdrRow);
    if (hdrRow >= 0 && data[hdrRow]) {
      Logger.log('Contrats headers: ' +
        JSON.stringify(data[hdrRow].map(function(x){ return String(x||''); })));
    }

    // ── Parcourir les lignes de données ──
    var firstDataRow = (hdrRow >= 0) ? hdrRow + 1 : 0;
    for (var i = firstDataRow; i < data.length; i++) {
      var row    = data[i] || [];
      var rowLen = row.length;

      // Skip lignes totalement vides
      var anyVal = false;
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { anyVal = true; break; }
      }
      if (!anyVal) continue;

      var idVal       = cellText(i, COL.id);
      var typeVal     = cellText(i, COL.type);
      var intituleVal = cellText(i, COL.intitule);
      var tiersVal    = cellText(i, COL.tiers);
      var dateDebutR  = (COL.dateDebut >= 0) ? row[COL.dateDebut] : '';
      var dateFinR    = (COL.dateFin   >= 0) ? row[COL.dateFin]   : '';
      var statutVal   = cellText(i, COL.statut);
      var commentVal  = cellText(i, COL.commentaire);

      // URL : colonne dédiée d'abord, puis intitulé, puis n'importe où sur la ligne
      var url = cellUrl(i, COL.lien);
      if (!url) url = cellUrl(i, COL.intitule);
      if (!url) url = anyUrlOnRow(i, -1);

      // Sauter la ré-impression de l'entête (« contrat », « lien », etc.)
      var firstTxt = '';
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { firstTxt = String(row[c]).trim(); break; }
      }
      var lcFirst = normalize(firstTxt);
      if (lcFirst === 'n' || lcFirst === 'no' || lcFirst === 'num' ||
          lcFirst === 'id' || lcFirst.indexOf('total') === 0 ||
          (lcFirst.indexOf('contrat') > -1 && !intituleVal && !tiersVal && !url)) {
        continue;
      }

      // Sauter lignes informatives sans contenu utile
      if (!idVal && !intituleVal && !tiersVal && !url && !typeVal &&
          !statutVal && !commentVal && isNull(dateDebutR) && isNull(dateFinR)) {
        continue;
      }

      // Sécurité : si on n'a vraiment AUCUN texte signifiant ET aucune URL,
      // on saute (évite les lignes décoratives / fusionnées).
      var hasSomething = idVal || intituleVal || tiersVal || url ||
                         typeVal || statutVal || commentVal ||
                         !isNull(dateDebutR) || !isNull(dateFinR);
      if (!hasSomething) continue;

      // Capturer aussi la ligne brute pour affichage générique côté client
      // (utile si la feuille a des colonnes qu'on n'a pas mappées).
      var rawCells = [];
      for (var c = 0; c < rowLen; c++) {
        rawCells.push(safeText(row[c]));
      }

      entries.push({
        id:          idVal,
        type:        typeVal,
        intitule:    intituleVal,
        tiers:       tiersVal,
        dateDebut:   fmtDate(dateDebutR),
        isoDateDebut:getIso(dateDebutR),
        dateFin:     fmtDate(dateFinR),
        isoDateFin:  getIso(dateFinR),
        statut:      statutVal,
        commentaire: commentVal,
        url:         url,
        raw:         rawCells
      });
    }

    // Capturer aussi les entêtes brutes pour un affichage générique
    // côté client si l'utilisateur veut voir les colonnes non mappées.
    var rawHeaders = [];
    if (hdrRow >= 0 && data[hdrRow]) {
      for (var c = 0; c < data[hdrRow].length; c++) {
        rawHeaders.push(String(data[hdrRow][c] || '').trim());
      }
    }

    Logger.log('Contrats: ' + entries.length + ' entrées');
    return { entries: entries, headers: rawHeaders, sheetName: sh.getName() };
  } catch(e) {
    Logger.log('readContracts error: ' + e.message);
    return { entries: entries, headers: [], sheetName: '' };
  }
}

// ════════════════════════════════════════════════════════════
// ── NUISIBLES (4 feuilles) ──────────────────────────────────
// Lecteur générique pour les feuilles « Nuisibles » :
//   • FK                  → Insectes volants
//   • boites à raticide   → Rongeurs / Voie extérieure
//   • pièges mécanique    → Rongeurs / Intérieur usine
//   • boites à colles     → Rongeurs / Intérieur usine
//
// La structure exacte de ces feuilles n'est pas figée — on lit
// donc TOUT en mode générique :
//   1) Détection souple de la feuille (par mots-clés normalisés)
//   2) Détection dynamique de la ligne d'entête
//   3) Classification automatique des colonnes :
//        date, zone/emplacement, statut, action, numérique
//        (consommation, vérifications, etc.), texte
//   4) Capture des cellules brutes + entêtes pour rendu côté client
// Cela permet d'afficher tableaux, KPI et graphiques sans
// dépendre d'un schéma précis des colonnes.
// ════════════════════════════════════════════════════════════
function readPestSheet(ss, kwds, label) {
  var entries    = [];
  var rawHeaders = [];
  var sheetName  = '';
  var meta       = {
    dateCols:   [],   // index des colonnes contenant une date
    zoneCols:   [],   // emplacement / zone / numéro piège
    statutCols: [],   // statut / état / conformité
    actionCols: [],   // action / mesure
    numCols:    [],   // colonnes purement numériques (compteurs, captures…)
    primaryDate:   -1,
    primaryZone:   -1,
    primaryStatut: -1,
    primaryAction: -1
  };
  try {
    var sheets = ss.getSheets();
    var sh     = null;
    // Recherche par mots-clés normalisés
    for (var i = 0; i < sheets.length; i++) {
      var n = normalize(sheets[i].getName());
      for (var k = 0; k < kwds.length; k++) {
        if (n.indexOf(kwds[k]) > -1) { sh = sheets[i]; break; }
      }
      if (sh) break;
    }
    if (!sh) {
      Logger.log('[Nuisibles] feuille NON trouvée pour ' + label + ' / kwds=' + JSON.stringify(kwds));
      return { entries: entries, headers: rawHeaders, sheetName: '', meta: meta };
    }
    sheetName = sh.getName();
    Logger.log('[Nuisibles] ' + label + ' → feuille trouvée : ' + sheetName);

    var range    = sh.getDataRange();
    var data     = range.getValues();
    if (!data || data.length < 1) {
      return { entries: entries, headers: rawHeaders, sheetName: sheetName, meta: meta };
    }

    // ── Détection de la ligne d'entête ────────────────────
    // Heuristique : on prend la première ligne où ≥ 2 cellules
    // contiennent un mot-clé reconnu (date / zone / statut /
    // action / quantité / numéro / type).
    var hdrKeys = [
      'date', 'zone', 'emplace', 'lieu', 'secteur', 'piege', 'piège',
      'numero', 'numéro', 'n°', 'ref',
      'statut', 'etat', 'état', 'conform',
      'action', 'mesure', 'observ', 'remarque', 'commentaire',
      'quantit', 'nombre', 'capture', 'consom',
      'type', 'produit', 'app', 'verif', 'vérif',
      'responsable', 'controle', 'contrôle'
    ];
    var hdrRow = -1;
    for (var i = 0; i < Math.min(data.length, 12); i++) {
      var row  = data[i] || [];
      var hits = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||'').replace(/[\n\r#]/g,'').trim());
        if (!h) continue;
        for (var kk = 0; kk < hdrKeys.length; kk++) {
          if (h.indexOf(hdrKeys[kk]) > -1) { hits++; break; }
        }
      }
      if (hits >= 2) { hdrRow = i; break; }
    }
    if (hdrRow < 0) hdrRow = 0;

    // Capturer les entêtes brutes
    var hdrRowVals = data[hdrRow] || [];
    for (var c = 0; c < hdrRowVals.length; c++) {
      rawHeaders.push(String(hdrRowVals[c] || '').trim());
    }

    // ── Classification automatique des colonnes ───────────
    var ncols = rawHeaders.length;
    for (var c = 0; c < ncols; c++) {
      var hh = normalize(rawHeaders[c].replace(/[\n\r#]/g,'').trim());
      if (!hh) continue;
      // Date
      if (hh.indexOf('date') > -1) {
        meta.dateCols.push(c);
        if (meta.primaryDate < 0) meta.primaryDate = c;
      }
      // Zone / emplacement / piège
      if (meta.primaryZone < 0 &&
          (hh.indexOf('zone') > -1 || hh.indexOf('emplace') > -1 ||
           hh.indexOf('lieu')  > -1 || hh.indexOf('secteur') > -1 ||
           hh.indexOf('piege') > -1 || hh.indexOf('boite')  > -1)) {
        meta.zoneCols.push(c);
        meta.primaryZone = c;
      } else if ((hh.indexOf('zone') > -1 || hh.indexOf('emplace') > -1 ||
                  hh.indexOf('secteur') > -1) && meta.zoneCols.indexOf(c) < 0) {
        meta.zoneCols.push(c);
      }
      // Statut / conformité
      if (hh.indexOf('statut') > -1 || hh.indexOf('etat') > -1 ||
          hh.indexOf('conform') > -1) {
        meta.statutCols.push(c);
        if (meta.primaryStatut < 0) meta.primaryStatut = c;
      }
      // Action / mesure / observation
      if (hh.indexOf('action') > -1 || hh.indexOf('mesure') > -1 ||
          hh.indexOf('observ') > -1 || hh.indexOf('remarque') > -1 ||
          hh.indexOf('commentaire') > -1) {
        meta.actionCols.push(c);
        if (meta.primaryAction < 0) meta.primaryAction = c;
      }
    }

    // ── Parcourir les lignes de données ───────────────────
    var firstDataRow = hdrRow + 1;
    var numHits = {}; // colIdx → count of numeric cells

    for (var i = firstDataRow; i < data.length; i++) {
      var row    = data[i] || [];
      var rowLen = row.length;

      // Skip lignes complètement vides
      var anyVal = false;
      for (var c2 = 0; c2 < rowLen; c2++) {
        if (!isNull(row[c2])) { anyVal = true; break; }
      }
      if (!anyVal) continue;

      // Sauter ré-impression de l'entête
      var firstTxt = '';
      for (var c3 = 0; c3 < rowLen; c3++) {
        if (!isNull(row[c3])) { firstTxt = String(row[c3]).trim(); break; }
      }
      var lcFirst = normalize(firstTxt);
      if (lcFirst === 'date' || lcFirst === 'zone' || lcFirst === 'n' ||
          lcFirst === 'no'   || lcFirst === 'num'  || lcFirst === 'id' ||
          lcFirst.indexOf('total') === 0) continue;

      // Capturer les cellules brutes + détecter les colonnes numériques
      var rawCells = [];
      var dateRaw = (meta.primaryDate >= 0) ? row[meta.primaryDate] : '';
      for (var c4 = 0; c4 < rowLen; c4++) {
        rawCells.push(safeText(row[c4]));
        var num = toNum(row[c4]);
        if (num !== null && !isNull(row[c4])) {
          // Exclure les colonnes manifestement « date » ou « zone »
          if (meta.dateCols.indexOf(c4) < 0 && meta.zoneCols.indexOf(c4) < 0) {
            numHits[c4] = (numHits[c4] || 0) + 1;
          }
        }
      }

      var entry = {
        raw:        rawCells,
        date:       fmtDate(dateRaw),
        isoDate:    getIso(dateRaw),
        zone:       (meta.primaryZone   >= 0) ? safeText(row[meta.primaryZone])   : '',
        statut:     (meta.primaryStatut >= 0) ? safeText(row[meta.primaryStatut]) : '',
        action:     (meta.primaryAction >= 0) ? safeText(row[meta.primaryAction]) : '',
        rowIdx:     i + 1
      };
      entries.push(entry);
    }

    // Finaliser numCols : colonnes avec >= 2 valeurs numériques
    Object.keys(numHits).forEach(function(k){
      if (numHits[k] >= 2) meta.numCols.push(parseInt(k, 10));
    });
    meta.numCols.sort(function(a,b){ return a - b; });

    Logger.log('[Nuisibles] ' + label + ' : ' + entries.length + ' entrées · meta=' + JSON.stringify(meta));
  } catch(e) {
    Logger.log('readPestSheet error (' + label + '): ' + e.message);
  }
  return { entries: entries, headers: rawHeaders, sheetName: sheetName, meta: meta };
}

function readPests(ss) {
  var fk = readPestSheet(
    ss,
    ['fk', 'flying', 'insecte volant', 'insectes volants'],
    'FK / Insectes volants'
  );
  var rat = readPestSheet(
    ss,
    ['boites a raticide', 'boite a raticide', 'raticide'],
    'Boîtes à raticide'
  );
  var mech = readPestSheet(
    ss,
    ['pieges mecanique', 'piege mecanique', 'pieges mecaniques', 'mecanique'],
    'Pièges mécanique'
  );
  var glue = readPestSheet(
    ss,
    ['boites a colles', 'boite a colle', 'boites a colle', 'boite a colles', 'colle'],
    'Boîtes à colles'
  );
  return {
    fk:   fk,
    rat:  rat,
    mech: mech,
    glue: glue
  };
}

// ════════════════════════════════════════════════════════════
// ── LECTURE FEUILLE "Gestion des Permis de Travail" ──────────
// Renvoie un objet :
//   { entries:[{...}], headers:[...], sheetName:'...' }
//
// Chaque entry contient :
//   - id (référence)
//   - titre, type, statut
//   - dateDebut, dateFin (validité)  + isoDateDebut, isoDateFin
//   - demandePar, interventions
//   - zone (alias de "interventions" — pour les graphiques par zone)
//   - photoRef         : valeur brute de la cellule photo
//   - photoUrl         : URL exploitable (Drive / http) si trouvée
//   - hasPhoto         : booléen
//   - month, year      : index/année déduits de la date de début
//   - state            : 'active' | 'pending' | 'closed' | 'expired'
//   - daysLeft         : jours restants avant expiration (si actif)
//   - raw[]            : toutes les colonnes brutes
//
// Détection tolérante des colonnes :
//   RÉFÉRENCE / ID    : « ref » / « n° » / « numero » / « id »
//   TITRE             : « titre » / « libelle » / « designation » /
//                       « intitule » / « objet » / « description »
//   TYPE              : « type » / « categorie » / « nature »
//   STATUT            : « statut » / « etat »
//   DATE DÉBUT        : « valide a partir » / « debut » / « delivrance » /
//                       « emis » / « date du » / « partir »
//   DATE FIN          : « valide jusqu » / « fin » / « echeance » /
//                       « expiration » / « validite » / « jusqu »
//   DEMANDÉ PAR       : « demande » / « demandeur » / « par »
//   INTERVENTIONS     : « intervention » / « lieu » / « emplacement » /
//                       « zone » / « localisation » / « atelier »
//   PHOTO             : « photo » / « image » / « piece jointe » /
//                       « justificatif » / « lien » / « document »
//
// Les hyperliens sont extraits dans cet ordre :
//   1) =HYPERLINK("url",…)
//   2) RichTextValues.getLinkUrl()
//   3) Objet Drive natif (getUrl)
//   4) URL en clair dans la valeur brute
// ════════════════════════════════════════════════════════════
function readPermits(ss) {
  var entries    = [];
  var rawHeaders = [];
  var sheetName  = '';
  try {
    var sheets = ss.getSheets();
    var sh = null;

    // 1. Cherche par GID
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === PERMITS_SHEET_GID) { sh = sheets[i]; break; }
    }
    // 2. Sinon par nom exact
    if (!sh) {
      var target = normalize(PERMITS_SHEET_NAME);
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()) === target) { sh = sheets[i]; break; }
      }
    }
    // 3. Sinon premier onglet contenant "permis"
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        var n = normalize(sheets[i].getName());
        if (n.indexOf('permis') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) {
      Logger.log('Permis sheet NOT found');
      return { entries: entries, headers: rawHeaders, sheetName: '' };
    }
    sheetName = sh.getName();
    Logger.log('Permis sheet found: ' + sheetName);

    var range    = sh.getDataRange();
    var data     = range.getValues();
    var rich     = null;
    var formulas = null;
    try { rich     = range.getRichTextValues(); } catch(e0) { rich = null; }
    try { formulas = range.getFormulas();        } catch(eF) { formulas = null; }
    if (!data || data.length < 1) {
      return { entries: entries, headers: rawHeaders, sheetName: sheetName };
    }

    // ── Helper : extrait l'URL d'une cellule ─────────────────
    function cellUrl(rIdx, cIdx) {
      if (cIdx < 0) return '';
      // 1) Formule =HYPERLINK("url", "libellé")
      if (formulas && formulas[rIdx] && formulas[rIdx][cIdx]) {
        var f = String(formulas[rIdx][cIdx] || '');
        if (f) {
          var mH = f.match(/=\s*HYPERLINK\s*\(\s*["']([^"']+)["']/i);
          if (mH) return mH[1].trim();
          var mU = f.match(/(https?:\/\/[^\s"'),]+)/i);
          if (mU) return mU[1].trim();
        }
      }
      // 2) RichText
      if (rich && rich[rIdx] && rich[rIdx][cIdx]) {
        var rt = rich[rIdx][cIdx];
        try {
          var u = rt.getLinkUrl && rt.getLinkUrl();
          if (u) return String(u).trim();
          var runs = rt.getRuns && rt.getRuns();
          if (runs && runs.length) {
            for (var k = 0; k < runs.length; k++) {
              var ru = runs[k].getLinkUrl && runs[k].getLinkUrl();
              if (ru) return String(ru).trim();
            }
          }
        } catch(eR) {}
      }
      // 3) Objet Drive natif
      var v = data[rIdx][cIdx];
      if (isNull(v)) return '';
      if (typeof v === 'object' && v && typeof v.getUrl === 'function') {
        try { return String(v.getUrl() || '').trim(); } catch(eU) {}
      }
      // 4) URL en clair dans la valeur brute
      var s = String(v).trim();
      if (/^https?:\/\//i.test(s)) return s;
      var mIn = s.match(/(https?:\/\/[^\s,;]+)/i);
      if (mIn) return mIn[1].trim();
      return '';
    }
    function cellText(rIdx, cIdx) {
      if (cIdx < 0) return '';
      var v = data[rIdx][cIdx];
      return isNull(v) ? '' : safeText(v);
    }
    // Cherche une URL n'importe où sur la ligne (toutes colonnes)
    function anyUrlOnRow(rIdx, exceptCol) {
      var rowLen = (data[rIdx] || []).length;
      for (var c = 0; c < rowLen; c++) {
        if (c === exceptCol) continue;
        var u = cellUrl(rIdx, c);
        if (u) return u;
      }
      return '';
    }

    // ── Détecter la ligne d'entête + colonnes ──
    var COL = { id:-1, titre:-1, type:-1, statut:-1, dateDebut:-1,
                dateFin:-1, demandePar:-1, interventions:-1, photo:-1 };
    var hdrRow = -1;
    for (var i = 0; i < Math.min(data.length, 8); i++) {
      var row = data[i];
      var cur = { id:-1, titre:-1, type:-1, statut:-1, dateDebut:-1,
                  dateFin:-1, demandePar:-1, interventions:-1, photo:-1 };
      var hits = 0;
      for (var c = 0; c < row.length; c++) {
        var h = normalize(String(row[c]||'').replace(/[\n\r#]/g,'').trim());
        if (!h) continue;

        // PHOTO / Pièce jointe / lien (testé tôt)
        if (cur.photo < 0 &&
            (h.indexOf('photo') > -1 || h.indexOf('image') > -1 ||
             h.indexOf('piece jointe') > -1 || h.indexOf('pj') === 0 ||
             h.indexOf('justificatif') > -1 || h.indexOf('document') > -1 ||
             h.indexOf('fichier') > -1 ||
             (h.indexOf('lien') > -1 && h.indexOf('jusqu') < 0) ||
             h.indexOf('url') > -1 || h.indexOf('drive') > -1)) {
          cur.photo = c; hits++; continue;
        }
        // DATE FIN (testé avant date début pour éviter de voler "valide à partir du")
        if (cur.dateFin < 0 &&
            (h.indexOf('valide jusqu') > -1 || h.indexOf('jusqu') > -1 ||
             h.indexOf('echeance')      > -1 || h.indexOf('expiration') > -1 ||
             h.indexOf('fin')           > -1 || h.indexOf('validite')   > -1)) {
          cur.dateFin = c; hits++; continue;
        }
        // DATE DÉBUT
        if (cur.dateDebut < 0 &&
            (h.indexOf('valide a partir') > -1 || h.indexOf('a partir') > -1 ||
             h.indexOf('partir du')       > -1 || h.indexOf('debut')    > -1 ||
             h.indexOf('emis')            > -1 || h.indexOf('delivrance') > -1 ||
             h.indexOf('signature')       > -1 || h === 'date' ||
             h.indexOf('date de')         > -1 || h.indexOf('date du')  > -1)) {
          cur.dateDebut = c; hits++; continue;
        }
        // STATUT
        if (cur.statut < 0 && (h.indexOf('statut') > -1 || h.indexOf('etat') > -1)) {
          cur.statut = c; hits++; continue;
        }
        // RÉFÉRENCE / ID
        if (cur.id < 0 &&
            (h.indexOf('reference') > -1 || h.match(/^n.?\s*$/) ||
             h === 'no' || h === 'num' || h.indexOf('numero') > -1 ||
             h === 'id' ||
             (h.indexOf('ref') > -1 && h.length < 12))) {
          cur.id = c; hits++; continue;
        }
        // DEMANDÉ PAR
        if (cur.demandePar < 0 &&
            (h.indexOf('demande par') > -1 || h.indexOf('demandeur') > -1 ||
             h.indexOf('demande') > -1 || h === 'par' || h.indexOf('emis par') > -1 ||
             h.indexOf('responsable') > -1 || h.indexOf('signataire') > -1)) {
          cur.demandePar = c; hits++; continue;
        }
        // INTERVENTIONS / Zone / Lieu
        if (cur.interventions < 0 &&
            (h.indexOf('intervention') > -1 || h.indexOf('lieu') > -1 ||
             h.indexOf('emplace') > -1 || h.indexOf('zone') > -1 ||
             h.indexOf('localisation') > -1 || h.indexOf('atelier') > -1 ||
             h.indexOf('secteur') > -1 || h.indexOf('site') > -1)) {
          cur.interventions = c; hits++; continue;
        }
        // TYPE
        if (cur.type < 0 &&
            (h.indexOf('type') > -1 || h.indexOf('categorie') > -1 ||
             h.indexOf('nature') > -1)) {
          cur.type = c; hits++; continue;
        }
        // TITRE / Intitulé / Objet
        if (cur.titre < 0 &&
            (h.indexOf('titre') > -1 || h.indexOf('libelle') > -1 ||
             h.indexOf('designation') > -1 || h.indexOf('intitule') > -1 ||
             h.indexOf('objet') > -1 || h.indexOf('description') > -1 ||
             h.indexOf('travaux') > -1)) {
          cur.titre = c; hits++; continue;
        }
      }
      if (hits >= 2) { hdrRow = i; COL = cur; break; }
    }

    // Fallback : pas d'entête détectée → mapping positionnel basé sur la
    // structure observée :
    //   A=Référence  B=Titre  C=Type  D=Statut  E=Valide à partir
    //   F=Valide jusqu'au  G=Demandé par  H=Interventions  I=Photo
    if (hdrRow < 0) {
      hdrRow            = -1;
      COL.id            = 0;
      COL.titre         = 1;
      COL.type          = 2;
      COL.statut        = 3;
      COL.dateDebut     = 4;
      COL.dateFin       = 5;
      COL.demandePar    = 6;
      COL.interventions = 7;
      COL.photo         = 8;
    }
    Logger.log('Permis COL: ' + JSON.stringify(COL) + ' hdrRow=' + hdrRow);
    if (hdrRow >= 0 && data[hdrRow]) {
      for (var c = 0; c < data[hdrRow].length; c++) {
        rawHeaders.push(String(data[hdrRow][c] || '').trim());
      }
      Logger.log('Permis headers: ' + JSON.stringify(rawHeaders));
    }

    // Aujourd'hui (heure locale GMT+1) — pour calculer l'état
    var todayIso = Utilities.formatDate(new Date(), 'GMT+1', 'yyyy-MM-dd');

    // ── Parcourir les lignes de données ──
    var firstDataRow = (hdrRow >= 0) ? hdrRow + 1 : 0;
    for (var i = firstDataRow; i < data.length; i++) {
      var row    = data[i] || [];
      var rowLen = row.length;

      // Skip lignes totalement vides
      var anyVal = false;
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { anyVal = true; break; }
      }
      if (!anyVal) continue;

      var idVal       = cellText(i, COL.id);
      var titreVal    = cellText(i, COL.titre);
      var typeVal     = cellText(i, COL.type);
      var statutVal   = cellText(i, COL.statut);
      var dateDebutR  = (COL.dateDebut >= 0) ? row[COL.dateDebut] : '';
      var dateFinR    = (COL.dateFin   >= 0) ? row[COL.dateFin]   : '';
      var demandeVal  = cellText(i, COL.demandePar);
      var intervVal   = cellText(i, COL.interventions);
      var photoRef    = cellText(i, COL.photo);

      // Sauter ré-impression de l'entête
      var firstTxt = '';
      for (var c = 0; c < rowLen; c++) {
        if (!isNull(row[c])) { firstTxt = String(row[c]).trim(); break; }
      }
      var lcFirst = normalize(firstTxt);
      if (lcFirst === 'reference' || lcFirst === 'ref' ||
          lcFirst === 'n' || lcFirst === 'no' || lcFirst === 'num' ||
          lcFirst === 'id' || lcFirst.indexOf('total') === 0) {
        continue;
      }

      // Sauter lignes informatives sans contenu utile
      if (!idVal && !titreVal && !typeVal && !statutVal &&
          !demandeVal && !intervVal && !photoRef &&
          isNull(dateDebutR) && isNull(dateFinR)) {
        continue;
      }

      // URL Photo : colonne dédiée d'abord, puis n'importe où sur la ligne
      var photoUrl = cellUrl(i, COL.photo);
      if (!photoUrl) photoUrl = cellUrl(i, COL.titre);
      if (!photoUrl) photoUrl = anyUrlOnRow(i, -1);

      // Capturer toutes les colonnes brutes
      var rawCells = [];
      for (var c = 0; c < rowLen; c++) {
        rawCells.push(safeText(row[c]));
      }

      var isoDeb = getIso(dateDebutR);
      var isoFin = getIso(dateFinR);

      // Mois / année (basé sur dateDebut, sinon dateFin)
      var monthIdx = -1, yearVal = '';
      var refDate = parseDate(dateDebutR) || parseDate(dateFinR);
      if (refDate) {
        monthIdx = refDate.getMonth();
        yearVal  = String(refDate.getFullYear());
      }

      // ── Calcul de l'état (state) ────────────────────────
      // Combine le champ statut explicite ET les dates de validité.
      var state = '';
      var lcStat = String(statutVal).toLowerCase();
      if (lcStat.indexOf('expir') > -1 || lcStat.indexOf('échu') > -1 ||
          lcStat.indexOf('echu')  > -1) {
        state = 'expired';
      } else if (lcStat.indexOf('cloture') > -1 || lcStat.indexOf('clôtur') > -1 ||
                 lcStat.indexOf('termin')  > -1 || lcStat.indexOf('ferm')   > -1 ||
                 lcStat.indexOf('clos')    > -1) {
        state = 'closed';
      } else if (lcStat.indexOf('attente') > -1 || lcStat.indexOf('approbation') > -1 ||
                 lcStat.indexOf('pending') > -1) {
        state = 'pending';
      } else if (lcStat.indexOf('actif')    > -1 || lcStat.indexOf('en cours') > -1 ||
                 lcStat.indexOf('ouvert')   > -1 || lcStat.indexOf('open')     > -1 ||
                 lcStat.indexOf('approuv')  > -1 || lcStat.indexOf('signe')    > -1 ||
                 lcStat.indexOf('signé')    > -1 || lcStat.indexOf('valid')    > -1) {
        state = 'active';
      }
      // Override par la date de fin si on a une date d'échéance dépassée
      if (isoFin && isoFin < todayIso && state !== 'closed') {
        state = 'expired';
      }
      // Si pas de statut explicite, déduire depuis les dates
      if (!state) {
        if (isoFin && isoFin < todayIso) state = 'expired';
        else if (isoDeb && isoDeb > todayIso) state = 'pending';
        else state = 'active';
      }
      // Jours restants si encore actif
      var daysLeft = null;
      if (isoFin && state === 'active') {
        var t  = new Date(todayIso + 'T00:00:00');
        var f  = new Date(isoFin   + 'T00:00:00');
        daysLeft = Math.round((f - t) / 86400000);
      }

      entries.push({
        id:             idVal,
        titre:          titreVal,
        type:           typeVal,
        statut:         statutVal,
        dateDebut:      fmtDate(dateDebutR),
        isoDateDebut:   isoDeb,
        dateFin:        fmtDate(dateFinR),
        isoDateFin:     isoFin,
        demandePar:     demandeVal,
        interventions:  intervVal,
        zone:           intervVal,
        photoRef:       photoRef,
        photoUrl:       photoUrl,
        hasPhoto:       !!photoUrl,
        month:          monthIdx,
        year:           yearVal,
        daysLeft:       daysLeft,
        state:          state,
        raw:            rawCells
      });
    }

    // Tri : plus récent en haut (date début décroissante puis ID)
    entries.sort(function(a, b) {
      var ad = a.isoDateDebut || '';
      var bd = b.isoDateDebut || '';
      if (ad && bd && ad !== bd) return bd.localeCompare(ad);
      var na = parseInt(String(a.id).match(/\d+/), 10);
      var nb = parseInt(String(b.id).match(/\d+/), 10);
      if (isNaN(na) && isNaN(nb)) return String(b.id).localeCompare(String(a.id));
      if (isNaN(na)) return 1;
      if (isNaN(nb)) return -1;
      return nb - na;
    });

    Logger.log('Permis: ' + entries.length + ' entrées');
    return { entries: entries, headers: rawHeaders, sheetName: sheetName };
  } catch(e) {
    Logger.log('readPermits error: ' + e.message);
    return { entries: entries, headers: rawHeaders, sheetName: sheetName };
  }
}

    for (var k = 0; k < kwds.length; k++)
      for (var i = 0; i < keys.length; i++)
        if (keys[i].indexOf(kwds[k]) > -1) return byName[keys[i]];

  }

    ts: new Date().toISOString(),

    cleaningPlan:[],

    procedures:[],
    proceduresExtras:[],
    contracts:[],
    contractsHeaders:[],
    contractsSheetName:'',
    contractsSourceUrl: 'https://docs.google.com/spreadsheets/d/' + SHEET_ID + '/edit?gid=' + CONTRACTS_SHEET_GID + '#gid=' + CONTRACTS_SHEET_GID,
    permits:[],
    permitsHeaders:[],
    permitsSheetName:'',
    permitsSourceUrl: 'https://docs.google.com/spreadsheets/d/' + SHEET_ID + '/edit?gid=' + PERMITS_SHEET_GID + '#gid=' + PERMITS_SHEET_GID,
    monthHeaders:[],
    logoUrl: 'https://drive.google.com/uc?export=view&id=' + LOGO_FILE_ID

  var driveCache  = driveResult.cache;   // { "431": "fileId1", ... }
  out.driveTagsCount = driveResult.count;

  var cleanPhotos = buildCleaningPhotosCache();
  out.cleaning = readCleaning(ss, cleanPhotos);
  out.cleaningPhotosCount = cleanPhotos.count;

  var cp = readCleaningPlan(ss);
  out.cleaningPlan     = (cp && cp.entries)  ? cp.entries  : [];
  out.cleaningProducts = (cp && cp.products) ? cp.products : [];

  // ── Procédures HSE (feuille "procédures") ────────────────
  var pr = readProcedures(ss);
  out.procedures        = (pr && pr.entries) ? pr.entries : [];
  out.proceduresExtras  = (pr && pr.extras)  ? pr.extras  : [];

  // ── Contrats (feuille "Suivi des Contrats et Liens Drive") ─
  var co = readContracts(ss);
  out.contracts          = (co && co.entries)    ? co.entries    : [];
  out.contractsHeaders   = (co && co.headers)    ? co.headers    : [];
  out.contractsSheetName = (co && co.sheetName)  ? co.sheetName  : CONTRACTS_SHEET_NAME;

  var pe = readPermits(ss);
  out.permits          = (pe && pe.entries)   ? pe.entries   : [];
  out.permitsHeaders   = (pe && pe.headers)   ? pe.headers   : [];
  out.permitsSheetName = (pe && pe.sheetName) ? pe.sheetName : PERMITS_SHEET_NAME;
  out.permitsSourceUrl = 'https://docs.google.com/spreadsheets/d/' + SHEET_ID +
                         '/edit?gid=' + PERMITS_SHEET_GID + '#gid=' + PERMITS_SHEET_GID;

  var epi = readEPI(ss);
  out.epi          = (epi && epi.entries)   ? epi.entries   : [];
  out.epiHeaders   = (epi && epi.headers)   ? epi.headers   : [];
  out.epiSheetName = (epi && epi.sheetName) ? epi.sheetName : EPI_SHEET_NAME;
  out.epiSourceUrl = 'https://docs.google.com/spreadsheets/d/' + SHEET_ID +
                     '/edit?gid=' + EPI_SHEET_GID + '#gid=' + EPI_SHEET_GID;
  out.epiSlideUrl  = 'https://docs.google.com/presentation/d/' + EPI_SLIDES_ID +
                     '/edit?slide=' + EPI_SLIDES_SLIDE + '#slide=' + EPI_SLIDES_SLIDE;
  out.epiSlideEmbedUrl = 'https://docs.google.com/presentation/d/' + EPI_SLIDES_ID +
                         '/embed?slide=' + EPI_SLIDES_SLIDE +
                         '&start=false&loop=false&delayms=60000';

  // ── NUISIBLES (4 feuilles) ───────────────────────────────
  // FK · Boîtes à raticide · Pièges mécanique · Boîtes à colles
  var pests = readPests(ss);
  out.pests = pests;   // { fk, rat, mech, glue } — chacun = { entries, headers, sheetName, meta }
  out.pestsSourceUrl = 'https://docs.google.com/spreadsheets/d/' + SHEET_ID +
                       '/edit?gid=1333145847#gid=1333145847';

  // ── KPI ──────────────────────────────────────────────────
  var kpiSheet = findSheet(['kpi']) || sheets[0];
  var kpiData  = kpiSheet.getDataRange().getValues();
  var ytdIdx   = 15;
  for (var c = 0; c < kpiData[0].length; c++)
    if (String(kpiData[0][c]).toLowerCase().indexOf('ytd') > -1) { ytdIdx = c; break; }
  for (var c = 3; c < ytdIdx; c++) out.monthHeaders.push(String(kpiData[0][c]||'').trim());
  for (var i = 1; i < kpiData.length; i++) {
    var r = kpiData[i];
    var ind = String(r[0]||'').trim();
    if (!ind) continue;
    if (ind.toLowerCase() === 'date') break;
    var months = [];
    for (var c = 3; c < ytdIdx; c++) months.push(toNum(r[c]));
    out.kpis[ind] = {
      unite:String(r[1]||'').trim(), objectif:String(r[2]||'').trim(), months:months,
      jan:toNum(r[3]),fev:toNum(r[4]),mar:toNum(r[5]),avr:toNum(r[6]),
      mai:toNum(r[7]),juin:toNum(r[8]),juil:toNum(r[9]),aout:toNum(r[10]),
      sep:toNum(r[11]),oct:toNum(r[12]),nov:toNum(r[13]),dec:toNum(r[14]),
      ytd:toNum(r[ytdIdx])
    };
  }

  // ── INCIDENTS ─────────────────────────────────────────────
  var incSheet = findSheet(['incident']) || (sheets.length>1?sheets[1]:null);
  if (incSheet) {
    var incData = incSheet.getDataRange().getValues();

    // Détection de la colonne « lien vers fichier d'enquête » via l'entête.
    // On reste tolérant : tout entête qui contient « enquet » est candidat,
    // de préférence combiné à « lien », « fichier », « rapport », « url »,
    // « drive » ou « doc ».
    var incHdr  = incData[0] || [];
    var COL_ENQ = -1;
    for (var ch = 0; ch < incHdr.length; ch++) {
      var hh = normalize(String(incHdr[ch]||'').replace(/[\n\r#]/g,'').trim());
      if (!hh) continue;
      if (hh.indexOf('enquet') > -1 &&
          (hh.indexOf('lien') > -1    || hh.indexOf('fichier') > -1 ||
           hh.indexOf('rapport') > -1 || hh.indexOf('url') > -1 ||
           hh.indexOf('drive') > -1   || hh.indexOf('doc') > -1)) {
        COL_ENQ = ch; break;
      }
    }
    if (COL_ENQ < 0) {
      for (var ch2 = 0; ch2 < incHdr.length; ch2++) {
        var hh2 = normalize(String(incHdr[ch2]||'').replace(/[\n\r#]/g,'').trim());
        if (hh2.indexOf('enquet') > -1) { COL_ENQ = ch2; break; }
      }
    }

    // Helper : récupère une URL http(s) depuis une cellule, qu'elle
    // contienne du texte brut, un lien Drive collé, ou une formule
    // =HYPERLINK("url","libellé").
    function extractIncUrl_(cellVal, rowIdx, colIdx) {
      if (!isNull(cellVal)) {
        var s = String(cellVal).trim();
        var m = s.match(/https?:\/\/[^\s"'<>)]+/i);
        if (m) return m[0];
      }
      try {
        if (colIdx >= 0) {
          var f = incSheet.getRange(rowIdx + 1, colIdx + 1).getFormula();
          if (f) {
            var mf = f.match(/https?:\/\/[^\s"'<>)]+/i);
            if (mf) return mf[0];
          }
        }
      } catch(e) { /* ignore */ }
      return '';
    }

    for (var i = 1; i < incData.length; i++) {
      var r  = incData[i];
      var id = String(r[0]||'').trim();
      if (!id || (id.toLowerCase().indexOf('id')>-1 && id.length<6)) continue;
      if (isNull(r[0]) && isNull(r[7]) && isNull(r[6])) continue;
      var pd  = parseDate(r[1]);
      var iso = (pd&&!isNaN(pd.getTime())) ? Utilities.formatDate(pd, 'GMT+1', 'yyyy-MM-dd') : '';
      var enqueteUrl = (COL_ENQ >= 0) ? extractIncUrl_(r[COL_ENQ], i, COL_ENQ) : '';
      out.incidents.push({
        id:id, date:fmtDate(r[1]), isoDate:iso,
        heure:safeText(r[2]),
        lieu:isNull(r[3])?'':String(r[3]).trim(),
        type:isNull(r[4])?'':String(r[4]).trim(),
        gravite:toNumZ(r[5])||1,
        description:isNull(r[6])?'':String(r[6]).trim(),
        victime:isNull(r[7])?'':String(r[7]).trim(),
        causes:isNull(r[8])?'':String(r[8]).trim(),
        responsable:isNull(r[9])?'':String(r[9]).trim(),
        delai:safeText(r[10]),
        statut:isNull(r[11])?'':String(r[11]).trim(),
        analyse5m:isNull(r[13])?'':String(r[13]).trim(),
        actions:isNull(r[14])?'':String(r[14]).trim(),
        enqueteUrl:enqueteUrl
      });
    }
    out.incidents.sort(function(a,b){ return !a.isoDate?1:!b.isoDate?-1:b.isoDate.localeCompare(a.isoDate); });
  }

  // ── TAGs ─────────────────────────────────────────────────
  // Structure confirmée (0-based getValues) :
  // r[0]=N°Cas  r[1]=DateCréation  r[2]=DateCible  r[3]=TypeDanger
  // r[4]=Gravité  r[5]=Emplacement  r[6]=Zone  r[7]=Description
  // r[8]=RisquePrincipal  r[9]=ActionUrgente  r[10]=Propositions
  // r[11]=RéférencePhoto  r[12]=Statut  r[13]=Responsable
  var tagSheet = findSheet(['tag','danger','situation']) || (sheets.length>2?sheets[2]:null);
  if (tagSheet) {
    var tagData = tagSheet.getDataRange().getValues();

    // Détecter les colonnes depuis l'entête (ligne 0)
    var hdr = tagData[0];
    var COL = { id:0, dateCreate:1, dateCible:2, type:3, gravite:4,
                emplacement:5, zone:6, description:7, risque:8,
                action:9, propositions:10, photoRef:11, photoRefApres:-1,
                statut:12, responsable:13 };
    for (var c = 0; c < hdr.length; c++) {
      var h = normalize(String(hdr[c]||'').replace(/[\n\r#]/g,'').trim());
      // ── PHOTO APRÈS — colonne dédiée (à tester AVANT photoRef pour ne pas la voler)
      if ((h.indexOf('photo') > -1 || h.indexOf('reference') > -1) &&
          (h.indexOf('apres') > -1 || h.indexOf('after') > -1)) {
        COL.photoRefApres = c;
        continue;
      }
      if      (h.match(/^n.?\s*(cas|tag|$)/))            COL.id          = c;
      else if (h.indexOf('date c') === 0 && h.indexOf('cible') < 0 && h.indexOf('cr') > -1) COL.dateCreate = c;
      else if (h.indexOf('cible') > -1 || h.indexOf('echeance') > -1) COL.dateCible   = c;
      else if (h.indexOf('type') > -1 && h.indexOf('danger') > -1)     COL.type        = c;
      else if (h.indexOf('gravit') > -1)                                 COL.gravite     = c;
      else if (h.indexOf('emplace') > -1)                                COL.emplacement = c;
      else if (h === 'zone')                                              COL.zone        = c;
      else if (h.indexOf('description') > -1)                            COL.description = c;
      else if (h.indexOf('risque') > -1)                                 COL.risque      = c;
      else if (h.indexOf('action') > -1 && h.indexOf('urgente') > -1)   COL.action      = c;
      else if (h.indexOf('proposition') > -1)                            COL.propositions= c;
      else if (h.indexOf('reference') > -1 && h.indexOf('photo') > -1)  COL.photoRef    = c;
      else if (h.indexOf('statut') > -1)                                 COL.statut      = c;
      else if (h.indexOf('respens') > -1 || h.indexOf('respons') > -1)  COL.responsable = c;
    }
    Logger.log('TAGs COL: ' + JSON.stringify(COL));

    for (var i = 1; i < tagData.length; i++) {
      var r = tagData[i];
      if (isNull(r[COL.id])) continue;

      // ── Référence photo AVANT (colonne principale) ────────
      var refRaw   = isNull(r[COL.photoRef]) ? '' : String(r[COL.photoRef]).trim();
      var refClean = refRaw.replace(/^tags?\//i, '').trim();

      // ── Référence photo APRÈS (colonne dédiée si présente) ─
      var refApresRaw   = (COL.photoRefApres >= 0 && !isNull(r[COL.photoRefApres]))
                            ? String(r[COL.photoRefApres]).trim() : '';
      var refApresClean = refApresRaw.replace(/^tags?\//i, '').trim();

      // ── Résolution fileId depuis le cache Drive ───────────
      var driveFileIdAvant = '';
      var driveFileIdApres = '';

      // AVANT : extraire le numéro de la réf principale et chercher le fichier
      // Priorité : fichier sans mot-clé "après" → puis n'importe quel fichier avec ce numéro
      var numAvant = refClean.match(/(\d+)/);
      if (numAvant && driveCache[numAvant[1]]) {
        var entryAv = driveCache[numAvant[1]];
        driveFileIdAvant = entryAv.avant || entryAv.any || '';
      }

      // APRÈS : 2 cas selon que la feuille a une colonne dédiée
      if (refApresClean) {
        // Cas 1 — Colonne « Réf. photo après » fournie : on prend le fichier
        // qui porte ce numéro, sans contrainte de mot-clé (l'utilisateur a explicité)
        var numApres = refApresClean.match(/(\d+)/);
        if (numApres && driveCache[numApres[1]]) {
          var entryAp = driveCache[numApres[1]];
          driveFileIdApres = entryAp.apres || entryAp.any || '';
        }
      } else if (numAvant && driveCache[numAvant[1]]) {
        // Cas 2 — Une seule colonne photo : la photo « après » doit avoir
        // un mot-clé apres/après/after dans son nom, sinon il n'y en a pas.
        driveFileIdApres = driveCache[numAvant[1]].apres || '';
      }

      var tag = {
        id:          String(r[COL.id]).trim(),
        date:        fmtDate(r[COL.dateCreate]),
        isoDate:     getIso(r[COL.dateCreate]),
        dateCible:   fmtDate(r[COL.dateCible]),
        type:        isNull(r[COL.type])         ? '' : String(r[COL.type]).trim(),
        gravite:     isNull(r[COL.gravite])      ? '' : String(r[COL.gravite]).trim(),
        emplacement: isNull(r[COL.emplacement])  ? '' : String(r[COL.emplacement]).trim(),
        zone:        isNull(r[COL.zone])         ? '' : String(r[COL.zone]).trim(),

    function sortByIdDesc(a){
      a.sort(function(x,y){
        var nx = parseInt(String(x.id).match(/\d+/), 10);
        var ny = parseInt(String(y.id).match(/\d+/), 10);
        if (isNaN(nx) && isNaN(ny)) return String(y.id).localeCompare(String(x.id));
        if (isNaN(nx)) return 1;
        if (isNaN(ny)) return -1;
        return ny - nx;
      });

        circonstance:isNull(r[9])?'':String(r[9]).trim()
      });
    }
    out.accidents.sort(function(a,b){ return !a.isoDate?1:!b.isoDate?-1:b.isoDate.localeCompare(a.isoDate); });
  }

  out.metrics = calcMetrics(out);
  return out;
}

function callGemini(prompt) {
  var models=['gemini-2.0-flash','gemini-1.5-flash'];
  for (var i=0;i<models.length;i++) {
    try {
      var url='https://generativelanguage.googleapis.com/v1/models/'+models[i]+':generateContent?key='+GEMINI_KEY;
      var resp=UrlFetchApp.fetch(url,{method:'post',contentType:'application/json',
        payload:JSON.stringify({contents:[{parts:[{text:prompt}]}],generationConfig:{temperature:0.3,maxOutputTokens:1024}}),muteHttpExceptions:true});
      var json=JSON.parse(resp.getContentText());
      if (json.candidates&&json.candidates[0]&&json.candidates[0].content) return json.candidates[0].content.parts[0].text;
      if (json.error&&json.error.code===404) continue;
      if (json.error) return 'Erreur API : '+json.error.message;
    } catch(ex) { return 'Erreur : '+ex.message; }
  }
  return 'Aucun modèle Gemini disponible.';
}

function initRBACSheet() {

    Logger.log('RBAC sheet created with default admin');
  }
  return sh;
}

function hashPassword(password) {
  var raw = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, password);
  var hash = '';
  for (var i = 0; i < raw.length; i++) {
    var byte = raw[i];
    if (byte < 0) byte += 256;
    var hex = byte.toString(16);
    if (hex.length === 1) hex = '0' + hex;
    hash += hex;
  }
  return hash;
}

function rbacLogin(username, password) {
  try {
    var sh = initRBACSheet();
    var data = sh.getDataRange().getValues();
    var passHash = hashPassword(password);
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      if (String(row[0]).trim().toLowerCase() === username.trim().toLowerCase() &&
          String(row[1]).trim() === passHash &&
          String(row[6]).trim().toUpperCase() === 'TRUE') {
        var sections = String(row[4]).trim();
        var allowedSections = (sections === 'ALL') ? ['ALL'] :
          sections.split(',').map(function(s){return s.trim();}).filter(function(s){return s!=='';});
        return {
          success: true,
          user: {
            username: String(row[0]).trim(),
            role: String(row[2]).trim().toLowerCase(),
            displayName: String(row[3]).trim(),
            allowedSections: allowedSections
          }
        };
      }
    }
    return { success: false, error: "Nom d'utilisateur ou mot de passe incorrect." };
  } catch(e) {
    Logger.log('rbacLogin error: ' + e.message);
    return { success: false, error: 'Erreur serveur.' };
  }
}

function rbacCreateUser(adminUsername, newUser) {
  try {
    var sh = initRBACSheet();
    var data = sh.getDataRange().getValues();
    var isAdmin = false;
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim().toLowerCase() === adminUsername.trim().toLowerCase() &&
          String(data[i][2]).trim().toLowerCase() === 'admin') { isAdmin = true; break; }
    }
    if (!isAdmin) return { success: false, error: 'Permission refusée.' };
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim().toLowerCase() === newUser.username.trim().toLowerCase())
        return { success: false, error: "Ce nom d'utilisateur existe déjà." };
    }
    var role = newUser.role.toLowerCase();
    if (role !== 'editor' && role !== 'viewer') return { success: false, error: 'Rôle invalide.' };
    var sections = 'ALL';
    if (role === 'editor' && newUser.allowedSections && newUser.allowedSections.length > 0)
      sections = newUser.allowedSections.join(',');
    sh.appendRow([
      newUser.username.trim(), hashPassword(newUser.password), role,
      newUser.displayName || newUser.username, sections, new Date().toISOString(), 'TRUE'
    ]);
    return { success: true, message: 'Utilisateur "' + newUser.username + '" créé.' };
  } catch(e) { return { success: false, error: e.message }; }
}

function rbacUpdateUser(adminUsername, targetUsername, updates) {
  try {
    var sh = initRBACSheet();
    var data = sh.getDataRange().getValues();
    var isAdmin = false;
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim().toLowerCase() === adminUsername.trim().toLowerCase() &&
          String(data[i][2]).trim().toLowerCase() === 'admin') { isAdmin = true; break; }
    }
    if (!isAdmin) return { success: false, error: 'Permission refusée.' };
    var targetRow = -1;
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim().toLowerCase() === targetUsername.trim().toLowerCase())
        { targetRow = i + 1; break; }
    }
    if (targetRow < 0) return { success: false, error: 'Utilisateur non trouvé.' };
    if (updates.displayName) sh.getRange(targetRow, 4).setValue(updates.displayName);
    if (updates.role) {
      var r = updates.role.toLowerCase();
      if (r!=='admin'&&r!=='editor'&&r!=='viewer') return {success:false,error:'Rôle invalide.'};
      sh.getRange(targetRow, 3).setValue(r);
    }
    if (updates.allowedSections !== undefined) {
      var sec = (updates.allowedSections && updates.allowedSections.length > 0)
        ? updates.allowedSections.join(',') : 'ALL';
      sh.getRange(targetRow, 5).setValue(sec);
    }
    if (updates.password) sh.getRange(targetRow, 2).setValue(hashPassword(updates.password));
    if (updates.active !== undefined) sh.getRange(targetRow, 7).setValue(updates.active ? 'TRUE' : 'FALSE');

// ── Helper : vérifier permission d'écriture ──
function checkWritePermission(username, sectionId) {
  var sh = initRBACSheet();
  var data = sh.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim().toLowerCase() === username.trim().toLowerCase()) {
      var role = String(data[i][2]).trim().toLowerCase();
      if (role === 'viewer') return { ok: false, error: 'Accès refusé : compte en lecture seule.' };
      if (role === 'admin') return { ok: true, role: 'admin' };
      if (role === 'editor') {
        var sections = String(data[i][4]).trim();
        if (sections === 'ALL') return { ok: true, role: 'editor' };
        var arr = sections.split(',').map(function(s){return s.trim();});
        if (arr.indexOf(sectionId) > -1) return { ok: true, role: 'editor' };
        return { ok: false, error: 'Accès refusé : cette section ne vous est pas attribuée.' };
      }
    }
  }
  return { ok: false, error: 'Utilisateur non trouvé.' };
}

// ── Helper : vérifier permission de suppression (admin only) ──
function checkDeletePermission(username) {
  var sh = initRBACSheet();
  var data = sh.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim().toLowerCase() === username.trim().toLowerCase()) {
      var role = String(data[i][2]).trim().toLowerCase();
      if (role === 'admin') return { ok: true };
      return { ok: false, error: 'Seul un administrateur peut supprimer des données.' };
    }
  }
  return { ok: false, error: 'Utilisateur non trouvé.' };
}

// ── Helper : trouver une feuille par mots-clés ──
function findSheetByKeywords(ss, kwds) {
  var sheets = ss.getSheets();
  for (var k = 0; k < kwds.length; k++) {
    for (var i = 0; i < sheets.length; i++) {
      if (normalize(sheets[i].getName()).indexOf(kwds[k]) > -1) return sheets[i];
    }
  }
  return null;
}

// ── Helper : upload fichier (base64) vers Drive et retourner l'ID ──
function uploadFileToDrive(folderId, fileName, base64Data, mimeType) {

    var blob = Utilities.newBlob(decoded, mimeType || 'image/jpeg', fileName);

    return { success: true, fileId: file.getId(), url: file.getUrl() };
  } catch(e) {
    Logger.log('uploadFileToDrive error: ' + e.message);
    return { success: false, error: e.message };
  }
}

      '',
      incData.analyse5m || '',
      incData.actions || ''

    sh.appendRow(newRow);

// ── NETTOYAGE EDIT ───────────────────────────────────────────
// ════════════════════════════════════════════════════════════
function crudEditCleaning(username, cleanId, cleanData) {
  try {
    var perm = checkWritePermission(username, 'nettoyage');
    if (!perm.ok) return { success: false, error: perm.error };
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var sheets = ss.getSheets();
    var sh = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === CLEAN_SHEET_GID) { sh = sheets[i]; break; }
    }
    if (!sh) {
      for (var i = 0; i < sheets.length; i++) {
        if (normalize(sheets[i].getName()).indexOf('nettoyage') > -1) { sh = sheets[i]; break; }
      }
    }
    if (!sh) return { success: false, error: 'Feuille Nettoyage non trouvée.' };
    var data = sh.getDataRange().getValues();
    var targetRow = -1;
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim() === String(cleanId).trim()) { targetRow = i + 1; break; }
    }
    if (targetRow < 0) return { success: false, error: 'Entrée non trouvée.' };
    var hdr = data[0];
    var cols = { zone:-1, emplacement:-1, description:-1, action:-1, responsable:-1, statut:-1 };
    for (var c = 0; c < hdr.length; c++) {
      var h = normalize(String(hdr[c]||''));
      if (h.indexOf('zone')>-1 && cols.zone<0) cols.zone=c;
      else if ((h.indexOf('emplace')>-1||h.indexOf('lieu')>-1) && cols.emplacement<0) cols.emplacement=c;
      else if ((h.indexOf('description')>-1||h.indexOf('observation')>-1||h.indexOf('constat')>-1) && cols.description<0) cols.description=c;
      else if ((h.indexOf('action')>-1||h.indexOf('mesure')>-1) && cols.action<0) cols.action=c;
      else if ((h.indexOf('respons')>-1||h.indexOf('charg')>-1) && cols.responsable<0) cols.responsable=c;
      else if ((h.indexOf('statut')>-1||h.indexOf('etat')>-1) && cols.statut<0) cols.statut=c;
    }
    if (cleanData.zone!==undefined && cols.zone>=0) sh.getRange(targetRow,cols.zone+1).setValue(cleanData.zone);
    if (cleanData.emplacement!==undefined && cols.emplacement>=0) sh.getRange(targetRow,cols.emplacement+1).setValue(cleanData.emplacement);
    if (cleanData.description!==undefined && cols.description>=0) sh.getRange(targetRow,cols.description+1).setValue(cleanData.description);
    if (cleanData.action!==undefined && cols.action>=0) sh.getRange(targetRow,cols.action+1).setValue(cleanData.action);
    if (cleanData.responsable!==undefined && cols.responsable>=0) sh.getRange(targetRow,cols.responsable+1).setValue(cleanData.responsable);
    if (cleanData.statut!==undefined && cols.statut>=0) sh.getRange(targetRow,cols.statut+1).setValue(cleanData.statut);
    Logger.log('Cleaning '+cleanId+' updated by '+username);
    return { success: true, message: 'Entrée nettoyage mise à jour.' };
  } catch(e) { return { success: false, error: e.message }; }
}
