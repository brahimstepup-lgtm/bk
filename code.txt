function doGet(e) {
  var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : '';
  if (action === 'data') return ContentService
    .createTextOutput(JSON.stringify(getData()))

// ── PROXY IMAGE BASE64 ───────────────────────────────────────
// Appelé depuis le client via google.script.run
// Retourne "data:image/jpeg;base64,..." ou ''
// ════════════════════════════════════════════════════════════
function getTagPhotoForClient(fileId) {
  if (!fileId || fileId === '') return '';
  try {
    var file = DriveApp.getFileById(fileId);
    var blob = file.getBlob();
    var b64  = Utilities.base64Encode(blob.getBytes());
    var mime = blob.getContentType() || 'image/jpeg';
    return 'data:' + mime + ';base64,' + b64;
  } catch(e) {
    Logger.log('getTagPhotoForClient error: ' + e.message);
    return '';
  }
}

// ── RÉSOLUTION PHOTO : référence → fileId Drive ──────────────
// Stratégie : extraire le numéro de la référence (ex: "431")
// puis chercher dans le dossier Drive avec title contains "431"
// Évite totalement le préfixe "tags/" et les accents
// ════════════════════════════════════════════════════════════
function resolveTagFileId(refRaw) {
  if (!refRaw || refRaw.trim() === '') return '';
  try {
    // 1. Nettoyer le préfixe tags/ ou tag/
    var clean = String(refRaw).replace(/^tags?\//i, '').trim();

    // 2. Extraire le numéro principal (premiers chiffres consécutifs)
    var numMatch = clean.match(/^(\d+)/);
    if (!numMatch) {
      // Pas de numéro → essai recherche par nom exact nettoyé
      numMatch = clean.match(/(\d{3,})/); // au moins 3 chiffres quelque part
    }

    if (numMatch) {
      var num = numMatch[1];
      var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
      // Recherche Drive par numéro dans le titre — fonctionne avec accents
      var files = folder.searchFiles('title contains "' + num + '"');
      if (files.hasNext()) {
        return files.next().getId();
      }
    }
  } catch(e) {
    Logger.log('resolveTagFileId error: ' + e.message + ' | ref: ' + refRaw);
  }
  return '';
}

// ── SCAN INITIAL : pré-charger les IDs de tous les TAGs ──────
// Appelé une seule fois dans getData() pour éviter N appels Drive
// Construit un cache { "431": "fileId", "432": "fileId", ... }
// ════════════════════════════════════════════════════════════
function buildDriveTagsCache() {
  var cache  = {};  // num → fileId
  var count  = 0;
  try {
    var folder = DriveApp.getFolderById(TAGS_FOLDER_ID);
    var files  = folder.getFiles();
    while (files.hasNext()) {
      var file = files.next();
      var name = file.getName();
      var id   = file.getId();
      count++;
      // Extraire le numéro en début de nom : "431.Référence Photo..." → "431"
      var m = name.match(/^(\d+)/);
      if (m) {
        var num = m[1];
        if (!cache[num]) cache[num] = id; // Premier fichier pour ce numéro
      }
    }
    Logger.log('Drive cache: ' + count + ' fichiers, ' + Object.keys(cache).length + ' numéros');
  } catch(e) {
    Logger.log('buildDriveTagsCache error: ' + e.message);
  }
  return { cache: cache, count: count };
}

// ── Utilitaires ───────────────────────────────────────────────
function isNull(v) {
  if (v === null || v === undefined) return true;
  var s = String(v).trim(); return s === '' || s === '/' || s === '-';
}
function toNum(v) {
  if (isNull(v)) return null;
  var n = parseFloat(String(v).replace(/\s/g,'').replace(/,/g,''));
  return isNaN(n) ? null : n;
}
function toNumZ(v) { var n = toNum(v); return n === null ? 0 : n; }

function parseDate(v) {
  if (isNull(v)) return null;
  if (v instanceof Date && !isNaN(v.getTime())) return v;
  var s = String(v).trim();
  var m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})/);
  if (m) {
    var d = parseInt(m[1]), mo = parseInt(m[2]), yr = parseInt(m[3]);
    if (d > 12)  return new Date(yr, mo-1, d);
    if (mo > 12) return new Date(yr, d-1, mo);
    return new Date(yr, mo-1, d);
  }
  try { var r = new Date(v); return isNaN(r.getTime()) ? null : r; } catch(e2) { return null; }
}
function fmtDate(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return String(v).trim();
  try { return Utilities.formatDate(d, 'GMT+1', 'dd/MM/yyyy'); } catch(e) { return String(v).trim(); }
}
function getIso(v) {
  if (isNull(v)) return '';
  var d = (v instanceof Date) ? v : parseDate(v);
  if (!d || isNaN(d.getTime())) return '';
  try { return d.toISOString().split('T')[0]; } catch(e) { return ''; }
}
function isFumigation(a) { return a && a.toLowerCase().indexOf('fumig') > -1; }
function normalize(s) {
  if (!s) return '';
  return s.toLowerCase()
    .replace(/[àáâãäå]/g,'a').replace(/[èéêë]/g,'e')
    .replace(/[ìíîï]/g,'i').replace(/[òóôõö]/g,'o')
    .replace(/[ùúûü]/g,'u').replace(/[ç]/g,'c').replace(/[ñ]/g,'n')
    .replace(/\s+/g,' ').trim();
}

// ── Planning ─────────────────────────────────────────────────
function readPlanning(ss) {
  var planning = [];
  try {
    var sheets = ss.getSheets();
    var planSheet = null;
    for (var i = 0; i < sheets.length; i++) {
      if (sheets[i].getSheetId() === PLANNING_GID) { planSheet = sheets[i]; break; }
    }
    if (!planSheet) {
      for (var i = 0; i < sheets.length; i++) {
        var n = sheets[i].getName().toLowerCase();
        if (n.indexOf('planning') > -1 || n.indexOf('activit') > -1 || n.indexOf('suivi') > -1) {
          planSheet = sheets[i]; break;
        }
      }
    }
    if (!planSheet) { Logger.log('Planning sheet NOT found'); return planning; }
    Logger.log('Planning sheet found: ' + planSheet.getName());
    var data = planSheet.getDataRange().getValues();
    var startRow = 1;
    for (var i = 0; i < Math.min(data.length, 10); i++) {
      var h0 = String(data[i][0]||'').toLowerCase();
      if (h0.indexOf('date') > -1) { startRow = i + 1; break; }
    }
    for (var i = startRow; i < data.length; i++) {
      var r = data[i];
      if (isNull(r[0])) continue;
      var pd = parseDate(r[0]);
      if (!pd || isNaN(pd.getTime())) continue;
      var act = isNull(r[1]) ? '' : String(r[1]).trim();
      if (!act) continue;
      planning.push({
        date: fmtDate(r[0]),
        isoDate: pd.toISOString().split('T')[0],
        activite: act,
        details: isNull(r[2]) ? '' : String(r[2]).trim(),
        fumigation: isFumigation(act)
      });
    }
    Logger.log('Planning: ' + planning.length + ' entrées');
  } catch(e) { Logger.log('Planning error: ' + e.message); }
  return planning;
}

function callGemini(prompt) {
  var models=['gemini-2.0-flash','gemini-1.5-flash'];
  for (var i=0;i<models.length;i++) {
    try {
      var url='https://generativelanguage.googleapis.com/v1/models/'+models[i]+':generateContent?key='+GEMINI_KEY;
      var resp=UrlFetchApp.fetch(url,{method:'post',contentType:'application/json',
        payload:JSON.stringify({contents:[{parts:[{text:prompt}]}],generationConfig:{temperature:0.3,maxOutputTokens:1024}}),muteHttpExceptions:true});
      var json=JSON.parse(resp.getContentText());
      if (json.candidates&&json.candidates[0]&&json.candidates[0].content) return json.candidates[0].content.parts[0].text;
      if (json.error&&json.error.code===404) continue;
      if (json.error) return 'Erreur API : '+json.error.message;
    } catch(ex) { return 'Erreur : '+ex.message; }
  }
  return 'Aucun modèle Gemini disponible.';
}

// PHOTO PROXY — retourne base64 de l'image Drive
function getTagPhotoForClient(fileId) { return getTagPhotoForClient_(fileId); }
function getTagPhotoForClient_(fileId) {
  if (!fileId) return '';
  try {
    var blob = DriveApp.getFileById(fileId).getBlob();
    return 'data:' + (blob.getContentType()||'image/jpeg') + ';base64,' + Utilities.base64Encode(blob.getBytes());
  } catch(e) { Logger.log('photo error: '+e.message); return ''; }
}
